101  /**
102  * Read from the device into a buffer
103  *
104  * @param buf buffer to read into
105  * @param count the number of bytes to read
106  *
107  * @return the number of bytes that were actually read, or -1 if fail. See ::read
108  */
109  ssize_t read(void* buf, size_t count);
110 
111  /**
112  * @brief Read from the device into a structure
113  *
114  * @param data the structure to read in to
115  *
116  * @return the number of bytes that were actually read, or -1 if fail. See ::read
117  *
118  * @note Implementation in header file to stop the compiler from optimising it away
119  */
120  template <typename T>
121  ssize_t read(T& data) requires std::is_trivially_copyable_v<T> {
122  return read(static_cast<void*>(&data), sizeof(T));
123  }
124 
125  /**
126  * Write bytes to the uart
127  *
128  * @param buf the buffer to write bytes from
129  * @param count the number of bytes to write
130  *
131  * @return the number of bytes that were written
132  */
133  ssize_t write(const void* buf, size_t count);
134 
135  /**
136  * @brief Write bytes to the uart
137  *
138  * @param data the data to write
139  *
140  * @return the number of bytes that were written
141  *
142  * @note Implementation in header file to stop the compiler from optimising it away
143  */
144  template <typename T>
145  ssize_t write(const T& data) requires std::is_trivially_copyable_v<T> {
146  return write(static_cast<const void*>(&data), sizeof(T));
147  }
148 
149  /**
150  * Read a single character from the uart device, returning -1 if there is no data
151  *
152  * @return the character read from the device, or -1 if there is no data
153  */
154  [[nodiscard]] int get();
155 
156  /**
157  * Close the open file descriptor then reset fd = -1
158  */
159  void close();
160

165  */
166  [[nodiscard]] int available() const;
