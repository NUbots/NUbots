37 // Resampling techniques!!!
38 // http://users.isy.liu.se/rt/schon/Publications/HolSG2006.pdf
39 namespace utility::math::stats::resample {
40 
41  /**
42  * @brief The core resampling algorithm for multinomial, stratified, and systematic resampling schemes.
43  *
44  * @details The algorithm takes the following steps
45  * 1. Normalise weights so they sum to 1
46  * 2. Calculate cumulative sum of the normalised weights
47  * 3. For each of the count particles
48  * a. Get a new resampling weight from the generator
49  * b. Find the first index in the cumulative sum where the generated resampling weight is >= cumulative sum
50  *
51  * The index in step 3b is the index of the particle that should be sampled
52  *
53  * @tparam Generator Type of the generator to use for resampling weights
54  * @tparam Iterator The iterator type for the weights array
55  *
56  * @param count The number of particles being resampled
57  * @param begin Iterator to the first weight to use for resampling
58  * @param end Iterator to the last weight to use for resampling
59  */
60  template <template <typename> class Generator, typename Iterator>
61  [[nodiscard]] std::vector<int> resample(const int& count, Iterator begin, Iterator end) {
62  using Scalar = std::remove_reference_t<decltype(*begin)>;
63 
64  // Get number of weights
65  const auto n_weights = std::distance(begin, end);
66 
67  // Create our distribution generator
68  Generator<Scalar> fn(count);
69 
70  // Normalise the weights
71  std::vector<Scalar> normalised(n_weights, Scalar(0));
72  const Scalar sum = std::accumulate(begin, end, Scalar(0));
73  std::transform(begin, end, normalised.begin(), [&](const Scalar& w) { return w / sum; });
74 
75  // Make a cumulative sum of the weights
76  std::vector<Scalar> cumsum(n_weights, Scalar(0));
77  std::partial_sum(normalised.begin(), normalised.end(), cumsum.begin());
78 
79  // Randomly sample from the distribution using our random number generator
80  std::vector<int> idx(count, 0);
81  for (int i = 0; i < count; ++i) {
82  idx[i] = std::distance(cumsum.begin(), std::lower_bound(cumsum.begin(), cumsum.end(), fn(i)));
83  }
84 
85  return idx;
86  }
