54  struct StartingSide {
55  enum Value { UNKNOWN = 0, LEFT = 1, RIGHT = 2, EITHER = 3, CUSTOM = 4 };
56  Value value = Value::UNKNOWN;
57 
58  // Constructors
59  StartingSide() = default;
60  StartingSide(int const& v) : value(static_cast<Value>(v)) {}
61  StartingSide(Value const& v) : value(v) {}
62  StartingSide(std::string const& str) {
63  // clang-format off
64  if (str == "LEFT") { value = Value::LEFT; }
65  else if (str == "RIGHT") { value = Value::RIGHT; }
66  else if (str == "EITHER") { value = Value::EITHER; }
67  else if (str == "CUSTOM") { value = Value::CUSTOM; }
68  else {
69  value = Value::UNKNOWN;
70  throw std::runtime_error("String " + str + " did not match any enum for StartingSide");
71  }
72  // clang-format on
73  }
74 
75  // Conversions
76  [[nodiscard]] operator Value() const {
77  return value;
78  }
79  [[nodiscard]] operator std::string() const {
80  switch (value) {
81  case Value::LEFT: return "LEFT";
82  case Value::RIGHT: return "RIGHT";
83  case Value::EITHER: return "EITHER";
84  case Value::CUSTOM: return "CUSTOM";
85  default: throw std::runtime_error("enum Method's value is corrupt, unknown value stored");
86  }
87  }
88  };
