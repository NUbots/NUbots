74  struct TrackedRobot {
75  /// @brief Time of the last time update
76  NUClear::clock::time_point last_time_update = NUClear::clock::now();
77  /// @brief Unscented Kalman Filter for this robot
78  utility::math::filter::UKF<double, RobotModel> ukf{};
79  /// @brief Whether the robot was and should have been seen in the last vision update
80  bool seen = true;
81  /// @brief The number of times the robot has been undetected in a row
82  long missed_count = 0;
83  /// @brief A unique identifier for the robot
84  unsigned long id;
85 
86  /// @brief The unique identifier of the robot if it is a teammate
87  /// If it is not a teammate, this will be 0
88  bool teammate = false;
89  message::purpose::Purpose purpose = message::purpose::Purpose();
90 
91  /// @brief Constructor that sets the state for the UKF
92  /// @param initial_rRWw The initial position of the robot in world coordinates
93  /// @param cfg_ukf The UKF configuration to use for the robot
94  /// @param next_id The unique id to assign to this new robot
95  /// @param p An optional purpose message that can be used to associate the robot with a teammate
96  TrackedRobot(const Eigen::Vector3d& initial_rRWw,
97  const Config::UKF& cfg_ukf,
98  const unsigned long next_id,
99  const std::unique_ptr<message::purpose::Purpose>& p = nullptr)
100  : id(next_id) {
101  NUClear::log<NUClear::LogLevel::DEBUG>("Making robot with id: ", id);
102  RobotModel<double>::StateVec initial_state;
103  initial_state.rRWw = initial_rRWw.head<2>();
104 
105  RobotModel<double>::StateVec initial_covariance;
106  initial_covariance.rRWw = cfg_ukf.initial_covariance.position;
107  initial_covariance.vRw = cfg_ukf.initial_covariance.velocity;
108  ukf.set_state(initial_state, initial_covariance.asDiagonal());
109 
110  RobotModel<double>::StateVec process_noise;
111  process_noise.rRWw = cfg_ukf.noise.process.position;
112  process_noise.vRw = cfg_ukf.noise.process.velocity;
113  ukf.model.process_noise = process_noise;
114 
115  // If a purpose is provided, set the teammate flag and purpose
116  if (p) {
117  teammate = true;
118  purpose = *p;
119  }
120  }
121 
122  // Get the robot's position in world space
123  Eigen::Vector2d get_rRWw() const {
124  return RobotModel<double>::StateVec(ukf.get_state()).rRWw;
125  };
126  };
