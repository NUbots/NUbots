36  /**
37  * @brief Allows for python like enumeration
38  * @warning Will cause problems if iterable is destroyed
39  * @tparam T The type of the iterable to iterate over, should be automatically deduced
40  */
41  template <std::ranges::range T>
42  struct enumerate {
43  private:
44  T& iterable;
45 
46  public:
47  /**
48  * @brief A wrapper for the actual iterator
49  * @warning Is invalidated like an iterator for T
50  */
51  struct iterator {
52  using iterator_type = std::ranges::iterator_t<T>;
53  int i{0};
54  iterator_type iter{};
55 
56  /// @brief Compares the actual iterators
57  bool operator!=(const iterator& other) const {
58  return iter != other.iter;
59  }
60 
61  /// @brief Increments the actual iterator and the counter
62  void operator++() {
63  i++;
64  iter++;
65  }
66 
67  /**
68  * @brief Gets the count and dereferences the actual iterator
69  * @return first is count, second is dereferenced value
70  */
71  auto operator*() {
72  return std::pair(i, *iter);
73  }
74 
75  iterator(const int& i, iterator_type&& iter) : i(i), iter(std::forward<iterator_type>(iter)) {}
76  };
77 
78 
79  /// @brief An iterator to the start of the iterable
80  constexpr iterator begin() noexcept {
81  return iterator(0, std::ranges::begin(iterable));
82  }
83 
84  /// @brief An iterator to the end of the iterable
85  constexpr iterator end() noexcept {
86  return iterator(std::numeric_limits<int>::max(), std::ranges::end(iterable));
87  }
88 
89  /**
90  * @brief Wraps an iterable to behave like python enumerate
91  * @param iterable An iterable container to wrap
92  */
93  enumerate(T& iterable) : iterable(iterable) {}
94  };
95 } // namespace utility::support
