34 namespace dynamixel::v2 {
35  /**
36  * @brief This struct mimics the expected data structure for a RegWrite command.
37  *
38  * @details
39  * Instruction that registers the Instruction Packet to a standby status; Packet is later executed through
40  * the Action command. This type has it's members arranged in the same way as a raw array of this command would.
41  * Because of this you cannot add or remove members from this type. The template argument and parameter allows you
42  * to write any type. For example if you write with a uint16_t then it will write two bytes to the device. And if
43  * you use a struct with 3 uint16_t's in it, then you directly write to something with an x, y and z byte.
44  *
45  * @tparam T the type of data to be written
46  *
47  * @author Alex Biddulph
48  */
49  template <typename T>
50  struct RegWriteCommand {
51 
52  RegWriteCommand(uint8_t id, uint16_t address, T data)
53  : magic(0x00FDFFFF)
54  , id(id)
55  , length(3 + sizeof(address) + sizeof(T))
56  , instruction(Instruction::REG_WRITE)
57  , address(address)
58  , data(data)
59  , checksum(calculate_checksum(this)) {}
60 
61  /// Magic number that heads up every packet
62  const uint32_t magic;
63  /// The ID of the device that we are communicating with
64  const uint8_t id;
65  /// The total length of the data packet (3 plus however many bytes we are writing)
66  const uint16_t length;
67  /// The instruction that we will be executing
68  const uint8_t instruction;
69  /// The address we are writing to
70  const uint16_t address;
71  /// The bytes that we are writing
72  const T data;
73  /// Our checksum for this command
74  const uint16_t checksum;
75  } __attribute__((packed)); // Make it so that the compiler reads this struct "as is" (no padding bytes)
76 
77 } // namespace dynamixel::v2
