47  /// The ID of the provider that requested the task

49  // If the requester of this task is a root provider
50  bool root = false;
51  };

53  /// The ID of the active provider

56  TaskInfo active_task = TaskInfo();
57

60

67  template <typename T>
68  struct GroupInfoStore {
69  private:
70  using GlobalStore = NUClear::util::TypeMap<T, T, const GroupInfo>;
71 
72  public:
73  /**
74  * Set the group info for this thread and return a lock that when destroyed will upgrade the data to the
75  * global store.
76  *
77  * @param info the group info to set
78  *
79  * @return a lock object that once destroyed will upgrade the data to the global store and clear the thread
80  * store
81  */
82  static Lock set(const std::shared_ptr<const GroupInfo>& info) {
83  auto lock = Lock([info] {
84  GlobalStore::set(info);
85  local_data = nullptr;
86  });
87  local_data = info;
88  return lock;
89  }
90 
91  static std::shared_ptr<const GroupInfo> get() {
92  return local_data == nullptr ? GlobalStore::get() : local_data;
93  }
94 
95  private:
96  static thread_local std::shared_ptr<const GroupInfo> local_data;
97  };
98 
99  template <typename T>
100  thread_local std::shared_ptr<const GroupInfo> GroupInfoStore<T>::local_data = nullptr;
