59  /**
60  * Binds a Provider object, sending it to the Director so it can control its execution.
61  *
62  * @tparam DSL the NUClear DSL for the on statement
63  *
64  * @param reaction the reaction object that we are binding the Provider to
65  */
66  template <typename DSL>
67  static void bind(const std::shared_ptr<NUClear::threading::Reaction>& reaction) {
68 
69  // Tell the Director
70  reaction->reactor.powerplant.emit<NUClear::dsl::word::emit::Inline>(
71  std::make_unique<commands::ProvideReaction>(
72  reaction,
73  typeid(T),
74  classification,
75  [](const NUClear::id_t& allowed_id,
76  const RunReason& run_reason,
77  const std::shared_ptr<const void>& data,
78  const std::shared_ptr<const GroupInfo>& info) -> Lock {
79  return {
80  information::RunReasonStore::set(run_reason), // Make clang-format not be dumb
81  information::TaskDataStore<T>::set(allowed_id, data),
82  information::GroupInfoStore<T>::set(info),
83  };
84  }));
85 
86  // Add our unbinder
87  reaction->unbinders.emplace_back([](const NUClear::threading::Reaction& r) {
88  r.reactor.emit<NUClear::dsl::word::emit::Inline>(
89  std::make_unique<NUClear::dsl::operation::Unbind<commands::ProvideReaction>>(r.id));
90  });
91  }

181  /**
182  * Limit access to this Provider unless a condition is true.
183  * This will prevent a Provider from running if the state is false.
184  * However if there is a `Provide` reaction that has a Causing relationship for this When, then
185  * depending on the priority of this task, it can force a change in which provider will be run
186  *
187  * @tparam State the smart enum that is being monitored for the when condition
188  * @tparam expr the function used for the comparison (e.g. std::less)
189  * @tparam value the value that the when condition is looking for
190  */
191  template <typename State, template <typename> class expr, State::Value value>
192  struct When {
193 
194  /**
195  * Bind a when expression in the Director.
196  *
197  * @tparam DSL the DSL from NUClear
198  *
199  * @param reaction the reaction that is having this when condition bound to it
200  */
201  template <typename DSL>
202  static void bind(const std::shared_ptr<NUClear::threading::Reaction>& reaction) {
203 
204  // Tell the director about this when condition
205  reaction->reactor.emit<NUClear::dsl::word::emit::Inline>(std::make_unique<commands::WhenExpression>(
206  reaction,
207  // typeindex of the enum value
208  typeid(State),
209  // Function that uses expr to determine if the passed value v is valid
210  [](const int& v) -> bool { return expr<int>()(v, value); },
211  // Function that uses get to get the current state of the condition
212  [reaction]() -> int {
213  // Check if there is cached data, and if not throw an exception
214  auto ptr = NUClear::dsl::operation::CacheGet<State>::template get<DSL>(*reaction->get_task());
215  if (ptr == nullptr) {
216  throw std::runtime_error("The state requested has not been emitted yet");
217  }
218  return static_cast<int>(*ptr);
219  },
220  // Binder function that lets a reactor bind a function that is called when the state changes
221  [](NUClear::Reactor& reactor,
222  const std::function<void(const int&)>& fn) -> NUClear::threading::ReactionHandle {
223  return reactor.on<NUClear::dsl::word::Trigger<State>>().then(
224  [fn](const State& s) { fn(static_cast<int>(s)); });
225  }));
226  }
227  };

264  /**
265  * Adds a Uses object to the callback to allow access to information from the context of this provider.
266  *
267  * The uses object provides information about what would happen if you were to emit a specific task from this
268  * provider as well as what has happened in previous calls.
269  *
270  * It has two main uses
271  * It allows you to see what would happen if you emitted a task (run, block proxy, etc)
272  * It allows you to see what has happened in previous calls, e.g. are we re-running because our previous task
273  * emitted that it was done
274  *
275  * @tparam T the type of the provider this uses is for
276  */
277  template <typename T>
278  struct Uses {
279 
280  /// The current run state of the group
281  RunState run_state;
282  /// Whether the task is done or not, regardless of if this provider group is running it
283  bool done;
284 
285  template <typename DSL>
286  static std::optional<Uses<T>> get(NUClear::threading::ReactionTask& t) {
287 
288  // Default values if we can't find the group info
289  Uses<T> data;
290  data.run_state = RunState::NO_TASK;
291  data.done = false;
292 
293  auto group_info = information::GroupInfoStore<T>::get();
294  bool root_provider = t.id != state::current_task_id;
295  data.done = group_info->done;
296 
297  // If we can't find the group info then cannot fill the data
298  if (group_info == nullptr) {
299  return data;
300  }
301 
302  // If it is a root provider, then the requester_id will not match the parent id
303  // Then if the active task is run from root, then it is running
304  // If a watcher is root, then it is queued
305  if (root_provider) {
306  data.run_state = group_info->active_task.root
307  ? RunState::RUNNING
308  : (std::any_of(group_info->watchers.begin(),
309  group_info->watchers.end(),
310  [&t](const auto& watcher) { return watcher.root; })
311  ? RunState::QUEUED
312  : RunState::NO_TASK);
313 
314  return data;
315  }
316 
317  // Not root, check against ids
318  auto is_target_task = [&t](const auto& task) {
319  return task.requester_id == t.parent->id && task.type == typeid(T);
320  };
321  data.run_state = group_info->active_task.requester_id == t.parent->id ? RunState::RUNNING
322  : std::any_of(group_info->watchers.begin(), group_info->watchers.end(), is_target_task)
323  ? RunState::QUEUED
324  : RunState::NO_TASK;
325 
326  return data;
327  }
328  };

356  /**
357  * A Task to be executed by the director and a priority with which to execute this task.
358  * Higher priority tasks are preferred over lower priority tasks.
359  *
360  * There are two different types of tasks that can be created using this emit, root level tasks and subtasks.
361  *
362  * Root level tasks:
363  * These are created when a reaction that is not a Provider emits the task. These tasks form the root of the
364  * execution tree and their needs will be met on a highest priority first basis. These tasks will persist until the
365  * Provider that they use emits a done task, or the task is re-emitted from anywhere with nullptr as the data. They
366  * will also be overridden if a new task is emitted anywhere in the system even if it is at a lower priority.
367  *
368  * Subtasks:
369  * These are created when a Provider task emits a task to complete. These tasks must be emitted each time that
370  * reaction is run to persist as if that reaction runs again and does not emit these tasks they will be
371  * cancelled. Within these tasks the priority is used to break ties between two subtasks that share the same
372  * root task. In these cases the subtask that requested the Provider with the highest priority will have its
373  * task executed. The other subtask will be blocked until the active task is no longer in the call queue.
374  *
375  * If a subtask is emitted with optional then it is compared differently to other tasks when it comes to priority.
376  * This task and all its descendants will be considered optional. If it is compared to a task that does not have
377  * optional in its parentage, the non optional task will win. However, descendants of this task that are not
378  * optional will compare to each other as normal. If two tasks both have optional in their parentage they will be
379  * compared as normal.
380  *
381  * @tparam T the Provider type that this task is for
382  */
383  template <typename T>
384  struct Task {
385 
386  /**
387  * Emits a new task for the behaviour system to handle.
388  *
389  * @param powerplant the powerplant context provided by NUClear
390  * @param data the data element of the task
391  * @param name a string identifier to help with debugging
392  * @param priority the priority that this task is to be executed with
393  * @param optional if this task is an optional task and can be taken over by lower priority non optional tasks
394  */
395  static void emit(NUClear::PowerPlant& powerplant,
396  std::shared_ptr<T> data,
397  const int& priority = 0,
398  const bool& optional = false,
399  const std::string& name = "") {
400 
401  /// Emit using the provider scope as it will know how to handle the task
402  ProviderScope::emit<T>(powerplant, data, priority, optional, name);
403  }
404  };
