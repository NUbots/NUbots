49  /**
50  * Binds a Provider object, sending it to the Director so it can control its execution.
51  *
52  * @tparam DSL the NUClear DSL for the on statement
53  *
54  * @param reaction the reaction object that we are binding the Provider to
55  */
56  template <typename DSL>
57  static inline void bind(const std::shared_ptr<NUClear::threading::Reaction>& reaction) {
58 
59  // Tell the Director
60  reaction->reactor.powerplant.emit<NUClear::dsl::word::emit::Direct>(
61  std::make_unique<commands::ProvideReaction>(reaction, typeid(T), classification));
62 
63  // Add our unbinder
64  reaction->unbinders.emplace_back([](const NUClear::threading::Reaction& r) {
65  r.reactor.emit<NUClear::dsl::word::emit::Direct>(
66  std::make_unique<NUClear::dsl::operation::Unbind<commands::ProvideReaction>>(r.id));
67  });
68  }

72  *
73  * @tparam DSL the NUClear DSL for the on statement

76  *
77  * @return the information needed by the on statement

83  auto run_info = std::make_shared<RunInfo>(information::InformationSource::get_run_info(r.id));

134  /**
135  * Limit access to this Provider unless a condition is true.
136  * This will prevent a Provider from running if the state is false.
137  * However if there is a `Provide` reaction that has a Causing relationship for this When, then
138  * depending on the priority of this task, it can force a change in which provider will be run
139  *
140  * @tparam State the smart enum that is being monitored for the when condition
141  * @tparam expr the function used for the comparison (e.g. std::less)
142  * @tparam value the value that the when condition is looking for
143  */
144  template <typename State, template <typename> class expr, State::Value value>
145  struct When {
146 
147  /**
148  * Bind a when expression in the Director.
149  *
150  * @tparam DSL the DSL from NUClear
151  *
152  * @param reaction the reaction that is having this when condition bound to it
153  */
154  template <typename DSL>
155  static inline void bind(const std::shared_ptr<NUClear::threading::Reaction>& reaction) {
156 
157  // Tell the director about this when condition
158  reaction->reactor.emit<NUClear::dsl::word::emit::Direct>(std::make_unique<commands::WhenExpression>(
159  reaction,
160  // typeindex of the enum value
161  typeid(State),
162  // Function that uses expr to determine if the passed value v is valid
163  [](const int& v) -> bool { return expr<int>()(v, value); },
164  // Function that uses get to get the current state of the condition
165  [reaction]() -> int {
166  // Check if there is cached data, and if not throw an exception
167  auto ptr = NUClear::dsl::operation::CacheGet<State>::template get<DSL>(*reaction);
168  if (ptr == nullptr) {
169  throw std::runtime_error("The state requested has not been emitted yet");
170  }
171  return static_cast<int>(*ptr);
172  },
173  // Binder function that lets a reactor bind a function that is called when the state changes
174  [](NUClear::Reactor& reactor,
175  const std::function<void(const int&)>& fn) -> NUClear::threading::ReactionHandle {
176  return reactor.on<NUClear::dsl::word::Trigger<State>>().then(
177  [fn](const State& s) { fn(static_cast<int>(s)); });
178  }));
179  }
180  };

208  /**
209  * Adds a Uses object to the callback to allow access to information from the context of this provider.
210  *
211  * The uses object provides information about what would happen if you were to emit a specific task from this
212  * provider as well as what has happened in previous calls.
213  *
214  * It has two main uses
215  * It allows you to see what would happen if you emitted a task (run, block proxy, etc)
216  * It allows you to see what has happened in previous calls, e.g. are we re-running because our previous task
217  * emitted that it was done
218  *
219  * @tparam Provider the type of the provider this uses is for
220  */
221  template <typename Provider>
222  struct Uses {
223  GroupInfo::RunState run_state;
224  bool done;
225 
226  template <typename DSL>
227  static inline std::shared_ptr<Uses<Provider>> get(NUClear::threading::Reaction& r) {
228 
229  auto group_info = information::InformationSource::get_group_info(r.id,
230  typeid(Provider),
231  typeid(commands::RootType<Provider>));
232 
233  auto data = std::make_shared<Uses<Provider>>();
234 
235  data->run_state = group_info.run_state;
236  data->done = group_info.done;
237 
238  return data;
239  }
240  };

268  /**
269  * A Task to be executed by the director and a priority with which to execute this task.
270  * Higher priority tasks are preferred over lower priority tasks.
271  *
272  * There are two different types of tasks that can be created using this emit, root level tasks and subtasks.
273  *
274  * Root level tasks:
275  * These are created when a reaction that is not a Provider emits the task. These tasks form the root of the
276  * execution tree and their needs will be met on a highest priority first basis. These tasks will persist until the
277  * Provider that they use emits a done task, or the task is re-emitted from anywhere with nullptr as the data. They
278  * will also be overridden if a new task is emitted anywhere in the system even if it is at a lower priority.
279  *
280  * Subtasks:
281  * These are created when a Provider task emits a task to complete. These tasks must be emitted each time that
282  * reaction is run to persist as if that reaction runs again and does not emit these tasks they will be
283  * cancelled. Within these tasks the priority is used to break ties between two subtasks that share the same
284  * root task. In these cases the subtask that requested the Provider with the highest priority will have its
285  * task executed. The other subtask will be blocked until the active task is no longer in the call queue.
286  *
287  * If a subtask is emitted with optional then it is compared differently to other tasks when it comes to priority.
288  * This task and all its descendants will be considered optional. If it is compared to a task that does not have
289  * optional in its parentage, the non optional task will win. However, descendants of this task that are not
290  * optional will compare to each other as normal. If two tasks both have optional in their parentage they will be
291  * compared as normal.
292  *
293  * @tparam T the Provider type that this task is for
294  */
295  template <typename T>
296  struct Task {
297 
298  /**
299  * Emits a new task for the behaviour system to handle.
300  *
301  * @param powerplant the powerplant context provided by NUClear
302  * @param data the data element of the task
303  * @param name a string identifier to help with debugging
304  * @param priority the priority that this task is to be executed with
305  * @param optional if this task is an optional task and can be taken over by lower priority non optional tasks
306  */
307  static void emit(NUClear::PowerPlant& powerplant,
308  std::shared_ptr<T> data,
309  const int& priority = 0,
310  const bool& optional = false,
311  const std::string& name = "") {
312 
313  // Work out who is sending the task so we can determine if it's a subtask
314  const auto* task = NUClear::threading::ReactionTask::get_current_task();
315  uint64_t reaction_id = (task != nullptr) ? task->parent.id : -1;
316  uint64_t task_id = (task != nullptr) ? task->id : -1;
317 
318  NUClear::dsl::word::emit::Direct<commands::BehaviourTask>::emit(
319  powerplant,
320  std::make_shared<commands::BehaviourTask>(typeid(T),
321  typeid(commands::RootType<T>),
322  reaction_id,
323  task_id,
324  data,
325  name,
326  priority,
327  optional));
328  }
329  };
