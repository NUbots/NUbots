125  /// @see servo_states
126  struct ServoState {
127  /// @brief True if we need to write new values to the hardware
128  bool dirty = false;
129 
130  /// @brief Current error state of the servo
131  /// @note different to the dynamixel packet error status
132  uint8_t hardware_error = 0;

137  /// @brief True if we simulate where we think the servos should be
138  /// @note that we still write the commands to hardware
139  bool simulated = false;

141  /// @brief Our internal system torque target, this is never sent to the servo
142  int torque = 0;
143  /// @brief Whether the servo's torque is enabled, allowing it to move
144  bool torque_enabled = true;
145 
146  // Cached values that are never read and are sent to the servos
147  /// @brief The integral gain of the velocity
148  float velocity_i_gain = 1920.0f / 65536.0f; // ROBOTIS default
149  /// @brief The proportional gain of the velocity
150  float velocity_p_gain = 100.0f / 128.0f; // ROBOTIS default
151  /// @brief The derivative gain of the position
152  float position_d_gain = 0.0f;
153  /// @brief The integral gain of the position
154  float position_i_gain = 0.0f;
155  /// @brief The proportional gain of the position
156  float position_p_gain = 850.0f / 128.0f; // ROBOTIS default
157  /// @brief The first feedforward torque coefficient
158  float feedforward_1st_gain = 0.0f;
159  /// @brief The second feedforward torque coefficient
160  float feedforward_2nd_gain = 0.0f;
161  /// @brief The target pulse width modulation of the servo
162  float goal_pwm = 885.0f; // ROBOTIS default
163  /// @brief The target current of the servo
164  float goal_current = 6.52176f / 0.00336f; // ROBOTIS default
165  /// @brief The target velocity of the servo, not used by the servos
166  float goal_velocity = 1.08775f; // ROBOTIS default
167  /// @brief The target position of the servo
168  float goal_position = 0.0f;
169  /// @brief The target acceleration of the servo
170  float profile_acceleration = 0.0f;
171  /// @brief The target velocity of the servo, replacing moving speed in v1 protocol
172  float profile_velocity = 0.0f;
173 
174  // Values that are either simulated or read from the servos
175  /// @brief The last read pulse width modulation of the servo
176  float present_pwm = 0.0f;
177  /// @brief The last read current of the servo
178  float present_current = 0.0f;
179  /// @brief The last read velocity of the servo
180  float present_velocity = 0.0f;
181  /// @brief The last read position of the servo
182  float present_position = 0.0f;
183  /// @brief The last read voltage of the servo
184  float voltage = 0.0f;
185  /// @brief The last read temperature of the servo
186  float temperature = 0.0f;
187 
188  /// @brief Whether we have initialised this servo yet
189  bool initialised = false;
190  };
