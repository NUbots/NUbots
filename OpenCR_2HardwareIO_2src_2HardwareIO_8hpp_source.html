122  /// @see servo_states
123  struct ServoState {
124  /// @brief True if we need to write new values to the hardware
125  bool dirty = false;
126 
127  /// @brief Current error state of the servo
128  /// @note different to the dynamixel packet error status
129  uint8_t hardware_error = 0;
130 
131  /// @brief Most recent packet error received, to cache before RawSensors is emitted
132  uint8_t packet_error = 0;
133 
134  /// @brief True if we simulate where we think the servos should be
135  /// @note that we still write the commands to hardware
136  bool simulated = false;
137 
138  /// @brief Our internal system torque target, this is never sent to the servo
139  int torque = 0;
140  /// @brief Whether the servo's torque is enabled, allowing it to move
141  bool torque_enabled = true;
142 
143  // Cached values that are never read and are sent to the servos
144  /// @brief The integral gain of the velocity
145  float velocity_i_gain = 1920.0f / 65536.0f; // ROBOTIS default
146  /// @brief The proportional gain of the velocity
147  float velocity_p_gain = 100.0f / 128.0f; // ROBOTIS default
148  /// @brief The derivative gain of the position
149  float position_d_gain = 0.0f;
150  /// @brief The integral gain of the position
151  float position_i_gain = 0.0f;
152  /// @brief The proportional gain of the position
153  float position_p_gain = 850.0f / 128.0f; // ROBOTIS default
154  /// @brief The first feedforward torque coefficient
155  float feedforward_1st_gain = 0.0f;
156  /// @brief The second feedforward torque coefficient
157  float feedforward_2nd_gain = 0.0f;
158  /// @brief The target pulse width modulation of the servo
159  float goal_pwm = 885.0f; // ROBOTIS default
160  /// @brief The target current of the servo
161  float goal_current = 6.52176f / 0.00336f; // ROBOTIS default
162  /// @brief The target velocity of the servo, not used by the servos
163  float goal_velocity = 1.08775f; // ROBOTIS default
164  /// @brief The target position of the servo
165  float goal_position = 0.0f;
166  /// @brief The target acceleration of the servo
167  float profile_acceleration = 0.0f;
168  /// @brief The target velocity of the servo, replacing moving speed in v1 protocol
169  float profile_velocity = 0.0f;
170 
171  // Values that are either simulated or read from the servos
172  /// @brief The last read pulse width modulation of the servo
173  float present_pwm = 0.0f;
174  /// @brief The last read current of the servo
175  float present_current = 0.0f;
176  /// @brief The last read velocity of the servo
177  float present_velocity = 0.0f;
178  /// @brief The last read position of the servo
179  float present_position = 0.0f;
180  /// @brief The last read voltage of the servo
181  float voltage = 0.0f;
182  /// @brief The last read temperature of the servo
183  float temperature = 0.0f;
184 
185  /// @brief Whether we have initialised this servo yet
186  bool initialised = false;
187  };
