79  class Animator {

86  Animator(const std::vector<SixDOFFrame>& frames_) : frames(frames_) {}
87  int clampPrev(int k) const {
88  return std::max(std::min(k, int(frames.size() - 2)), 0);
89  }
90  int clampCurrent(int k) const {
91  return std::max(std::min(k, int(frames.size() - 1)), 0);
92  }
93  void next() {
94  i = clampPrev(i + 1);
95  }
96  void reset() {
97  i = 0;
98  }
99  const SixDOFFrame& currentFrame() const {

101  }

103  return frames[i];
104  }
105  const SixDOFFrame& startFrame() const {

107  }

191  Eigen::Isometry3d getTorsoPose(const message::input::Sensors& sensors) {
192  // Find position vector from support foot to torso in support foot coordinates.
193  return ((supportFoot == utility::input::LimbID::LEFT_LEG)
194  ? Eigen::Isometry3d(sensors.Htx[utility::input::ServoID::L_ANKLE_ROLL])
195  : Eigen::Isometry3d(sensors.Htx[utility::input::ServoID::R_ANKLE_ROLL]));
196  }
197 
198  Eigen::Isometry3d getFootPose(const message::input::Sensors& sensors) {
199  auto result = Eigen::Isometry3d::Identity();
200  if (stage == MotionStage::RUNNING || stage == MotionStage::STOPPING) {
201 
202  double elapsedTime =
203  std::chrono::duration_cast<std::chrono::microseconds>(sensors.timestamp - motionStartTime).count()
204  * 1e-6;
205  double alpha = (anim.currentFrame().duration != 0)
206  ? std::fmax(0, std::fmin(elapsedTime / anim.currentFrame().duration, 1))
207  : 1;
208 
209  result = interpolate(anim.previousFrame().pose, anim.currentFrame().pose, alpha);
210 
211  bool servosAtGoal = true;
212 
213  // Check all the servos between R_HIP_YAW and L_ANKLE_ROLL are within the angle threshold
214  for (int id = utility::input::ServoID::R_HIP_YAW; id <= utility::input::ServoID::L_ANKLE_ROLL; ++id) {
215  if (std::fabs(sensors.servo[id].goal_position - sensors.servo[id].present_position)
216  > servo_angle_threshold) {
217  servosAtGoal = false;
218  break;
219  }
220  }
221 
222  if (alpha >= 1.0 && servosAtGoal) {
223  stable = anim.stable();
224  if (!stable) {
225  anim.next();
226  motionStartTime = sensors.timestamp;
227  }
228  }
229  if (stable && stage == MotionStage::STOPPING) {
230  stage = MotionStage::FINISHED;
231  }
232  }
233  return result;
234  }
