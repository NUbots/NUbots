82  return std::max(std::min(k, int(frames.size() - 1)), 0);

182  Eigen::Isometry3d getTorsoPose(const message::input::Sensors& sensors) {
183  // Find position vector from support foot to torso in support foot coordinates.
184  return ((supportFoot == utility::input::LimbID::LEFT_LEG)
185  ? Eigen::Isometry3d(sensors.Htx[utility::input::ServoID::L_ANKLE_ROLL])
186  : Eigen::Isometry3d(sensors.Htx[utility::input::ServoID::R_ANKLE_ROLL]));
187  }
188 
189  Eigen::Isometry3d getFootPose(const message::input::Sensors& sensors) {
190  auto result = Eigen::Isometry3d::Identity();
191  if (stage == MotionStage::RUNNING || stage == MotionStage::STOPPING) {
192 
193  double elapsedTime =
194  std::chrono::duration_cast<std::chrono::microseconds>(sensors.timestamp - motionStartTime).count()
195  * 1e-6;
196  double alpha = (anim.currentFrame().duration != 0)
197  ? std::fmax(0, std::fmin(elapsedTime / anim.currentFrame().duration, 1))
198  : 1;
199 
200  result = interpolate(anim.previousFrame().pose, anim.currentFrame().pose, alpha);
201 
202  bool servosAtGoal = true;
203 
204  // Check all the servos between R_HIP_YAW and L_ANKLE_ROLL are within the angle threshold
205  for (int id = utility::input::ServoID::R_HIP_YAW; id <= utility::input::ServoID::L_ANKLE_ROLL; ++id) {
206  if (std::fabs(sensors.servo[id].goal_position - sensors.servo[id].present_position)
207  > servo_angle_threshold) {
208  servosAtGoal = false;
209  break;
210  }
211  }
212 
213  if (alpha >= 1.0 && servosAtGoal) {
214  stable = anim.stable();
215  if (!stable) {
216  anim.next();
217  motionStartTime = sensors.timestamp;
218  }
219  }
220  if (stable && stage == MotionStage::STOPPING) {
221  stage = MotionStage::FINISHED;
222  }
223  }
224  return result;
225  }
