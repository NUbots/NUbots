56 /**
57  * @brief This struct mimics the expected data structure for a Write command.
58  *
59  * @details
60  * This type has it's members arranged in the same way as a raw array of this command would. Because of this
61  * you cannot add or remove members from this type (unless for some reason the API to the CM740 changes). The
62  * template argument and parameter allows you to read any type. For example if you read with a uint16_t then
63  * it will read two bytes to the device. And if you use a struct with 3 uint16_t's in it, then you can directly
64  * read a paramter with an x, y and z bytes (e.g. the accelerometer)
65  * @tparam TType the type of data to be written
66  *
67  * @author Trent Houliston
68  */
69 #pragma pack(push, 1) // Make it so that the compiler reads this struct "as is" (no padding bytes)
70  template <typename TType>
71  struct ReadCommand {
72 
73  ReadCommand(uint8_t id, uint8_t address) : id(id), address(address), size(sizeof(TType)) {}
74 
75  /// Magic number that heads up every packet
76  const uint16_t magic = 0xFFFF;
77  /// The ID of the device that we are communicating with
78  const uint8_t id;
79  /// The total length of the data packet (always 4)
80  const uint8_t length = 4;
81  /// The instruction that we will be executing (The READ instruction)
82  const uint8_t instruction = Instruction::READ;
83  /// The address to read from
84  const uint8_t address;
85  /// The number of bytes to read
86  const uint8_t size;
87  /// Our checksum for this command
88  const uint8_t checksum = calculateChecksum(this);
89  };
90 #pragma pack(pop)

92 /**
93  * @brief This struct mimics the expected data structure for a Write command.

95  * @details
96  * This type has it's members arranged in the same way as a raw array of this command would. Because of this
97  * you cannot add or remove members from this type (unless for some reason the API to the CM740 changes). The
98  * template argument and parameter allows you to write any type. For example if you write with a uint16_t then
99  * it will write two bytes to the device. And if you use a struct with 3 uint16_t's in it, then you directly
100  * write to something with an x, y and z byte.
101  *
102  * @tparam TType the type of data to be written
103  *
104  * @author Trent Houliston
105  */
106 #pragma pack(push, 1) // Make it so that the compiler reads this struct "as is" (no padding bytes)
107  template <typename TType>
108  struct WriteCommand {
109 
110  WriteCommand(uint8_t id, uint8_t address, TType data)
111  : id(id), length(3 + sizeof(TType)), address(address), data(data) {}
112 
113  /// Magic number that heads up every packet
114  const uint16_t magic = 0xFFFF;
115  /// The ID of the device that we are communicating with
116  const uint8_t id;
117  /// The total length of the data packet (3 plus however many bytes we are writing)
118  const uint8_t length;
119  /// The instruction that we will be executing (The WRITE instruction)
120  const uint8_t instruction = Instruction::WRITE;
121  /// The address we are writing to
122  const uint8_t address;
123  /// The bytes that we are writing
124  const TType data;
125  /// Our checksum for this command
126  const uint8_t checksum = calculateChecksum(this);
127  };

165  /**
166  * @brief Reads from this device at the given memory address.
167  *
168  * @details
169  * This will read from the current device at the given memory address, and return an object of type TType.
170  * It will read enough consecutive bytes to fill the TType with data. For example, if you read a 2 byte wide
171  * datatype, it will read 2 consecutive bytes from the device to fill it.
172  *
173  * @tparam TType the type of data we are reading
174  *
175  * @param address the address of the data that we are reading
176  *
177  * @return the data that was read from the device
178  */
179  template <typename TType>
180  TType read(uint8_t address) {
181 
182  // Check that this struct is not cache alligned
183  static_assert(sizeof(ReadCommand<TType>) == 8,
184  "The compiler is adding padding to this struct, Bad compiler!");
185 
186  // Execute our Read command through the uart
187  CommandResult result = coms.executeRead(ReadCommand<TType>(id, address));
188 
189  // If there was an error then try reading again
190  if (result.data.size() != sizeof(TType)) {
191  result = coms.executeRead(ReadCommand<TType>(id, address));
192  }
193 
194  // If it's still bad then throw a runtime error
195  if (result.data.size() != sizeof(TType)) {
196  throw std::runtime_error("There was an error while trying to read from the device");
197  }
198 
199  // Copy our resulting data into our return type
200  TType data;
201  memcpy(&data, result.data.data(), sizeof(TType));
202 
203  // Return our data
204  return data;
205  }
206 
207  /**
208  * @brief Writes data to this device at the given memory address.
209  *
210  * @details
211  * This will write the passed object to the current device at the given memory address. It will write
212  * consecutive bytes for the size of TType. For example, if you write a 2 byte wide datatype, it will write 2
213  * consecutive bytes from the device.
214  *
215  * @tparam TType the type of data we are writing
216  *
217  * @param address the address of the data that we are reading
218  * @param data the data that we are writing to the device
219  */
220  template <typename TType>
221  void write(uint8_t address, TType data) {
222 
223  // Check that this struct is not cache alligned
224  static_assert(sizeof(WriteCommand<TType>) == 7 + sizeof(TType),
225  "The compiler is adding padding to this struct, Bad compiler!");
226 
227  // Write our data over the UART
228  coms.executeWrite(WriteCommand<TType>(id, address, data));
229  }

231  /**
232  * @brief Reads a specified number of bytes from this device.
233  *
234  * @param data Vector to store the read data in. Vector will be resized to hold the requested data.
235  * Contents of the vector will be overwritten.
236  *
237  * @param count Number of bytes to read from this device.
238  *
239  * @return the number of bytes.
240  */
241  size_t readBytes(std::vector<uint8_t>& data, size_t count) {
242 
243  // Make sure the vector has enough space in it.
244  data.resize(count);
245  return coms.readBytes(data.data(), count);
246  }
247 
248  /**
249  * @brief Writes a specified number of bytes to this device.
250  *
251  * @param data Vector of bytes to send.
252  *
253  * @return the number of bytes.
254  */
255  size_t writeBytes(const std::vector<uint8_t>& data) {
256 
257  return coms.writeBytes(data.data(), data.size());
258  }
