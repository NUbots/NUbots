53  [[nodiscard]] inline std::map<ServoID, Eigen::Isometry3d> calculateHeadJointPosition(const KinematicsModel& model,
54  const float& HEAD_PITCH,
55  const float& HEAD_YAW,
56  const ServoID& servoID) {
57  std::map<ServoID, Eigen::Isometry3d> positions{};
58 
59  Eigen::Isometry3d runningTransform = Eigen::Isometry3d::Identity();
60  const Eigen::Vector3d NECK_POS(model.head.NECK_BASE_POS_FROM_ORIGIN_X,
61  model.head.NECK_BASE_POS_FROM_ORIGIN_Y,
62  model.head.NECK_BASE_POS_FROM_ORIGIN_Z);
63  const float NECK_LENGTH = model.head.NECK_LENGTH;
64 
65  // Translate to base of neck from origin
66  runningTransform = runningTransform.translate(NECK_POS);
67  // Rotate to face out of base of neck
68  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(-M_PI_2, Eigen::Vector3d::UnitY()));
69  // Rotate head in yaw axis
70  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(HEAD_YAW, Eigen::Vector3d::UnitX()));
71  // Translate to top of neck (i.e. next motor axle)
72  runningTransform = runningTransform.translate(Eigen::Vector3d(NECK_LENGTH, 0.0, 0.0));
73  // YAW
74  // Return the basis pointing out of the top of the torso with z pointing out the back of the neck. Pos
75  // is top of neck (at hip pitch motor)
76  positions[ServoID::HEAD_YAW] = runningTransform;
77  if (servoID == ServoID::HEAD_YAW) {
78  return positions;
79  }
80 
81  // Rotate to face forward direction of neck
82  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(M_PI_2, Eigen::Vector3d::UnitY()));
83  // Rotate pitch
84  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(HEAD_PITCH, Eigen::Vector3d::UnitY()));
85  // PITCH
86  // Return basis pointing along camera vector (ie x is camera vector, z out of top of head). Pos at
87  // camera position
88  positions[ServoID::HEAD_PITCH] = runningTransform;
89  return positions;
90  }

107  [[nodiscard]] inline std::map<ServoID, Eigen::Isometry3d> calculateLegJointPosition(const KinematicsModel& model,
108  const Sensors& sensors,
109  const ServoID& servoID,
110  const BodySide& isLeft) {
111  std::map<ServoID, Eigen::Isometry3d> positions{};
112  Eigen::Isometry3d runningTransform = Eigen::Isometry3d::Identity();
113  // Variables to mask left and right leg differences:
114  ServoID HIP_YAW;
115  ServoID HIP_ROLL;
116  ServoID HIP_PITCH;
117  ServoID KNEE;
118  ServoID ANKLE_PITCH;
119  ServoID ANKLE_ROLL;
120  int negativeIfRight = 1;
121 
122  if (isLeft == BodySide::LEFT) {
123  HIP_YAW = ServoID::L_HIP_YAW;
124  HIP_ROLL = ServoID::L_HIP_ROLL;
125  HIP_PITCH = ServoID::L_HIP_PITCH;
126  KNEE = ServoID::L_KNEE;
127  ANKLE_PITCH = ServoID::L_ANKLE_PITCH;
128  ANKLE_ROLL = ServoID::L_ANKLE_ROLL;
129  }
130  else {
131  HIP_YAW = ServoID::R_HIP_YAW;
132  HIP_ROLL = ServoID::R_HIP_ROLL;
133  HIP_PITCH = ServoID::R_HIP_PITCH;
134  KNEE = ServoID::R_KNEE;
135  ANKLE_PITCH = ServoID::R_ANKLE_PITCH;
136  ANKLE_ROLL = ServoID::R_ANKLE_ROLL;
137  negativeIfRight = -1;
138  }
139 
140  // Hip pitch
141  runningTransform = runningTransform.translate(
142  Eigen::Vector3d(model.leg.HIP_OFFSET_X, negativeIfRight * model.leg.HIP_OFFSET_Y, -model.leg.HIP_OFFSET_Z));
143  // Rotate to face down the leg (see above for definitions of terms, including 'facing')
144  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(M_PI_2, Eigen::Vector3d::UnitY()));
145  // Using right hand rule along global z gives positive direction of yaw:
146  runningTransform = runningTransform.rotate(
147  Eigen::AngleAxisd(-sensors.servo[HIP_YAW].present_position, Eigen::Vector3d::UnitX()));
148  // Return basis facing from body to hip centre (down) with z aligned with the axis of the hip roll motor
149  // axis. Position at hip joint
150  positions[HIP_YAW] = runningTransform;
151  if (servoID == HIP_YAW) {
152  return positions;
153  }
154 
155  runningTransform = runningTransform.rotate(
156  Eigen::AngleAxisd(sensors.servo[HIP_ROLL].present_position, Eigen::Vector3d::UnitZ()));
157  // Return basis facing down leg plane, with z oriented through axis of roll motor. Position still hip
158  // joint
159  positions[HIP_ROLL] = runningTransform;
160  if (servoID == HIP_ROLL) {
161  return positions;
162  }
163 
164  // Rotate to face down upper leg
165  runningTransform = runningTransform.rotate(
166  Eigen::AngleAxisd(sensors.servo[HIP_PITCH].present_position, Eigen::Vector3d::UnitY()));
167  // Translate down upper leg
168  runningTransform = runningTransform.translate(Eigen::Vector3d(model.leg.UPPER_LEG_LENGTH, 0.0, 0.0));
169  // Return basis faces down upper leg, with z out of front of thigh. Pos = knee axis centre
170  positions[HIP_PITCH] = runningTransform;
171  if (servoID == HIP_PITCH) {
172  return positions;
173  }
174 
175 
176  // Rotate to face down lower leg
177  runningTransform =
178  runningTransform.rotate(Eigen::AngleAxisd(sensors.servo[KNEE].present_position, Eigen::Vector3d::UnitY()));
179  // Translate down lower leg
180  runningTransform = runningTransform.translate(Eigen::Vector3d(model.leg.LOWER_LEG_LENGTH, 0.0, 0.0));
181  // Return basis facing down lower leg, with z out of front of shin. Pos = ankle axis centre
182  positions[KNEE] = runningTransform;
183  if (servoID == KNEE) {
184  return positions;
185  }
186 
187 
188  // Rotate to face down foot (pitch)
189  runningTransform = runningTransform.rotate(
190  Eigen::AngleAxisd(sensors.servo[ANKLE_PITCH].present_position, Eigen::Vector3d::UnitY()));
191  // Return basis facing pitch down to foot with z out the front of the foot. Pos = ankle axis centre
192  positions[ANKLE_PITCH] = runningTransform;
193  if (servoID == ANKLE_PITCH) {
194  return positions;
195  }
196 
197  // Rotate to face down foot (roll)
198  runningTransform = runningTransform.rotate(
199  Eigen::AngleAxisd(sensors.servo[ANKLE_ROLL].present_position, Eigen::Vector3d::UnitZ()));
200  // Rotate so x faces toward toes
201  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(-M_PI_2, Eigen::Vector3d::UnitY()));
202  // Translate to ground
203  runningTransform = runningTransform.translate(Eigen::Vector3d(0.0, 0.0, -model.leg.FOOT_HEIGHT));
204  // Return basis with x out of the front of the toe and z out the top of foot. Pos = ankle axis centre
205  // projected to ground
206  positions[ANKLE_ROLL] = runningTransform;
207  return positions;
208  }

217  [[nodiscard]] inline std::map<ServoID, Eigen::Isometry3d> calculateArmJointPosition(const KinematicsModel& model,
218  const Sensors& sensors,
219  const ServoID& servoID,
220  const BodySide& isLeft) {
221  std::map<ServoID, Eigen::Isometry3d> positions{};
222  Eigen::Isometry3d runningTransform = Eigen::Isometry3d::Identity();
223  // Variables to mask left and right differences:
224  ServoID SHOULDER_PITCH;
225  ServoID SHOULDER_ROLL;
226  ServoID ELBOW;
227  int negativeIfRight = 1;
228 
229  if (isLeft == BodySide::LEFT) {
230  SHOULDER_PITCH = ServoID::L_SHOULDER_PITCH;
231  SHOULDER_ROLL = ServoID::L_SHOULDER_ROLL;
232  ELBOW = ServoID::L_ELBOW;
233  }
234  else {
235  SHOULDER_PITCH = ServoID::R_SHOULDER_PITCH;
236  SHOULDER_ROLL = ServoID::R_SHOULDER_ROLL;
237  ELBOW = ServoID::R_ELBOW;
238  negativeIfRight = -1;
239  }
240 
241  const float& shoulder_pitch = sensors.servo[SHOULDER_PITCH].present_position;
242  const float& shoulder_roll = sensors.servo[SHOULDER_ROLL].present_position;
243  const float& elbow = sensors.servo[ELBOW].present_position;
244 
245  // Translate to shoulder
246  runningTransform =
247  runningTransform.translate(Eigen::Vector3d(model.arm.SHOULDER_X_OFFSET,
248  negativeIfRight * model.arm.DISTANCE_BETWEEN_SHOULDERS * 0.5,
249  model.arm.SHOULDER_Z_OFFSET));
250  // Rotate about shoulder pitch with zero position Zombie arms
251  runningTransform =
252  runningTransform.rotate(Eigen::AngleAxisd(shoulder_pitch - M_PI_2, Eigen::Vector3d::UnitY()));
253  // Translate to end of shoulder part
254  runningTransform = runningTransform.translate(Eigen::Vector3d(model.arm.SHOULDER_LENGTH,
255  negativeIfRight * model.arm.SHOULDER_WIDTH,
256  -model.arm.SHOULDER_HEIGHT));
257  // Return matrix pointing forward out of shoulder, y same as global y. Pos = at centre of shoulder roll
258  // joint
259  positions[SHOULDER_PITCH] = runningTransform;
260  if (servoID == SHOULDER_PITCH) {
261  return positions;
262  }
263 
264  // Rotate by the shoulder roll
265  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(shoulder_roll, Eigen::Vector3d::UnitX()));
266  // Translate to centre of next joint
267  runningTransform = runningTransform.translate(Eigen::Vector3d(model.arm.UPPER_ARM_X_OFFSET,
268  negativeIfRight * model.arm.UPPER_ARM_Y_OFFSET,
269  -model.arm.UPPER_ARM_LENGTH));
270  // Rotate to face down arm
271  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(M_PI_2, Eigen::Vector3d::UnitY()));
272 
273  positions[SHOULDER_ROLL] = runningTransform;
274  if (servoID == SHOULDER_ROLL) {
275  return positions;
276  }
277 
278  // Rotate by the elbow angle
279  runningTransform = runningTransform.rotate(Eigen::AngleAxisd(elbow, Eigen::Vector3d::UnitY()));
280  // Translate to centre of end of arm, in line with joint
281  runningTransform = runningTransform.translate(Eigen::Vector3d(model.arm.LOWER_ARM_LENGTH,
282  negativeIfRight * model.arm.LOWER_ARM_Y_OFFSET,
283  -model.arm.LOWER_ARM_Z_OFFSET));
284  positions[ELBOW] = runningTransform;
285  return positions;
286  }

290  [[nodiscard]] inline std::map<ServoID, Eigen::Isometry3d> calculatePosition(const KinematicsModel& model,
291  const Sensors& sensors,
292  const ServoID& servoID) {
293  switch (servoID.value) {
294  case ServoID::HEAD_YAW:
295  case ServoID::HEAD_PITCH: return calculateHeadJointPosition(model, sensors, servoID);
296  case ServoID::R_SHOULDER_PITCH:
297  case ServoID::R_SHOULDER_ROLL:
298  case ServoID::R_ELBOW: return calculateArmJointPosition(model, sensors, servoID, BodySide::RIGHT);
299  case ServoID::L_SHOULDER_PITCH:
300  case ServoID::L_SHOULDER_ROLL:
301  case ServoID::L_ELBOW: return calculateArmJointPosition(model, sensors, servoID, BodySide::LEFT);
302  case ServoID::R_HIP_YAW:
303  case ServoID::R_HIP_ROLL:
304  case ServoID::R_HIP_PITCH:
305  case ServoID::R_KNEE:
306  case ServoID::R_ANKLE_PITCH:
307  case ServoID::R_ANKLE_ROLL: return calculateLegJointPosition(model, sensors, servoID, BodySide::RIGHT);
308  case ServoID::L_HIP_YAW:
309  case ServoID::L_HIP_ROLL:
310  case ServoID::L_HIP_PITCH:
311  case ServoID::L_KNEE:
312  case ServoID::L_ANKLE_PITCH:
313  case ServoID::L_ANKLE_ROLL: return calculateLegJointPosition(model, sensors, servoID, BodySide::LEFT);
314  default: return std::map<ServoID, Eigen::Isometry3d>();
315  }
316  }

318  [[nodiscard]] inline std::map<ServoID, Eigen::Isometry3d> calculateAllPositions(const KinematicsModel& model,
319  const Sensors& sensors) {
320  std::map<ServoID, Eigen::Isometry3d> result{};
321  for (const auto& r : calculatePosition(model, sensors, ServoID::L_ANKLE_ROLL)) {
322  result[r.first] = r.second;
323  }
324  for (const auto& r : calculatePosition(model, sensors, ServoID::R_ANKLE_ROLL)) {
325  result[r.first] = r.second;
326  }
327  for (const auto& r : calculatePosition(model, sensors, ServoID::HEAD_PITCH)) {
328  result[r.first] = r.second;
329  }
330  for (const auto& r : calculatePosition(model, sensors, ServoID::L_ELBOW)) {
331  result[r.first] = r.second;
332  }
333  for (const auto& r : calculatePosition(model, sensors, ServoID::R_ELBOW)) {
334  result[r.first] = r.second;
335  }
336 
337  return result;

339  /*! @brief Adds up the mass vectors stored in the robot model and normalises the resulting position
340  @return [x_com, y_com, z_com, total_mass] relative to the torso basis
341  */
342  [[nodiscard]] inline Eigen::Vector4d calculateCentreOfMass(const message::actuation::KinematicsModel& model,
343  const std::array<Eigen::Matrix4d, 20>& Htx) {
344 
345  // Convenience function to transform particle-space CoM to torso-space CoM
346  // Htx - transform from particle space to torso space
347  // particle - CoM coordinates in particle space
348  auto com = [](const Eigen::Matrix4d& Htx, const Eigen::Vector4d& particle) {
349  // Split out CoM and mass
350  Eigen::Vector4d com(particle.x(), particle.y(), particle.z(), 1.0);
351  const double mass = particle.w();
352 
353  // Calculate CoM in torso space
354  com = Htx * com;
355 
356  return std::pair<Eigen::Vector3d, double>{Eigen::Vector3d(com.x(), com.y(), com.z()), mass};
357  };
358 
359  // Get the centre of mass for each particle in torso space
360  // There are 16 particles in total
361  std::array<std::pair<Eigen::Vector3d, double>, 16> particles = {
362  com(Htx[utility::input::ServoID::HEAD_PITCH], model.mass_model.head),
363  com(Htx[utility::input::ServoID::L_SHOULDER_PITCH], model.mass_model.arm_upper),
364  com(Htx[utility::input::ServoID::R_SHOULDER_PITCH], model.mass_model.arm_upper),
365  com(Htx[utility::input::ServoID::L_SHOULDER_ROLL], model.mass_model.arm_lower),
366  com(Htx[utility::input::ServoID::R_SHOULDER_ROLL], model.mass_model.arm_lower),
367  com(Htx[utility::input::ServoID::L_HIP_ROLL], model.mass_model.hip_block),
368  com(Htx[utility::input::ServoID::R_HIP_ROLL], model.mass_model.hip_block),
369  com(Htx[utility::input::ServoID::L_HIP_PITCH], model.mass_model.leg_upper),
370  com(Htx[utility::input::ServoID::R_HIP_PITCH], model.mass_model.leg_upper),
371  com(Htx[utility::input::ServoID::L_KNEE], model.mass_model.leg_lower),
372  com(Htx[utility::input::ServoID::R_KNEE], model.mass_model.leg_lower),
373  com(Htx[utility::input::ServoID::L_ANKLE_PITCH], model.mass_model.ankle_block),
374  com(Htx[utility::input::ServoID::R_ANKLE_PITCH], model.mass_model.ankle_block),
375  com(Htx[utility::input::ServoID::L_ANKLE_ROLL], model.mass_model.foot),
376  com(Htx[utility::input::ServoID::R_ANKLE_ROLL], model.mass_model.foot),
377  std::pair<Eigen::Vector3d, double>{
378  Eigen::Vector3d{model.mass_model.torso.x(), model.mass_model.torso.y(), model.mass_model.torso.z()},
379  model.mass_model.torso.w()},
380  };
381 
382  // Calculate the CoM for the entire robot
383  std::pair<Eigen::Vector3d, double> robot_com = {Eigen::Vector3d::Zero(), 0.0};
384  for (const auto& particle : particles) {
385  robot_com.first = (robot_com.first * robot_com.second + particle.first * particle.second)
386  / (robot_com.second + particle.second);
387  robot_com.second = robot_com.second + particle.second;
388  }
389 
390  return Eigen::Vector4d{robot_com.first.x(), robot_com.first.y(), robot_com.first.z(), robot_com.second};
391  }
392 
393  /*! @brief Transforms inertial tensors for each robot particle into torso space and sums to find the total
394  inertial tensor
395  @return [[xx, xy, xz], relative to the torso basis
396  [xy, yy, yz],
397  [xz, yz, zz]]
398  */
399  [[nodiscard]] inline Eigen::Matrix3d calculateInertialTensor(const message::actuation::KinematicsModel& model,
400  const std::array<Eigen::Matrix4d, 20>& Htx) {
401 
402  // Convenience function to transform particle-space inertial tensors to torso-space inertial tensor
403  // Htx - transform from particle space to torso space
404  // particle - CoM coordinates in particle space
405  auto translateTensor = [](const Eigen::Matrix4d& Htx,
406  const Eigen::Matrix3d& tensor,
407  const Eigen::Vector4d& com_mass) {
408  Eigen::Vector4d com(com_mass.x(), com_mass.y(), com_mass.z(), 1.0);
409  com = Htx * com;
410 
411  // Calculate distance to particle CoM from particle origin, using skew-symmetric matrix
412  const double x = com.x();
413  const double y = com.y();
414  const double z = com.z();
415  Eigen::Matrix3d d;
416  // clang-format off
417  d << y * y + z * z, -x * y, -x * z,
418  -x * y, x * x + z * z, -y * z,
419  -x * z, -y * z, x * x + y * y;
420  // clang-format on
421 
422  // We need to rotate the tensor into our torso reference frame
423  // https://en.wikipedia.org/wiki/Moment_of_inertia#Body_frame
424  Eigen::Matrix3d torso_tensor = Htx.topLeftCorner<3, 3>() * tensor * Htx.topLeftCorner<3, 3>().transpose();
425 
426  // Translate tensor using the parallel axis theorem
427  Eigen::Matrix3d tensor_com = com_mass.w() * (torso_tensor - d);
428 
429  return tensor_com;
430  };
431 
432  // Get the centre of mass for each particle in torso space
433  // There are 16 particles in total
434  std::array<Eigen::Matrix3d, 16> particles = {
435  translateTensor(Htx[utility::input::ServoID::HEAD_PITCH], model.tensor_model.head, model.mass_model.head),
436  translateTensor(Htx[utility::input::ServoID::L_SHOULDER_PITCH],
437  model.tensor_model.arm_upper,
438  model.mass_model.arm_upper),
439  translateTensor(Htx[utility::input::ServoID::R_SHOULDER_PITCH],
440  model.tensor_model.arm_upper,
441  model.mass_model.arm_upper),
442  translateTensor(Htx[utility::input::ServoID::L_SHOULDER_ROLL],
443  model.tensor_model.arm_lower,
444  model.mass_model.arm_lower),
445  translateTensor(Htx[utility::input::ServoID::R_SHOULDER_ROLL],
446  model.tensor_model.arm_lower,
447  model.mass_model.arm_lower),
448  translateTensor(Htx[utility::input::ServoID::L_HIP_ROLL],
449  model.tensor_model.hip_block,
450  model.mass_model.hip_block),
451  translateTensor(Htx[utility::input::ServoID::R_HIP_ROLL],
452  model.tensor_model.hip_block,
453  model.mass_model.hip_block),
454  translateTensor(Htx[utility::input::ServoID::L_HIP_PITCH],
455  model.tensor_model.leg_upper,
456  model.mass_model.leg_upper),
457  translateTensor(Htx[utility::input::ServoID::R_HIP_PITCH],
458  model.tensor_model.leg_upper,
459  model.mass_model.leg_upper),
460  translateTensor(Htx[utility::input::ServoID::L_ANKLE_PITCH],
461  model.tensor_model.leg_lower,
462  model.mass_model.leg_lower),
463  translateTensor(Htx[utility::input::ServoID::R_ANKLE_PITCH],
464  model.tensor_model.leg_lower,
465  model.mass_model.leg_lower),
466  translateTensor(Htx[utility::input::ServoID::L_ANKLE_PITCH],
467  model.tensor_model.ankle_block,
468  model.mass_model.ankle_block),
469  translateTensor(Htx[utility::input::ServoID::R_ANKLE_PITCH],
470  model.tensor_model.ankle_block,
471  model.mass_model.ankle_block),
472  translateTensor(Htx[utility::input::ServoID::L_ANKLE_ROLL], model.tensor_model.foot, model.mass_model.foot),
473  translateTensor(Htx[utility::input::ServoID::R_ANKLE_ROLL], model.tensor_model.foot, model.mass_model.foot),
474  model.tensor_model.torso};
475 
476  // Calculate the inertial tensor for the entire robot
477  Eigen::Matrix3d inertia_tensor = Eigen::Matrix3d::Zero();
478  for (const auto& particle : particles) {
479  inertia_tensor += particle;
480  }
481 
482  return inertia_tensor;
483  } // namespace kinematics

485  [[nodiscard]] inline Eigen::Matrix2d calculateRobotToIMU(const Eigen::Isometry3d& orientation) {
486  const Eigen::Vector3d xRobotImu = orientation.rotation().topRows<1>();
487  const Eigen::Vector2d projXRobot = xRobotImu.head<2>().normalized();
488  const Eigen::Vector2d projYRobot = Eigen::Vector2d(-projXRobot.y(), projXRobot.x());
489 
490  Eigen::Matrix2d robotToImu;

496  template <typename T, typename Scalar = typename T::Scalar, typename MatrixType = typename T::LinearMatrixType>
497  [[nodiscard]] inline T calculateGroundSpace(const T& Htf, const T& Hwt) {
498  // Retrieve rotations needed for creating the space
499  // support foot to torso rotation, and world to torso rotation
500  const MatrixType Rtf(Htf.rotation());
501 
502  // Fix the foot in world space
503  const MatrixType Rwf(Hwt.rotation() * Rtf);
504 
505  // Dot product of foot z (in world space) with world z
506  const Scalar alpha = utility::math::angle::acos_clamped(Rwf(2, 2));
507 
508  Eigen::Matrix<Scalar, 3, 1> axis(Rwf.col(2).cross(Eigen::Matrix<Scalar, 3, 1>::UnitZ()).normalized());
509 
510  // Axis angle is foot to ground
511  const MatrixType Rwg(Eigen::AngleAxis<Scalar>(alpha, axis).toRotationMatrix() * Rwf);
512  const MatrixType Rtg(Hwt.rotation().transpose() * Rwg);
513 
514  // Ground space assemble!
515  T Htg;
516  Htg.linear() = Rtg;
517  Htg.translation() = Htf.translation();
518 
519  return Htg;
520  }
