138  inline void Session::setMultiformPart(const std::pair<std::string, std::string>& fieldfield_and_filepath,
139  const std::map<std::string, std::string>& fields) {
140  // https://curl.se/libcurl/c/curl_mime_init.html
141  if (curl_) {
142  if (mime_form_ != nullptr) {
143  curl_mime_free(mime_form_);
144  mime_form_ = nullptr;
145  }
146  curl_mimepart* field = nullptr;
147 
148  mime_form_ = curl_mime_init(curl_);
149 
150  field = curl_mime_addpart(mime_form_);
151  curl_mime_name(field, fieldfield_and_filepath.first.c_str());
152  curl_mime_filedata(field, fieldfield_and_filepath.second.c_str());
153 
154  for (const auto& field_pair : fields) {
155  field = curl_mime_addpart(mime_form_);
156  curl_mime_name(field, field_pair.first.c_str());
157  curl_mime_data(field, field_pair.second.c_str(), CURL_ZERO_TERMINATED);
158  }
159 
160  curl_easy_setopt(curl_, CURLOPT_MIMEPOST, mime_form_);
161  }
162  }

186  inline Response Session::makeRequest(const std::string& contentType) {
187  std::lock_guard<std::mutex> lock(mutex_request_);
188 
189  struct curl_slist* headers = NULL;
190  if (!contentType.empty()) {
191  headers = curl_slist_append(headers, std::string{"Content-Type: " + contentType}.c_str());
192  if (contentType == "multipart/form-data") {
193  headers = curl_slist_append(headers, "Expect:");
194  }
195  }
196  headers = curl_slist_append(headers, std::string{"Authorization: Bearer " + token_}.c_str());
197  if (!organization_.empty()) {
198  headers = curl_slist_append(headers, std::string{"OpenAI-Organization: " + organization_}.c_str());
199  }
200  curl_easy_setopt(curl_, CURLOPT_HTTPHEADER, headers);
201  curl_easy_setopt(curl_, CURLOPT_URL, url_.c_str());
202 
203  std::string response_string;
204  std::string header_string;
205  curl_easy_setopt(curl_, CURLOPT_WRITEFUNCTION, writeFunction);
206  curl_easy_setopt(curl_, CURLOPT_WRITEDATA, &response_string);
207  curl_easy_setopt(curl_, CURLOPT_HEADERDATA, &header_string);
208 
209  res_ = curl_easy_perform(curl_);
210 
211  bool is_error = false;
212  std::string error_msg{};
213  if (res_ != CURLE_OK) {
214  is_error = true;
215  error_msg = "OpenAI curl_easy_perform() failed: " + std::string{curl_easy_strerror(res_)};
216  if (throw_exception_) {
217  throw std::runtime_error(error_msg);
218  }
219  else {
220  std::cerr << error_msg << '\n';
221  }
222  }
223 
224  return {response_string, is_error, error_msg};
225  }

227  inline std::string Session::easyEscape(const std::string& text) {
228  char* encoded_output = curl_easy_escape(curl_, text.c_str(), static_cast<int>(text.length()));
229  const auto str = std::string{encoded_output};
230  curl_free(encoded_output);
231  return str;
232  }

701  // POST https://api.openai.com/v1/images/edits
702  // Creates an edited or extended image given an original image and a prompt.
703  inline Json CategoryImage::edit(Json input) {
704  std::string prompt = input["prompt"].get<std::string>(); // required
705  // Default values
706  std::string mask = "";
707  int n = 1;
708  std::string size = "1024x1024";
709  std::string response_format = "url";
710  std::string user = "";
711 
712  if (input.contains("mask")) {
713  mask = input["mask"].get<std::string>();
714  }
715  if (input.contains("n")) {
716  n = input["n"].get<int>();
717  }
718  if (input.contains("size")) {
719  size = input["size"].get<std::string>();
720  }
721  if (input.contains("response_format")) {
722  response_format = input["response_format"].get<std::string>();
723  }
724  if (input.contains("user")) {
725  user = input["user"].get<std::string>();
726  }
727  openai_.setMultiformPart({"image", input["image"].get<std::string>()},
728  std::map<std::string, std::string>{{"prompt", prompt},
729  {"mask", mask},
730  {"n", std::to_string(n)},
731  {"size", size},
732  {"response_format", response_format},
