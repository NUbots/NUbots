143  inline void Session::setMultiformPart(const std::pair<std::string, std::string>& fieldfield_and_filepath,
144  const std::map<std::string, std::string>& fields) {
145  // https://curl.se/libcurl/c/curl_mime_init.html
146  if (curl_) {
147  if (mime_form_ != nullptr) {
148  curl_mime_free(mime_form_);
149  mime_form_ = nullptr;
150  }
151  curl_mimepart* field = nullptr;
152 
153  mime_form_ = curl_mime_init(curl_);
154 
155  field = curl_mime_addpart(mime_form_);
156  curl_mime_name(field, fieldfield_and_filepath.first.c_str());
157  curl_mime_filedata(field, fieldfield_and_filepath.second.c_str());
158 
159  for (const auto& field_pair : fields) {
160  field = curl_mime_addpart(mime_form_);
161  curl_mime_name(field, field_pair.first.c_str());
162  curl_mime_data(field, field_pair.second.c_str(), CURL_ZERO_TERMINATED);
163  }
164 
165  curl_easy_setopt(curl_, CURLOPT_MIMEPOST, mime_form_);
166  }
167  }

191  inline Response Session::makeRequest(const std::string& contentType) {
192  std::lock_guard<std::mutex> lock(mutex_request_);
193 
194  struct curl_slist* headers = NULL;
195  if (!contentType.empty()) {
196  headers = curl_slist_append(headers, std::string{"Content-Type: " + contentType}.c_str());
197  if (contentType == "multipart/form-data") {
198  headers = curl_slist_append(headers, "Expect:");
199  }
200  }
201  headers = curl_slist_append(headers, std::string{"Authorization: Bearer " + token_}.c_str());
202  if (!organization_.empty()) {
203  headers = curl_slist_append(headers, std::string{"OpenAI-Organization: " + organization_}.c_str());
204  }
205  curl_easy_setopt(curl_, CURLOPT_HTTPHEADER, headers);
206  curl_easy_setopt(curl_, CURLOPT_URL, url_.c_str());
207 
208  std::string response_string;
209  std::string header_string;
210  curl_easy_setopt(curl_, CURLOPT_WRITEFUNCTION, writeFunction);
211  curl_easy_setopt(curl_, CURLOPT_WRITEDATA, &response_string);
212  curl_easy_setopt(curl_, CURLOPT_HEADERDATA, &header_string);
213 
214  res_ = curl_easy_perform(curl_);
215 
216  bool is_error = false;
217  std::string error_msg{};
218  if (res_ != CURLE_OK) {
219  is_error = true;
220  error_msg = "OpenAI curl_easy_perform() failed: " + std::string{curl_easy_strerror(res_)};
221  if (throw_exception_) {
222  throw std::runtime_error(error_msg);
223  }
224  else {
225  std::cerr << error_msg << '\n';
226  }
227  }
228 
229  return {response_string, is_error, error_msg};
230  }

232  inline std::string Session::easyEscape(const std::string& text) {
233  char* encoded_output = curl_easy_escape(curl_, text.c_str(), static_cast<int>(text.length()));
234  const auto str = std::string{encoded_output};
235  curl_free(encoded_output);
236  return str;
237  }

706  // POST https://api.openai.com/v1/images/edits
707  // Creates an edited or extended image given an original image and a prompt.
708  inline Json CategoryImage::edit(Json input) {
709  std::string prompt = input["prompt"].get<std::string>(); // required
710  // Default values
711  std::string mask = "";
712  int n = 1;
713  std::string size = "1024x1024";
714  std::string response_format = "url";
715  std::string user = "";
716 
717  if (input.contains("mask")) {
718  mask = input["mask"].get<std::string>();
719  }
720  if (input.contains("n")) {
721  n = input["n"].get<int>();
722  }
723  if (input.contains("size")) {
724  size = input["size"].get<std::string>();
725  }
726  if (input.contains("response_format")) {
727  response_format = input["response_format"].get<std::string>();
728  }
729  if (input.contains("user")) {
730  user = input["user"].get<std::string>();
731  }
732  openai_.setMultiformPart({"image", input["image"].get<std::string>()},
