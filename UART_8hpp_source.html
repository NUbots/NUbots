139  /**
140  * @brief Executes a passed packet and then waits for a response, Used for single commands (read write ping)
141  *
142  * @tparam TPacket the type of packet we are executing
143  *
144  * @param command the command we are executing
145  *
146  * @return the return value from executing this command
147  */
148  template <typename TPacket>
149  CommandResult executeRead(const TPacket& command) {
150 
151  // Lock the mutex
152  std::lock_guard<std::mutex> lock(mutex);
153 
154  // Write our command to the UART
155  int written = write(fd, &command, sizeof(TPacket));
156 
157  // Wait until we finish writing before continuing (no buffering)
158  tcdrain(fd);
159 
160  // We flush our buffer, just in case there was anything random in it
161  tcflush(fd, TCIFLUSH);
162 
163  assert(written == sizeof(TPacket));
164  // If compiled with NDEBUG then technically written is unused, suppress that warning
165  (void) written;
166 
167  // Read the packet that we get in response
168  return readPacket();
169  }
170 
171  /**
172  * @brief Executes a passed packet and does not wait for a response (for writes)
173  *
174  * @tparam TPacket the type of packet we are executing
175  *
176  * @param command the command we are executing
177  */
178  template <typename TPacket>
179  void executeWrite(const TPacket& command) {
180 
181  // Lock the mutex
182  std::lock_guard<std::mutex> lock(mutex);
183 
184  // Write our command to the UART
185  int written = write(fd, &command, sizeof(TPacket));
186 
187  // Wait until we finish writing before continuing (no buffering)
188  tcdrain(fd);
189 
190  assert(written == sizeof(TPacket));
191  // If compiled with NDEBUG then technically written is unused, suppress that warning
192  (void) written;
193  }
