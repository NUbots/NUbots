36 namespace extension::behaviour {
37  /**
38  * This struct holds all the scope information for a provider.
39  *
40  * While running in a ProviderScope, the behaviour of task emits will be different.
41  */
42  struct ProviderScope {
43  ProviderScope(const std::type_index& provider_type, NUClear::threading::ReactionTask& reaction_task)
44  : provider_type(provider_type)
45  , reaction_task(reaction_task)
46  , previous_scope(std::exchange(current_scope, this)) {}
47 
48  ProviderScope(ProviderScope&& other) = delete;
49  ProviderScope& operator=(ProviderScope&& other) = delete;
50  ProviderScope(const ProviderScope&) = delete;
51  ProviderScope& operator=(const ProviderScope&) = delete;
52 
53  ~ProviderScope() {
54  // Upon desctruction restore the previous scope as the current scope
55  current_scope = previous_scope;
56 
57  // Sort the task pack so highest priority tasks come first
58  // Optional tasks are always considered lower priority than non optional tasks
59  std::stable_sort(tasks.begin(), tasks.end(), [](const auto& a, const auto& b) {
60  return a.optional == b.optional ? a.priority > b.priority : b.optional;
61  });
62 
63  // If there are any Continue tasks, do nothing else
64  // If there are other tasks with Continue, they will be ignored
65  for (const auto& task : tasks) {
66  if (task.type == typeid(::extension::behaviour::Continue)) {
67  // Tell the user if there is a Continue task with other tasks
68  if (tasks.size() > 1) {
69  NUClear::log<NUClear::LogLevel::WARN>(
70  "Continue task was emitted with other tasks, the other tasks will be ignored");
71  }
72 
73  // Do nothing else
74  return;
75  }
76  }
77 
78  // Emit all the tasks that were accumulated in this scope
79  reaction_task.parent->reactor.emit(std::make_unique<commands::BehaviourTasks>(provider_type,
80  reaction_task.parent->id,
81  reaction_task.id,
82  false,
83  std::move(tasks)));
84  }
85 
86  template <typename T>
87  static void emit(NUClear::PowerPlant& powerplant,
88  const std::shared_ptr<T>& data,
89  const int priority,
90  const bool optional,
91  const std::string& name) {
92 
93  // Create the task object
94  auto task = commands::BehaviourTask(typeid(T), data, name, priority, optional);
95 
96  // If we are in a scope (running in a provider) accumulate the task to be emitted later
97  if (current_scope != nullptr) {
98  current_scope->tasks.push_back(std::move(task));
99  }
100  /// Root tasks emit the task immediately as a single pack
101  else {
102  // Get the current reaction
103  auto current_task = NUClear::threading::ReactionTask::get_current_task();
104  auto reaction_id = current_task != nullptr ? current_task->parent->id : 0;
105  auto task_id = current_task != nullptr ? current_task->id : 0;
106 
107  powerplant.emit(std::make_unique<commands::BehaviourTasks>(typeid(commands::RootProvider<T>),
108  reaction_id,
109  task_id,
110  true,
111  std::vector<commands::BehaviourTask>{task}));
112  }
113  }
114 
115  private:
116  /// The type of the provider that this scope is for
117  std::type_index provider_type;
118  /// The reaction task object this scope is for
119  NUClear::threading::ReactionTask& reaction_task;
120  /// The previous scope that was active before this one so it can be restored
121  /// This should only not be nullptr if somehow an Inline emit forces another provider to run while still in
122  /// another provider
123  ProviderScope* previous_scope = nullptr;
124  /// The tasks that have been accumulated in this scope
125  std::vector<commands::BehaviourTask> tasks;
126 
127  /// Holds the scope object which is currently active
128  static thread_local ProviderScope* current_scope;
129  };
