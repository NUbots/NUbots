147  /// @brief Config for the foot down detector
148  struct FootDown {
149  FootDown() = default;
150  FootDown(const FootDownMethod& method, const std::map<FootDownMethod, float>& thresholds) {
151  set_method(method, thresholds);
152  }
153  void set_method(const FootDownMethod& method, const std::map<FootDownMethod, float>& thresholds) {
154  if (thresholds.count(method) == 0) {
155  throw std::runtime_error(fmt::format("Invalid foot down method '{}'", std::string(method)));
156  }
157  current_method = method;
158  certainty_thresholds = thresholds;
159  }
160  [[nodiscard]] float threshold() const {
161  return certainty_thresholds.at(current_method);
162  }
163  [[nodiscard]] FootDownMethod method() const {
164  return current_method;
165  }
166  FootDownMethod current_method = FootDownMethod::Z_HEIGHT;
167  std::map<FootDownMethod, float> certainty_thresholds = {
168  {FootDownMethod::Z_HEIGHT, 0.01f},
169  {FootDownMethod::LOAD, 0.05f},
170  {FootDownMethod::FSR, 60.0f},
171  };
172  } footDown;
