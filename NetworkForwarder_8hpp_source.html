51  on<Trigger<T>, Buffer<10>, Priority::LOW>()

55  using namespace std::chrono; // NOLINT(google-build-using-namespace) fine in function scope
56  int id = id::get(*msg);
57 
58  auto now = NUClear::clock::now();
59 
60  // Loop through all the targets that are interested in this type
61  for (auto& target_handle : handle->targets) {
62  const auto& target = target_handle.first;
63  auto& h = target_handle.second;
64 
65  if (h.last_message.count(id) == 0
66  || duration_cast<duration<double>>(now - h.last_message[id]).count() > h.period) {
67  powerplant.emit_shared<Scope::NETWORK>(std::move(msg), target, false);
68  h.last_message[id] = now;
69  log<NUClear::TRACE>("Forwarding", type, "to", target);
70  }
