43  /**
44  * @brief Registers an emitter for a specific message type `T`.
45  *
46  * This function will add a callback function for the message type `T` to the decoder such that the decoder
47  * will emit the message when it is iterated over.
48  *
49  * @tparam T The type of message to register the emitter for.
50  */
51  template <typename T>
52  void register_emitter() {
53  uint64_t hash = NUClear::util::serialise::Serialise<T>::hash();
54  emitters[hash] = [this](utility::nbs::Decoder& decoder) {
55  decoder.on<T>([this](const T& msg) { emit(std::make_unique<T>(msg)); });
56  };
57  }
