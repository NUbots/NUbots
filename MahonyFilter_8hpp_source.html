37  template <typename Scalar>
38  class MahonyFilter {
39  private:
40  /// @brief Proportional gain
41  Scalar Kp = 0;
42 
43  /// @brief Integral gain
44  Scalar Ki = 0;
45 
46  /// @brief Gyroscope bias
47  Eigen::Matrix<Scalar, 3, 1> bias = Eigen::Matrix<Scalar, 3, 1>::Zero();
48 
49  /// @brief Rotation matrix from world to body frame of IMU
50  Eigen::Matrix<Scalar, 3, 3> Rwb = Eigen::Matrix<Scalar, 3, 3>::Identity();
51 
52  /// @brief Skew function
53  static Eigen::Matrix<Scalar, 3, 3> skew(const Eigen::Matrix<Scalar, 3, 1>& v) {
54  Eigen::Matrix<Scalar, 3, 3> skew;
55  skew << 0, -v(2), v(1), v(2), 0, -v(0), -v(1), v(0), 0;
56  return skew;
57  }
58 
59  public:
60  // Constructor
61  MahonyFilter(const Scalar Kp = 0,
62  const Scalar Ki = 0,
63  const Eigen::Matrix<Scalar, 3, 1>& bias = Eigen::Matrix<Scalar, 3, 1>::Zero(),
64  const Eigen::Matrix<Scalar, 3, 3>& Rwb = Eigen::Matrix<Scalar, 3, 3>::Identity())
65  : Kp(Kp), Ki(Ki), bias(bias), Rwb(Rwb) {}
66 
67 
68  /// @brief Update function
69  Eigen::Matrix<Scalar, 3, 3> update(const Eigen::Matrix<Scalar, 3, 1>& acc,
70  const Eigen::Matrix<Scalar, 3, 1>& gyro,
71  const Scalar dt) {
72  // Normalize the accelerometer reading
73  Eigen::Matrix<Scalar, 3, 1> uGBb = acc.normalized();
74 
75  // Invert the rotation matrix to get the body-to-world transformation
76  Eigen::Matrix<Scalar, 3, 3> Rbw = Rwb.transpose();
77 
78  // Rotate the world gravity vector in the world frame into the body frame
79  Eigen::Matrix<Scalar, 3, 1> est_uGBb = Rbw * Eigen::Matrix<Scalar, 3, 1>::UnitZ();
80 
81  // Calculate the error between the measured and estimated acceleration vectors
82  Eigen::Matrix<Scalar, 3, 3> a_corr = uGBb * est_uGBb.transpose() - est_uGBb * uGBb.transpose();
83  Eigen::Matrix<Scalar, 3, 1> omega_mes =
84  -1 * Eigen::Matrix<Scalar, 3, 1>(a_corr(2, 1), a_corr(0, 2), a_corr(1, 0));
85 
86  // Integrate the error vector to estimate the gyro bias
87  bias += Ki * omega_mes * dt;
88 
89  // Depolarizing the gyroscope bias
90  Eigen::Matrix<Scalar, 3, 1> l_omega = gyro + Kp * omega_mes + bias;
91 
92  // Find the quaternions rate of change
93  Eigen::Matrix<Scalar, 4, 4> ome = Eigen::Matrix<Scalar, 4, 4>::Zero();
94  ome.block(0, 0, 3, 3) = -skew(l_omega);
95  ome.block(3, 0, 1, 3) = -l_omega.transpose();
96  ome.block(0, 3, 3, 1) = l_omega;
97  // Calculate integral to find the attitude quaternion
98  Eigen::Matrix<Scalar, 4, 1> quat = Eigen::Quaternion<Scalar>(Rwb).coeffs();
99  quat += dt * Eigen::Matrix<Scalar, 4, 1>(0.5 * ome * quat);
100 
101  // Set the rotation matrix
102  Rwb = Eigen::Quaternion<Scalar>(quat.normalized()).toRotationMatrix();
103  return Rwb;
104  }
105 
106  /// @brief Get the rotation matrix
107  Eigen::Matrix<Scalar, 3, 3> get_state() const {
108  return Rwb;
109  }
110 
111  /// @brief Set the rotation matrix
112  void set_state(const Eigen::Matrix<Scalar, 3, 3>& Rwb) {
113  this->Rwb = Rwb;
114  }
115 
116  /// @brief Set the proportional gain
117  void set_Kp(const Scalar new_Kp) {
118  this->Kp = new_Kp;
119  }
120 
121  /// @brief Get the current proportional gain
122  Scalar get_Kp() const {
123  return Kp;
124  }
125 
126  /// @brief Set the integral gain
127  void set_Ki(const Scalar new_Ki) {
128  this->Ki = new_Ki;
129  }
130 
131  /// @brief Get the current integral gain
132  Scalar get_Ki() const {
133  return Ki;
134  }
135  };
