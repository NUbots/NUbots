42  /// @brief Update function
43  Eigen::Matrix<Scalar, 3, 3> update(const Eigen::Matrix<Scalar, 3, 1>& acc,
44  const Eigen::Matrix<Scalar, 3, 1>& gyro,
45  const Scalar dt) {
46  // Normalize the accelerometer reading
47  Eigen::Matrix<Scalar, 3, 1> uGBb = acc.normalized();
48 
49  // Invert the rotation matrix to get the body-to-world transformation
50  Eigen::Matrix<Scalar, 3, 3> Rbw = Rwb.transpose();
51 
52  // Rotate the world gravity vector in the world frame into the body frame
53  Eigen::Matrix<Scalar, 3, 1> est_uGBb = Rbw * Eigen::Matrix<Scalar, 3, 1>::UnitZ();
54 
55  // Calculate the error between the measured and estimated acceleration vectors
56  Eigen::Matrix<Scalar, 3, 3> a_corr = uGBb * est_uGBb.transpose() - est_uGBb * uGBb.transpose();
57  Eigen::Matrix<Scalar, 3, 1> omega_mes =
58  -1 * Eigen::Matrix<Scalar, 3, 1>(a_corr(2, 1), a_corr(0, 2), a_corr(1, 0));
59 
60  // Integrate the error vector to estimate the gyro bias
61  bias += Ki * omega_mes * dt;
62 
63  // Depolarizing the gyroscope bias
64  Eigen::Matrix<Scalar, 3, 1> l_omega = gyro + Kp * omega_mes + bias;
65 
66  // Find the quaternions rate of change
67  Eigen::Matrix<Scalar, 4, 4> ome = Eigen::Matrix<Scalar, 4, 4>::Zero();
68  ome.block(0, 0, 3, 3) = -skew(l_omega);
69  ome.block(3, 0, 1, 3) = -l_omega.transpose();
70  ome.block(0, 3, 3, 1) = l_omega;
71  // Calculate integral to find the attitude quaternion
72  Eigen::Matrix<Scalar, 4, 1> quat = Eigen::Quaternion<Scalar>(Rwb).coeffs();
73  quat += dt * Eigen::Matrix<Scalar, 4, 1>(0.5 * ome * quat);
74 
75  // Set the rotation matrix
76  Rwb = Eigen::Quaternion<Scalar>(quat.normalized()).toRotationMatrix();
77  return Rwb;
78  }
