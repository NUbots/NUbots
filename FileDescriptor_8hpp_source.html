34  /**
35  * @brief An RAII file descriptor.
36  * @details This class represents an RAII file descriptor.
37  * It will close the file descriptor it holds on
38  * destruction.
39  */
40  class FileDescriptor {
41  public:
42  /**
43  * @brief Constructs a new RAII file descriptor.
44  *
45  * @param fd [description]
46  */
47  FileDescriptor();
48  FileDescriptor(const int& fd_, std::function<void(int)> cleanup_ = nullptr);
49 
50  // Don't allow copy construction or assignment
51  FileDescriptor(const FileDescriptor&) = delete;
52  FileDescriptor& operator=(const FileDescriptor&) = delete;
53 
54  // Allow move construction or assignment
55  FileDescriptor(FileDescriptor&& rhs) noexcept;
56  FileDescriptor& operator=(FileDescriptor&& rhs) noexcept;
57 
58  /**
59  * @brief Destruct the file descriptor, closes the held fd
60  */
61  ~FileDescriptor();
62 
63  /**
64  * @brief Get the currently held file descriptor
65  *
66  * @return the file descriptor
67  */
68  // No Lint: As we are giving access to a variable which can change state.
69  // NOLINTNEXTLINE(readability-make-member-function-const)
70  [[nodiscard]] int get();
71 
72  /**
73  * @brief Returns if the currently held file descriptor is valid
74  *
75  * @return true if the file descriptor is valid
76  * @return false if the file descriptor is invalid
77  */
78  [[nodiscard]] bool valid() const;
79 
80  /**
81  * @brief Close the currently held file descriptor
82  */
83  void close();
84 
85  /**
86  * @brief Return the number of bytes available for reading
87  *
88  * @return -2 if no current connection, -1 for some other error. Check errno if -1 is returned.
89  */
90  [[nodiscard]] int available() const;
91 
92  private:
93  /// @brief The held file descriptor
94  int fd{-1};
95  /// @brief An optional cleanup function to call on close
96  std::function<void(int)> cleanup;
97  };
