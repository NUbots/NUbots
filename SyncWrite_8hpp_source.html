13  /**
14  * @brief This struct mimics the expected data structure for a Sync Write command.
15  *
16  * @details
17  * This type has it's members arranged in the same way as a raw array of this command would. Because of this
18  * you cannot add or remove members from this type. The template argument and parameter allows you to read any
19  * type. For example if you read with a uint16_t then it will read two bytes to the device. And if you use a
20  * struct with 3 uint16_t's in it, then you can directly read a parameter with an x, y and z bytes (e.g. the
21  * accelerometer)
22  * @tparam T the type of data to be written
23  * @tparam N the number of devices to read from
24  *
25  * @author Alex Biddulph
26  */
27  template <typename T>
28  struct SyncWriteData {
29  static_assert(std::is_trivial<T>::value && std::is_standard_layout<T>::value, "Values must be trivial data");
30  SyncWriteData() : id(0), data() {}
31  SyncWriteData(uint8_t id, T data) : id(id), data(data) {}

37  } __attribute__((packed)); // Make it so that the compiler reads this struct "as is" (no padding bytes)
