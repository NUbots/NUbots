35  template <typename Scalar>
36  class OccupancyMap {
37 
38  public:
39  /// @brief Getter for the map data
40  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> get_map() {
41  return map;
42  }
43 
44  /// @brief Get the width (number of columns) of the map
45  int get_width() {
46  return map.cols();
47  }
48 
49  /// @brief Get the length (number of rows) of the map
50  int get_length() {
51  return map.rows();
52  }
53 
54  /// @brief Get the value of the map at the specified coordinates or return -1 if the coordinates are outside
55  /// the map bounds
56  /// @param x The x-coordinate of the map
57  /// @param y The y-coordinate of the map
58  /// @return The value of the map at the specified coordinates
59  Scalar get_occupancy_value(int x, int y) {
60  // Check if the observation is within the map
61  if (x < 0 || x >= get_length() || y < 0 || y >= get_width()) {
62  return -1;
63  }
64  else {
65  return map(x, y);
66  }
67  }

79  /**
80  * @brief Add a horizontal line to the map
81  * @param X0 The x-coordinate of the origin of the line
82  * @param Y0 The y-coordinate of the origin of the line
83  * @param length The length of the line
84  * @param width The width of the line
85  */
86  void add_horizontal_line(int x0, int y0, int length) {

90  /**
91  * @brief Add a vertical line to the map
92  * @param X0 The x-coordinate of the origin of the line
93  * @param Y0 The y-coordinate of the origin of the line
94  * @param length The length of the line
95  * @param width The width of the line
96  */
97  void add_vertical_line(int x0, int y0, int length) {
98  map.block(y0, x0, length, 1).setOnes();
99  }

101  /**
102  * @brief Add a rectangle to the map with width of 1 grid cell
103  * @param X0 The x-coordinate of the origin of the rectangle
104  * @param Y0 The y-coordinate of the origin of the rectangle
105  * @param width The width of the rectangle
106  * @param height The height of the rectangle
107  */
108  void add_rectangle(int x0, int y0, int width, int height) {
109  // Add the top and bottom lines
110  add_horizontal_line(x0, y0, width);
111  add_horizontal_line(x0, y0 + height - 1, width);
112  // Add the left and right lines
113  add_vertical_line(x0, y0, height);
114  add_vertical_line(x0 + width - 1, y0, height);
115  }
116 
117  /**
118  * @brief Add a rectangle to the map with specified inner width
119  * @param X0 The x-coordinate of the origin of the rectangle
120  * @param Y0 The y-coordinate of the origin of the rectangle
121  * @param width The width of the rectangle
122  * @param height The height of the rectangle
123  * @param inner_width The width of the inner rectangle
124  */
125  void add_rectangle(int x0, int y0, int width, int height, int inner_width) {
126  for (int i = 0; i < inner_width; i++) {
127  int new_width = width - (2 * i);
128  int new_height = height - (2 * i);
129  if (new_width > 0 && new_height > 0) {
130  add_rectangle(x0 + i, y0 + i, new_width, new_height);
131  }
132  }
133  }

135  /**
136  * @brief Add a circle to the map with specified origin, radius and line width
137  * @param origin The origin (x,y) of the circle
138  * @param radius The radius of the circle
139  * @param inner_width The width of the circle line
140  */
141  void add_circle(const int X0, const int Y0, const int radius, const int inner_width) {
142  for (int y = Y0 - radius; y <= Y0 + radius; y++) {
143  for (int x = X0 - radius; x <= X0 + radius; x++) {
144  if (std::pow(x - X0, 2) + std::pow(y - Y0, 2) <= std::pow(radius, 2)) {
145  int dist = std::sqrt(std::pow(x - X0, 2) + std::pow(y - Y0, 2));
146  if (dist >= radius - inner_width && dist <= radius) {
147  if (x >= 0 && x < map.cols() && y >= 0 && y < map.rows()) {
148  map(y, x) = 1;
149  }
150  }
151  }
152  }
153  }
154  }
155 
156  /**
157  * @brief Add a rectangular cross to the map with specified origin, radius and line width
158  * @param origin The origin (x,y) of the circle
159  * @param radius The radius of the circle
160  * @param inner_width The width of the circle line
161  */
162  void add_cross(const int X0, const int Y0, const int radius, const int inner_width) {
163  for (int i = Y0 - radius; i <= Y0 + radius; i++) {
164  if (i >= 0 && i < map.rows()) {
165  int width = inner_width / 2;
166  int start_x = X0 - width;
167  int end_x = X0 + width;
168  if (start_x >= 0 && end_x < map.cols()) {
169  map.block(i, start_x, 1, end_x - start_x + 1) =
170  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Ones(1, end_x - start_x + 1);
171  }
172  }
173  }
174 
175  for (int i = X0 - radius; i <= X0 + radius; i++) {
176  if (i >= 0 && i < map.cols()) {
177  int width = inner_width / 2;
178  int start_y = Y0 - width;
179  int end_y = Y0 + width;
180  if (start_y >= 0 && end_y < map.rows()) {
181  map.block(start_y, i, end_y - start_y + 1, 1) =
182  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Ones(end_y - start_y + 1, 1);
183  }
184  }
185  }
186  }

188  /**
189  * @brief Replace map with a new map with occupancy values which encode the minimum distance to the
190  * closest occupied cell
191  * @param map The pre-filled Map with occupancy values of 0 or 1
192  * @param grid_size The size of the grid cells
193  */
194  void create_distance_map(Scalar grid_size) {
195  // Initialize the distance map with infinite distances
196  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> dist_map =
197  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Constant(map.rows(),
198  map.cols(),
199  std::numeric_limits<float>::infinity());
200 
201  // Create a queue to hold the cells to be visited
202  std::queue<std::pair<int, int>> q;
203 
204  // Mark all occupied cells as visited and set their distances to 0
205  for (int y = 0; y < map.rows(); y++) {
206  for (int x = 0; x < map.cols(); x++) {
207  if (map(y, x) == 1) {
208  dist_map(y, x) = 0;
209  q.push(std::make_pair(y, x));
210  }
211  }
212  }
213 
214  // Run BFS algorithm to compute the distances of all unoccupied cells to the nearest occupied cell
215  while (!q.empty()) {
216  std::pair<int, int> curr = q.front();
217  q.pop();
218  int y = curr.first;
219  int x = curr.second;
220 
221  // Update the distances of the neighbors of the current cell
222  for (int j = -1; j <= 1; j++) {
223  for (int i = -1; i <= 1; i++) {
224  // Skip the current cell
225  if (i == 0 && j == 0) {
226  continue;
227  }
228 
229  int new_y = y + j;
230  int new_x = x + i;
231 
232  // Check if the neighbor is within the map boundaries
233  if (new_y >= 0 && new_y < map.rows() && new_x >= 0 && new_x < map.cols()) {
234  float weight = std::sqrt(std::pow(i, 2) + std::pow(j, 2));
235 
236  // Check if the neighbor is unoccupied and has a shorter distance through the current cell
237  if (map(new_y, new_x) == 0 && dist_map(new_y, new_x) > dist_map(y, x) + weight) {
238  dist_map(new_y, new_x) = dist_map(y, x) + weight;
239  q.push(std::make_pair(new_y, new_x));
240  }
241  }
242  }
243  }
244  }
245 
246  // Replace the original map with the distance map
247  map = dist_map * grid_size;
248  }
249 
250  private:
251  /// @brief Eigen matrix which stores the map data
252  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> map;
253  };
