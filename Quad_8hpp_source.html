38  template <typename Scalar, int R, int C>
39  class Quad {
40  public:
41  using T = typename Eigen::Matrix<Scalar, R, C>;
42  Quad() : bl(T::Zero()), br(T::Zero()), tr(T::Zero()), tl(T::Zero()) {}
43  Quad(const Quad& other) : bl(other.bl), br(other.br), tr(other.tr), tl(other.tl) {}
44  Quad(const Quad&& other) noexcept : bl(other.bl), br(other.br), tr(other.tr), tl(other.tl) {}
45  [[nodiscard]] Quad& operator=(const Quad& other) {
46  bl = other.bl;
47  br = other.br;
48  tr = other.tr;
49  tl = other.tl;
50  }
51  [[nodiscard]] Quad& operator=(const Quad&& other) noexcept {
52  bl = other.bl;
53  br = other.br;
54  tr = other.tr;
55  tl = other.tl;
56  }
57  Quad(T bottomLeft, T topLeft, T topRight, T bottomRight)
58  : bl(std::move(bottomLeft)), br(std::move(bottomRight)), tr(std::move(topRight)), tl(std::move(topLeft)) {}
59  Quad(const Scalar& left, const Scalar& top, const Scalar& right, const Scalar& bottom)
60  : bl(left, bottom), br(right, bottom), tr(right, top), tl(left, top) {}

62  /**
63  * Sets the Quad as a screen aligned rectangle given the specified positions.
64  * @param left The left x pixel value.
65  * @param top The top y pixel value.
66  * @param right The right x pixel value.
67  * @param bottom The bottom y pixel value.
68  */
69  void set(const Scalar& left, const Scalar& top, const Scalar& right, const Scalar& bottom) {
70  bl.x() = left;

94  //! Returns the bottom centre pixel location of the Quad.
95  [[nodiscard]] T getTopCentre() const {
96  return ((tl + tr) * 0.5);

99  [[nodiscard]] T getBottomCentre() const {

131  // Returns the bounding box width and height
132  [[nodiscard]] T getSize() const {
133  Quad boundingBox = getBoundingBox(getVertices());

138  return (0.5 * (bl.x() + tl.x()));

160  //! Returns the average width of the Quad in pixels.
161  [[nodiscard]] Scalar getAverageWidth() const {
162  return ((0.5 * ((br - bl).norm() + (tr - tl).norm())));
163  }
164 
165  //! Returns the average height of the Quad in pixels.
166  [[nodiscard]] Scalar getAverageHeight() const {
167  return ((0.5 * ((br - tr).norm() + (bl - tl).norm())));
168  }

170  [[nodiscard]] Scalar area() const {
171  // Area of a quadrilateral: A = 0.5 * |diag1 X diag2|
172  // In two dimensions, this equates to: A = 0.5 * |(diag1.x)(diag2.y) - (diag2.x)(diag2.y)|
173  T diag1 = bl - tr;
174  T diag2 = tl - br;
175  return std::abs(0.5 * ((diag1.x() * diag2.y()) - (diag1.y() * diag2.x())));
176  }
177  [[nodiscard]] Scalar aspectRatio() const {
178  return (((br - tr).norm() + (bl - tl).norm() + 2) / ((br - bl).norm() + (tr - tl).norm() + 2));
179  }

181  [[nodiscard]] std::vector<T> getVertices() const {
182  std::vector<T> vert = {tr, br, bl, tl};
183  return vert;
184  }

186  [[nodiscard]] bool overlapsHorizontally(const Quad& other) const {
187  // Rough for now.
188  Scalar farRight = std::max(tr.x(), br.x());
189  Scalar farLeft = std::min(tl.x(), bl.x());
190  Scalar o_farRight = std::max(other.tr.x(), other.br.x());
191  Scalar o_farLeft = std::min(other.tl.x(), other.bl.x());
192 
193  return !((farRight < o_farLeft) || (o_farRight < farLeft));
194  }
195 
196  [[nodiscard]] bool checkCornersValid() const {
197  return br.innerSize() == 2 && bl.innerSize() == 2 && tr.innerSize() == 2 && tl.innerSize() == 2;
198  }

200  static Quad getBoundingBox(const std::vector<T>& points) {
201  // Check for
202  if (points.empty()) {
203  throw std::domain_error("Request made for bounding box for empty list of points!");
204  }
205 
206  Scalar min_x = points[0].x();
207  Scalar max_x = points[0].x();
208  Scalar min_y = points[0].y();
209  Scalar max_y = points[0].y();
210  for (uint i = 1; i < points.size(); i++) {
211  auto& p = points[i];
212  max_x = std::max(max_x, p.x());
213  min_x = std::min(min_x, p.x());
214  max_y = std::max(max_y, p.y());
215  min_y = std::min(min_y, p.y());
216  }
217  return Quad(T(min_x, min_y), T(min_x, max_y), T(max_x, max_y), T(max_x, min_y));
218  }
