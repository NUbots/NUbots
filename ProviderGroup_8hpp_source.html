46  struct WatchHandle {
47  WatchHandle(const std::function<void()>& deleter_) : deleter(deleter_) {}
48 
49  WatchHandle(WatchHandle&& other) {
50  deleter = std::exchange(other.deleter, std::function<void()>());
51  }
52 
53  WatchHandle& operator=(WatchHandle&& other) {
54  if (this != &other) {
55  deleter = std::exchange(other.deleter, std::function<void()>());
56  }
57  return *this;
58  }
59 
60  WatchHandle(const WatchHandle&) = delete;
61  WatchHandle& operator=(const WatchHandle&) = delete;
62  ~WatchHandle() {
63  if (deleter) {
64  deleter();
65  }
66  }
67 
68  std::function<void()> deleter;
69  };

71  std::shared_ptr<WatchHandle> add_watcher(const std::shared_ptr<DirectorTask>& task) {
72  watchers.push_back(task);
73 
74  return std::make_shared<WatchHandle>([this, task] {
75  auto it = std::find(watchers.begin(), watchers.end(), task);
76  if (it != watchers.end()) {
77  watchers.erase(it);
78  }
79  });
80  }
