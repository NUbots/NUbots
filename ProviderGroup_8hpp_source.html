38  struct WatchHandle {
39  WatchHandle(const std::function<void()>& deleter_) : deleter(deleter_) {}
40 
41  WatchHandle(WatchHandle&& other) {
42  deleter = std::exchange(other.deleter, std::function<void()>());
43  }
44 
45  WatchHandle& operator=(WatchHandle&& other) {
46  if (this != &other) {
47  deleter = std::exchange(other.deleter, std::function<void()>());
48  }
49  return *this;
50  }
51 
52  WatchHandle(const WatchHandle&) = delete;
53  WatchHandle& operator=(const WatchHandle&) = delete;
54  ~WatchHandle() {
55  if (deleter) {
56  deleter();
57  }
58  }
59 
60  std::function<void()> deleter;
61  };

63  std::shared_ptr<WatchHandle> add_watcher(const std::shared_ptr<DirectorTask>& task) {
64  watchers.push_back(task);
65 
66  return std::make_shared<WatchHandle>([this, task] {
67  auto it = std::find(watchers.begin(), watchers.end(), task);
68  if (it != watchers.end()) {
69  watchers.erase(it);
70  }
71  });
72  }
