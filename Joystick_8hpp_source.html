33 /**
34  * Encapsulates all data relevant to a sampled joystick event.
35  */
36 class JoystickEvent {
37 private:
38  static constexpr unsigned char JS_EVENT_BUTTON{0x01}; // button pressed/released
39  static constexpr unsigned char JS_EVENT_AXIS{0x02}; // joystick moved
40  static constexpr unsigned char JS_EVENT_INIT{0x80}; // initial state of device
41 
42 public:
43  JoystickEvent() = default;
44 
45  /**
46  * The timestamp of the event, in milliseconds.
47  */
48  uint32_t time = 0;
49 
50  /**
51  * The value associated with this joystick event.
52  * For buttons this will be either 1 (down) or 0 (up).
53  * For axes, this will range between -32768 and 32767.
54  */
55  int16_t value = 0;
56 
57  /**
58  * The event type.
59  */
60  uint8_t type = 0;
61 
62  /**
63  * The axis/button number.
64  */
65  uint8_t number = 0;
66 
67  /**
68  * Returns true if this event is the result of a button press.
69  */
70  [[nodiscard]] bool isButton() const {
71  return (type & JS_EVENT_BUTTON) != 0;
72  }
73 
74  /**
75  * Returns true if this event is the result of an axis movement.
76  */
77  [[nodiscard]] bool isAxis() const {
78  return (type & JS_EVENT_AXIS) != 0;
79  }
80 
81  /**
82  * Returns true if this event is part of the initial state obtained when
83  * the joystick is first connected to.
84  */
85  [[nodiscard]] bool isInitialState() const {
86  return (type & JS_EVENT_INIT) != 0;
87  }
88 };
