138  template <typename T, typename U>
139  std::string set_setting(T* setting, U&& value) {
140  GError* error = nullptr;
141 
142  // Get our current value
143  auto current = SettingsFunctions<T>::read(setting, &error);
144  if (error) {
145  std::string msg = fmt::format("Failed to read the current value: \"{}\"", error->message);
146  g_error_free(error);
147  throw std::runtime_error(msg);
148  }
149 
150  // Get the unit that it is measured in
151  std::string unit = SettingsFunctions<T>::unit(setting);
152 
153  // Check if this feature is locked
154  bool locked = arv_gc_feature_node_is_locked(reinterpret_cast<ArvGcFeatureNode*>(setting), &error);
155  if (error) {
156  g_error_free(error);
157  error = nullptr;
158  }
159  if (locked) {
160  throw std::runtime_error(
161  fmt::format("Cannot change from {0}{2} to {1}{2} as it is locked", current, value, unit));
162  }
163 
164  // If we are planning to change the value
165  if (value != current) {
166 
167  if (SettingsFunctions<T>::valid(setting, std::forward<U>(value))) {
168  SettingsFunctions<T>::write(setting, std::forward<U>(value), &error);
169  if (error) {
170  std::string msg = fmt::format("Failed changing from {0}{2} to {1}{2}: \"{3}\"",
171  current,
172  value,
173  unit,
174  error->message);
175  g_error_free(error);
176  throw std::runtime_error(msg);
177  }
178  return fmt::format("changed {0}{2} to {1}{2}", current, value, unit);
179  }
180  }
181  return "";
182  }
