63  /**
64  * Message used to tell the behaviour system about a new Provides reaction
65  */
66  struct ProvideReaction {
67 
68  /// A function that can be used to set data into the correct caches and return a Lock for when the task is
69  /// executed. This is provided as a function pointer rather than an std::function to ensure that it is stateless
70  /// for the given Provider type (ProviderGroup).
71  using DataSetter = Lock (*)(const NUClear::id_t&,
72  const RunReason&,
73  const std::shared_ptr<const void>&,
74  const std::shared_ptr<const GroupInfo>&);
75 
76  /**
77  * @brief Construct a new Provide Reaction object to send to the behaviour system
78  *
79  * @param reaction_ the reaction for this provides object
80  * @param type_ the type that this Provider provides for
81  * @param classification_ what kind of provider this ProvideReaction is for
82  * @param data_setter_ a function that will set the data for this Provider Group
83  */
84  ProvideReaction(std::shared_ptr<NUClear::threading::Reaction> reaction_,
85  const std::type_index& type_,
86  const ProviderClassification& classification_,
87  const DataSetter& data_setter_)
88  : reaction(std::move(reaction_)), type(type_), classification(classification_), data_setter(data_setter_) {}
89 
90  /// The reaction for this Provider
91  std::shared_ptr<NUClear::threading::Reaction> reaction;
92  /// The data type that this Provider services
93  std::type_index type;
94  /// The action type that this Provider is using
95  ProviderClassification classification;
96  /// A function that will place data into the correct caches and return a Lock
97  DataSetter data_setter;
98  };

100  /**
101  * Message used to tell the behaviour system about a new When condition on a Provider
102  */
103  struct WhenExpression {
104 
105  /**
106  * Constructs a new WhenExpression object to send to the behaviour system
107  *
108  * @param reaction_ the reaction this when condition is for
109  * @param type_ the enum type that this when condition is monitoring
110  * @param validator_ a function that will determine if the passed value satisfies this when condition
111  * @param current_ a function that will get the current state of the reaction
112  * @param binder_ a function that can be used to bind a reaction that monitors when the state changes
113  */
114  WhenExpression(std::shared_ptr<NUClear::threading::Reaction> reaction_,
115  const std::type_index& type_,
116  std::function<bool(const int&)> validator_,
117  std::function<int()> current_,
118  std::function<NUClear::threading::ReactionHandle(NUClear::Reactor&,
119  std::function<void(const int&)>)> binder_)
120  : reaction(std::move(reaction_))
121  , type(type_)
122  , validator(std::move(validator_))
123  , current(std::move(current_))
124  , binder(std::move(binder_)) {}
125 
126  /// The Provider reaction that is contingent on this when condition
127  std::shared_ptr<NUClear::threading::Reaction> reaction;
128  /// The enum type that this when expression is looking at
129  std::type_index type;
130  /// Function to determine if the passed state is valid
131  std::function<bool(const int&)> validator;
132  /// Function to get the current state from the global cache
133  std::function<int()> current;
134  /// Function to bind a reaction to monitor when this state changes
135  std::function<NUClear::threading::ReactionHandle(NUClear::Reactor&, std::function<void(const int&)>)> binder;
136  };

206  /**
207  * Construct a new Task object to send to the behaviour system

212  * @param priority_ the priority that this task is to run with
213  * @param optional_ whether this task is optional or not
214  */

223  std::type_index type;

225  std::shared_ptr<void> data;
226  /// A name for this task to be shown in debugging systems
227  std::string name;
228  /// The priority with which this task is to be executed
229  int priority;
230  /// If this task is optional

234  /**
235  * Represents a group of tasks that are emitted by a single provider and should be considered as siblings.
236  */
237  struct BehaviourTasks {
238  /**
239  * Construct a new Behaviour Tasks object to send to the behaviour system
240  *
241  * @param requester_type_ the type of the provider that emitted this task or RootProvider<T> for root tasks
242  * @param requester_id_ the id of the reaction that emitted this task
243  * @param requester_task_id_ the task_id of the reaction task that emitted this
244  * @param root_ if this task pack is a root task
245  * @param tasks_ the tasks that are to be executed
246  */
247  BehaviourTasks(const std::type_index& requester_type_,
248  const NUClear::id_t& requester_id_,
249  const NUClear::id_t& requester_task_id_,
250  const bool& root_,
251  std::vector<BehaviourTask>&& tasks_)
252  : requester_type(requester_type_)
253  , requester_reaction_id(requester_id_)
254  , requester_task_id(requester_task_id_)
255  , root(root_)
256  , tasks(std::move(tasks_)) {}
257 
258  /// The type of the requester, will be either the provider type or RootProvider<type> for root tasks
259  std::type_index requester_type;
260  /// The reaction id of the requester if it is a Provider, otherwise it will be 0 (for root tasks)
261  NUClear::id_t requester_reaction_id;
262  /// The reaction task id of the provider
263  NUClear::id_t requester_task_id;
264  /// If this task pack is a root task
265  bool root;
266  /// The tasks that are to be executed
267  std::vector<BehaviourTask> tasks;
268  };
