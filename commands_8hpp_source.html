85  /**
86  * Message used to tell the behaviour system about a new When condition on a Provider
87  */
88  struct WhenExpression {
89 
90  /**
91  * Constructs a new WhenExpression object to send to the behaviour system
92  *
93  * @param reaction_ the reaction this when condition is for
94  * @param type_ the enum type that this when condition is monitoring
95  * @param validator_ a function that will determine if the passed value satisfies this when condition
96  * @param current_ a function that will get the current state of the reaction
97  * @param binder_ a function that can be used to bind a reaction that monitors when the state changes
98  */
99  WhenExpression(std::shared_ptr<NUClear::threading::Reaction> reaction_,
100  const std::type_index& type_,
101  std::function<bool(const int&)> validator_,
102  std::function<int()> current_,
103  std::function<NUClear::threading::ReactionHandle(NUClear::Reactor&,
104  std::function<void(const int&)>)> binder_)
105  : reaction(std::move(reaction_))
106  , type(type_)
107  , validator(std::move(validator_))
108  , current(std::move(current_))
109  , binder(std::move(binder_)) {}
110 
111  /// The Provider reaction that is contingent on this when condition
112  std::shared_ptr<NUClear::threading::Reaction> reaction;
113  /// The enum type that this when expression is looking at
114  std::type_index type;
115  /// Function to determine if the passed state is valid
116  std::function<bool(const int&)> validator;
117  /// Function to get the current state from the global cache
118  std::function<int()> current;
119  /// Function to bind a reaction to monitor when this state changes
120  std::function<NUClear::threading::ReactionHandle(NUClear::Reactor&, std::function<void(const int&)>)> binder;
121  };

186  /**
187  * A task message that is to be sent to the behaviour system for execution
188  */
189  struct BehaviourTask {
190 
191  /**
192  * Construct a new Task object to send to the behaviour system
193  *
194  * @param type_ the type that this task is for
195  * @param root_type_ a secondary type to use if this is a root task
196  * @param requester_id_ the id of the reaction that emitted this task
197  * @param requester_task_id_ the task_id of the reaction task that emitted this
198  * @param data_ the task data to be sent to the provider
199  * @param name_ a string name for this task for use in debugging
200  * @param priority_ the priority that this task is to run with
201  * @param optional_ whether this task is optional or not
202  */
203  BehaviourTask(const std::type_index& type_,
204  const std::type_index& root_type_,
205  const uint64_t& requester_id_,
206  const uint64_t& requester_task_id_,
207  std::shared_ptr<void> data_,
208  std::string name_,
209  const int& priority_,
210  const bool& optional_)
211  : type(type_)
212  , root_type(root_type_)
213  , requester_id(requester_id_)
214  , requester_task_id(requester_task_id_)
215  , data(std::move(data_))
216  , name(std::move(name_))
217  , priority(priority_)
218  , optional(optional_) {}
219 
220  /// The Provider type this task is for
221  std::type_index type;
222  /// A secondary provider type to use if this is a root task
223  std::type_index root_type;
224  /// The reaction id of the requester (could be the id of a Provider)
225  uint64_t requester_id;
226  /// The reaction task id of the requester (if it is a Provider later a ProviderDone will be emitted)
227  uint64_t requester_task_id;
228  /// The data for the command, (the data that will be given to the Provider) if null counts as no task
229  std::shared_ptr<void> data;
230  /// A name for this task to be shown in debugging systems
231  std::string name;
232  /// The priority with which this task is to be executed
233  int priority;
234  /// If this task is optional
235  bool optional;
236  };
