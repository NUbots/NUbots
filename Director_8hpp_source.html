179  /**
180  * An object which holds the possible solutions to running a task.
181  * Each solution represents how a single task could be executed. As each provider group can have more than one
182  * provider, there can be more than one option for how to run the task. For each option that we can choose, they
183  * can themselves have their own requirements for provider groups they need to execute. Therefore each of these
184  * requirements is themselves a solution to another task that could be emitted.
185  */
186  struct Solution {
187 
188  /**
189  * A single option for how a provider could be executed
190  */
191  struct Option {
192  /// The state of this provider if it is blocked along with a description of why it is blocked
193  enum State {
194  /// This option is not blocked to us
195  OK,
196  /// This provider is blocked as the task given didn't have enough authority to run it
197  BLOCKED_PRIORITY,
198  /// This provider is blocked due to a when condition that cannot be met
199  BLOCKED_WHEN,
200  /// This provider is blocked as executing it would cause a loop of providers
201  BLOCKED_LOOP,
202  };
203 
204  /// The provider we are executing in this
205  std::shared_ptr<component::Provider> provider;
206  /// The set of solutions needed for this option to be executed
207  std::vector<Solution> requirements;
208  /// The state of this option, holds if it is blocked and if so why
209  State state;
210  };
211 
212  /// This value is true if this solution is a pushing solution. Meaning that the main solution cannot run but
213  /// running solution can enable a when condition to be met
214  bool pushed;
215  /// The list of options for this solution
216  std::vector<Option> options;
217  };

275  /**
276  * Represents a solution that we can run now, i.e. we can run all of the providers in this solution.
277  *
278  * Or if there is no solution that can be run now, it will return that it is blocked and which groups we wanted
279  * to run but weren't able to.

281  struct OkSolution {
282  OkSolution(const bool& blocked_, std::set<std::type_index>&& blocking_groups_)
283  : blocked(blocked_), blocking_groups(blocking_groups_) {}
284  OkSolution(const bool& blocked_,
285  const std::shared_ptr<component::Provider>& provider_,
286  std::vector<std::shared_ptr<component::Provider>>&& requirements_,
287  std::set<std::type_index>&& used_,
288  std::set<std::type_index>&& blocking_groups_)
289  : blocked(blocked_)
290  , provider(provider_)
291  , requirements(requirements_)
292  , used(used_)
293  , blocking_groups(blocking_groups_) {}
294 
295  /// If this entire path is blocked and unusable
296  bool blocked;
297  /// The main provider that this solution is for
298  std::shared_ptr<component::Provider> provider;
299  /// The list of providers that are needed for each of the requirements
300  std::vector<std::shared_ptr<component::Provider>> requirements;
301  /// Which groups have been used in this solution
302  std::set<std::type_index> used;
303  /// Groups which we wanted to use but were blocked to us
304  std::set<std::type_index> blocking_groups;
305  };

384  * @param run_level the level of execution that we are allowed to do

394  /**
395  * Looks at all the tasks that are in the pack and determines if they should run, and if so runs them.
396  *
397  * This function will ensure that if this pack is to be executed it can be executed as a whole. That means that
398  * all of the non optional tasks in it are able to run now given the state of the system. If the system is
399  * unable to run due to priority or due to a when condition not met it will not run any of the tasks in this
400  * pack, but enqueue them on the relevant providers.
401  *
402  * In the event that `When` conditions needs to be met by this pack and it has sufficient priority, it may also
403  * place this provider group into a "Pushing" state whereby it forces another provider group into a state where
404  * it will make the system reach the causing it requires.
405  *
406  * @param pack the task pack that represents the queued tasks
407  */
