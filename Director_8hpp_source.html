164  * that group. This will terminate once there are no more subtasks that are active in other groups or when

171  /**
172  * An object which holds the possible solutions to running a task.
173  * Each solution represents how a single task could be executed. As each provider group can have more than one
174  * provider, there can be more than one option for how to run the task. For each option that we can choose, they
175  * can themselves have their own requirements for provider groups they need to execute. Therefore each of these
176  * requirements is themselves a solution to another task that could be emitted.
177  */
178  struct Solution {
179 
180  /**
181  * A single option for how a provider could be executed
182  */
183  struct Option {
184  /// The state of this provider if it is blocked along with a description of why it is blocked
185  enum State {
186  /// This option is not blocked to us
187  OK,
188  /// This provider is blocked as the task given didn't have enough authority to run it
189  BLOCKED_PRIORITY,
190  /// This provider is blocked due to a when condition that cannot be met
191  BLOCKED_WHEN,
192  /// This provider is blocked as executing it would cause a loop of providers
193  BLOCKED_LOOP,
194  };
195 
196  /// The provider we are executing in this
197  std::shared_ptr<component::Provider> provider;
198  /// The set of solutions needed for this option to be executed
199  std::vector<Solution> requirements;
200  /// The state of this option, holds if it is blocked and if so why
201  State state;
202  };
203 
204  /// This value is true if this solution is a pushing solution. Meaning that the main solution cannot run but
205  /// running solution can enable a when condition to be met
206  bool pushed;
207  /// The list of options for this solution
208  std::vector<Option> options;
209  };

267  /**
268  * Represents a solution that we can run now, i.e. we can run all of the providers in this solution.
269  *
270  * Or if there is no solution that can be run now, it will return that it is blocked and which groups we wanted
271  * to run but weren't able to.
272  */
273  struct OkSolution {
274  OkSolution(const bool& blocked_, std::set<std::type_index>&& blocking_groups_)
275  : blocked(blocked_), blocking_groups(blocking_groups_) {}
276  OkSolution(const bool& blocked_,
277  const std::shared_ptr<component::Provider>& provider_,
278  std::vector<std::shared_ptr<component::Provider>>&& requirements_,
279  std::set<std::type_index>&& used_,
280  std::set<std::type_index>&& blocking_groups_)
281  : blocked(blocked_)
282  , provider(provider_)
283  , requirements(requirements_)
284  , used(used_)
285  , blocking_groups(blocking_groups_) {}
286 
287  /// If this entire path is blocked and unusable
288  bool blocked;
289  /// The main provider that this solution is for
290  std::shared_ptr<component::Provider> provider;
291  /// The list of providers that are needed for each of the requirements
292  std::vector<std::shared_ptr<component::Provider>> requirements;
293  /// Which groups have been used in this solution
294  std::set<std::type_index> used;
295  /// Groups which we wanted to use but were blocked to us
296  std::set<std::type_index> blocking_groups;
297  };

370  /**
371  * Looks at all the tasks that are in the pack and determines if they should run, and if so runs them.
372  *
373  * This function will ensure that if this pack is to be executed it can be executed as a whole. That means that
374  * all of the non optional tasks in it are able to run now given the state of the system. If the system is
375  * unable to run due to priority or due to a when condition not met it will not run any of the tasks in this
376  * pack, but enqueue them on the relevant providers.
377  *
378  * In the event that `When` conditions needs to be met by this pack and it has sufficient priority, it may also
379  * place this provider group into a "Pushing" state whereby it forces another provider group into a state where
380  * it will make the system reach the causing it requires.
381  *
382  * @param pack the task pack that represents the queued tasks
383  */
