36  class DataPlayback : public NUClear::Reactor {
37  private:
38  // FILE FORMAT
39  // TYPE DATA
40  // char[3] RADIATION SYMBOL {{ 0xE2, 0x98, 0xA2 }}
41  // uint32_t SIZE OF NEXT PACKET
42  // uint64_t TIMESTAMP DATA WAS EMITTED IN MICROSECONDS
43  // uint64_t DATA TYPE HASH

45  struct Player {
46  // The function that will queue this message to be emitted
47  std::function<void(const NUClear::clock::time_point&, const std::vector<uint8_t>&)> emit;
48  // If this player should execute and send messages
49  bool enabled = false;
50  };
51 
52  template <typename T>
53  void add_player() {
54  uint64_t hash = NUClear::util::serialise::Serialise<T>::hash();
55  Player p;
56  p.emit = [this](const NUClear::clock::time_point& emit_time, const std::vector<uint8_t>& payload) {
57  // Deserialise our type
58  auto msg = std::make_unique<T>(NUClear::util::serialise::Serialise<T>::deserialise(payload));
59 
60  // Emit it after the delay
61  emit<Scope::DELAY>(msg, emit_time - NUClear::clock::now());
62  };
63 
64  // Default to not emitting this type
65  p.enabled = false;
66 
67  // Add to our players list
68  players.insert(std::make_pair(hash, p));
69  }
