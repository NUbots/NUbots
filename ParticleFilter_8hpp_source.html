47  /**
48  * @author Alex Biddulph, Trent Houliston
49  *
50  * @details Uses a set of particles (or samples) to represent possible states in a system. Each particle is randomly
51  * selected from a multivariate normal distribution. Each particle is weighted by how likely it is to be the real
52  * state of the system. Particles with a higher weight are more likely to be resampled (with noise) during the time
53  * update.
54  *
55  * Time Update: Resamples all particles, perturbing the resampled particles with the model's process noise, and asks
56  * the model to predict what should happen to each particle after an elapsed amount of time.
57  *
58  * Measurement Update: Asks the model to predict what the next measurement will be (based on current state/particle)
59  * and the finds the difference between the model's prediction and the actual measurement. The likelihood of this
60  * prediction occuring, given the measurement, is then determined using a multivariate normal distribution
61  * probability density function.
62  *
63  * @tparam Scalar The scalar type to use for all calculations. This puts an upper bound on the numerical precision
64  * that is achievable by the filter
65  * @tparam FilterModel The model to use in the filtering process. This model needs to provide the equations that
66  * models the system.
67  */
68  template <typename Scalar, template <typename> class FilterModel>
69  class ParticleFilter {
70  public:
71  /// The possible resampling methods
72  /// @see http://users.isy.liu.se/rt/schon/Publications/HolSG2006.pdf for an explanation on each one
73  enum class ResampleMethod { MULTINOMIAL, STRATIFIED, SYSTEMATIC };
74  ResampleMethod resample_method = ResampleMethod::SYSTEMATIC;
75 
76  bool use_residual_resampling = true;

83  using StateMat = Eigen::Matrix<Scalar, Model::size, Model::size>;

85  /**
86  * @brief Constructs the particle filter using default values. Mean is set to 0 and covariance is set to
87  * \f$0.1 * I_{n}\f$, where n is the size of the model. The residual method is used for resampling particles and
88  * the residual will use the systematic method for resampling the residual particles.
89  */
90  ParticleFilter() {
91  set_state(StateVec::Zero(), StateMat::Identity() * 0.1);
92  }

94  /**
95  * @brief Constructs the particle filter with the provided values. The resampling method with default to the
96  * residual method with the systematic method for resampling the residual particles. It is an error to use the
97  * residual method for resampling residuals and an exception will be thrown.
98  *
99  * @param mean The mean of the multivariate normal distribution. Used for sampling the initial particles in the
100  * filter.
101  * @param covariance The covariance of the multivariate normal distribution. Used for sampling the initial
102  * particles in the filter.
103  * @param resample_method_ The method to use when resampling particles during the time update step
104  * @param use_residual_resampling_ Whether residual resampling should be used with the other method
105  */
106  ParticleFilter(const StateVec& mean,
107  const StateMat& covariance,
108  const ResampleMethod& resample_method_ = ResampleMethod::SYSTEMATIC,
109  const bool& use_residual_resampling_ = true)
110  : resample_method(resample_method_), use_residual_resampling(use_residual_resampling_) {
111  set_state(mean, covariance);
112  }

114  /**
115  * @brief Constructs the particle filter with the provided values. The resampling method with default to the
116  * residual method with the systematic method for resampling the residual particles. It is an error to use the
117  * residual method for resampling residuals and an exception will be thrown.
118  *
119  * This version initialises the particles using multiple initial hypotheses. The particle count is split evenly
120  * across all hypotheses. If the division between the number of hypotheses and the number of particles is not
121  * even, then the remaining particles will be treated as rogues. It is an error for the numbers of means and
122  * covariances to differ.
123  *
124  * @param mean The mean of the multivariate normal distribution. Used for sampling the initial particles in the
125  * filter. There is one mean for each initial hypothesis.
126  * @param covariance The covariance of the multivariate normal distribution. Used for sampling the initial
127  * particles in the filter. There is one covariance for each initial hypothesis.
128  * @param resample_method_ The method to use when resampling particles during the time update step.
129  * @param use_residual_resampling_ Whether residual resampling should be used with the other method
130  */
131  ParticleFilter(const std::vector<std::pair<StateVec, StateMat>>& hypotheses,
132  const ResampleMethod& resample_method_ = ResampleMethod::SYSTEMATIC,
133  const bool& use_residual_resampling_ = true)
134  : resample_method(resample_method_), use_residual_resampling(use_residual_resampling_) {
135 
136  set_state(hypotheses);
137  }

139  /**
140  * @brief Clears all particles in the filter and samples new particles from the multivariate normal
141  * distribution with the given mean and covariance. Sampled particles are limited (using model::limit)
142  * function to ensure all particles are valid. Particle weights are initialised to 1.
143  *
144  * @param mean The mean of the multivariate normal distribution.
145  * @param covariance The covariance of the multivariate normal distribution.
146  */
147  void set_state(const StateVec& mean, const StateMat& covariance) {
148  // Initialise all of the particles
149  init(mean, covariance);
150 
151  // Limit the state of each particle to ensure they are still valid
152  for (unsigned int i = 0; i < particles.cols(); i++) {
153  particles.col(i) = model.limit(particles.col(i));
154  }
155  }

157  /**
158  * @brief Clears all particles in the filter and samples new particles from the multivariate normal
159  * distribution with the given mean and covariance. Sampled particles are limited (using model::limit)
160  * function to ensure all particles are valid. Particle weights are initialised to 1.
161  *
162  * This version initialises the particles using multiple initial hypotheses. The particle count is split evenly
163  * across all hypotheses. If the division between the number of hypotheses and the number of particles is not
164  * even, then the remaining particles will be treated as rogues. It is an error for the numbers of means and
165  * covariances to differ.
166  *
167  * @param mean The mean of the multivariate normal distribution. There is one mean for each initial hypothesis.
168  * @param covariance The covariance of the multivariate normal distribution. There is one covariance for each
169  * initial hypothesis.
170  */
171  void set_state(const std::vector<std::pair<StateVec, StateMat>>& hypotheses) {
172  // Make sure we have sane inputs
173  if (hypotheses.empty()) {
174  throw std::runtime_error("ParticleFilter::set_state called with invalid data");
175  }
176 
177  // Initialise all of the particles
178  if (hypotheses.size() == 1) {
179  init(hypotheses[0].first, hypotheses[0].second);
180  }
181  else {
182  init(hypotheses);
183  }
184 
185  // Limit the state of each particle to ensure they are still valid
186  for (unsigned int i = 0; i < particles.cols(); i++) {
187  particles.col(i) = model.limit(particles.col(i));
188  }
189  }

191  private:
192  using ParticleList = Eigen::Matrix<Scalar, Model::size, Eigen::Dynamic>;
193  using ParticleWeights = std::vector<Scalar>;
194 
195  ParticleList particles{};
196  ParticleWeights weights{};
197 
198  /**
199  * @brief Clears all particles in the filter and samples new particles from the multivariate normal
200  * distribution with the given mean and covariance. Particle weights are initialised to 1.
201  *
202  * @param mean The mean of the multivariate normal distribution.
203  * @param covariance The covariance of the multivariate normal distribution.
204  */
205  void init(const StateVec& mean, const StateMat& covariance) {
206  // Make sure our particle list has the right shape
207  if (particles.cols() != model.n_particles + model.n_rogues || particles.rows() != Model::size) {
208  particles.resize(Model::size, model.n_particles + model.n_rogues);
209  weights.resize(model.n_particles + model.n_rogues);
210  }
211 
212  // Setup our multivariate normal distribution so we can initialise the particles
213  MultivariateNormal<Scalar, Model::size> multivariate(mean, covariance);
214 
215  // Sample a random vector from the multivariate distribution for each particle
216  for (int i = 0; i < model.n_particles; ++i) {
217  particles.col(i) = multivariate.sample();
218  }
219 
220  // Get the model to give us some rogues
221  for (int i = 0; i < model.n_rogues; ++i) {
222  particles.col(model.n_particles + i) = model.get_rogue();
223  }
224 
225  // Start all weights at 1
226  std::fill(weights.begin(), weights.end(), Scalar(1));
227  }
228 
229  /**
230  * @brief Clears all particles in the filter and samples new particles from the multivariate normal
231  * distribution with the given mean and covariance. Particle weights are initialised to 1.
232  *
233  * This version initialises the particles using multiple initial hypotheses. The particle count is split evenly
234  * across all hypotheses. If the division between the number of hypotheses and the number of particles is not
235  * even, then the remaining particles will be treated as rogues.
236  *
237  * @param mean The mean of the multivariate normal distribution.
238  * @param covariance The covariance of the multivariate normal distribution.
239  */
240  void init(const std::vector<std::pair<StateVec, StateMat>>& hypotheses) {
241  // Make sure our particle list has the right shape
242  if (particles.cols() != model.n_particles + model.n_rogues || particles.rows() != Model::size) {
243  particles.resize(Model::size, model.n_particles + model.n_rogues);
244  weights.resize(model.n_particles + model.n_rogues);
245  }
246 
247  // We want to evenly distribute the particles across all hypotheses
248  // If the distribution is not even then the remaining particles will be treated as extra rogues
249  const int particles_per_state = model.n_particles / hypotheses.size();
250 
251  // For each hypothesis sample a random vector from the multivariate distribution for each particle
252  for (int hypothesis = 0; hypothesis < int(hypotheses.size()); ++hypothesis) {
253  // Setup our multivariate normal distribution for this hypothesis so we can initialise the particles
254  MultivariateNormal<Scalar, Model::size> multivariate(hypotheses[hypothesis].first,
255  hypotheses[hypothesis].second);
256 
257  const int start_col = hypothesis * particles_per_state;
258  for (int i = 0; i < particles_per_state; ++i) {
259  particles.col(start_col + i) = multivariate.sample();
260  }
261  }
262 
263  // Get the model to give us some rogues
264  const int num_particles = particles_per_state * hypotheses.size();
265  const int num_rogues = model.n_particles - num_particles + model.n_rogues;
266  for (int i = 0; i < num_rogues; ++i) {
267  particles.col(num_particles + i) = model.get_rogue();
268  }
269 
270  // Start all weights at 1
271  std::fill(weights.begin(), weights.end(), Scalar(1));
272  }

274  /**
275  * @brief Use the chosen resample and residual method to resample the particles in the filter based on their
276  * weights.
277  */
278  std::vector<int> resample_particles() {
279 
280  // Select the resampling method to use
281  if (use_residual_resampling) {
282  // The residual method uses a secondary method when it resamples the residual particles
283  switch (resample_method) {
284  case ResampleMethod::MULTINOMIAL:
285  return stats::resample::residual(model.n_particles,
286  weights.begin(),
287  weights.end(),
288  stats::resample::multinomial<decltype(weights.begin())>);
289  case ResampleMethod::STRATIFIED:
290  return stats::resample::residual(model.n_particles,
291  weights.begin(),
292  weights.end(),
293  stats::resample::stratified<decltype(weights.begin())>);
294  case ResampleMethod::SYSTEMATIC:
295  return stats::resample::residual(model.n_particles,
296  weights.begin(),
297  weights.end(),
298  stats::resample::systematic<decltype(weights.begin())>);
299  }
300  }
301  switch (resample_method) {
302  case ResampleMethod::MULTINOMIAL:
303  return stats::resample::multinomial(model.n_particles, weights.begin(), weights.end());
304  case ResampleMethod::STRATIFIED:
305  return stats::resample::stratified(model.n_particles, weights.begin(), weights.end());
306  case ResampleMethod::SYSTEMATIC:
307  return stats::resample::systematic(model.n_particles, weights.begin(), weights.end());
308  }
309  // This shouldn't happen unless a new enum element is added and the switches aren't updated
310  throw std::runtime_error("Invalid setting for resample method.");
311  }

314  public:
315  /**
316  * @brief Do a weighted resampling of all of the particles in the filter. The model is also asked to provide new
317  * rogues. All particle weights are reset to 1.
318  */
319  void resample() {
320 
321  // Get indexes to the particles that are being resampled
322  // Some particles may be resampled multiple times
323  std::vector<int> idx = resample_particles();
324 
325  // Resample the particles
326  // For each index get the corresponding particle and perturb it with the model's process noise
327  ParticleList resampled_particles(Model::size, particles.cols());
328  for (int i = 0; i < int(idx.size()); ++i) {
329  // Create a new particle by using the selected particle as the mean + value from process noise
330  resampled_particles.col(i) = particles.col(idx[i]);
331  }
332 
333  // Get the model to give us some rogue particles
334  for (int i = 0; i < model.n_rogues; ++i) {
335  resampled_particles.col(model.n_particles + i) = model.get_rogue();
336  }
337 
338  // Update our particles with our resampled particles
339  for (int i = 0; i < particles.cols(); i++) {
340  particles.col(i) = resampled_particles.col(i);
341  }
342 
343  // Reset all weights to 1
344  std::fill(weights.begin(), weights.end(), Scalar(1));
345  }

347  /**
348  * @brief Resample all particles in the filter and set their weights back to 1. The model is asked to give a
349  * prediction on what happens to each particle after dt time has elapsed (using model::time).
350  *
351  * @tparam Args the types of any extra arguments that are provided to the model's time update function.
352  *
353  * @param dt The amount of time that has elapsed since the previous time update step.
354  * @param params Any extra arguments that are to be forwarded to the model's time update function.
355  */
356  template <typename... Args>
357  void time(const Scalar& dt, Args&&... params) {
358 
359  // Resample our particles
360  resample();
361 
362  // Make a multivariate normal distribution with zero mean and the model's process noise as the covariance
363  // This will be used to perturb the resampled particles
364  MultivariateNormal<Scalar, Model::size> multivariate(model.noise(dt));
365 
366  // Perturb each particle and get the model to apply a time update. Then limit particle to ensure it is valid
367  for (int i = 0; i < particles.cols(); ++i) {
368  particles.col(i) = model.limit(
369  model.time(particles.col(i) + multivariate.sample(), dt, std::forward<Args>(params)...));
370  }
371  }

373  /**
374  * @brief Get the model to predict (using model::predict) what the next measurement will be then determine how
375  * accurate this prediction is (using model::difference). The likelihood of the measurement is determined using
376  * the multivariate normal distribution probability density function, this likelihood (determined for each
377  * particle) re-weights each particle.
378  *
379  * @tparam MeasurementScalar The underlying numerical type of the measurement
380  * @tparam S The dimensionality of the measurement (and associated covariance matrix)
381  * @tparam VArgs The types of any extra arguments associated with constructing the measurement vector.
382  * @tparam MArgs The types of any extra arguments associated with constructing the measurement covariance
383  * matrix.
384  * @tparam Args the types of any extra arguments that are provided to the model's predict function.
385  *
386  * @param measurement The measurement (from sensors, etc).
387  * @param measurement_covariance The covariance of the measurement.
388  * @param params Any extra arguments that are to be forwarded to the model's predict function.
389  */
390  template <typename MeasurementScalar, int S, int... VArgs, int... MArgs, typename... Args>
391  MeasurementScalar measure(const Eigen::Matrix<MeasurementScalar, S, 1, VArgs...>& measurement,
392  const Eigen::Matrix<MeasurementScalar, S, S, MArgs...>& measurement_covariance,
393  Args&&... params) {
394 
395  // Get the model to make a prediction for each particle
396  // Also get the model to tell us how much a prediction deviates from the observed measurement
397  Eigen::Matrix<MeasurementScalar, S, Eigen::Dynamic> differences(S, particles.cols());
398  for (int i = 0; i < particles.cols(); ++i) {
399  differences.col(i) =
400  model.difference(model.predict(particles.col(i), std::forward<Args>(params)...), measurement);
401  }
402 
403  // Recalculate the weights of each particle
404  // The weight update factor comes from the density function of a zero mean, measurement_covariance
405  // covariance multivariate normal distribution
406  MultivariateNormal<Scalar, S> multivariate(measurement_covariance);
407  for (int i = 0; i < particles.cols(); ++i) {
408  weights[i] *= multivariate.density(differences.col(i));
409  }
410 
411  // Calculate log probabilities
412  Eigen::Matrix<MeasurementScalar, Eigen::Dynamic, 1> logits =
413  -differences.cwiseProduct(measurement_covariance.inverse() * differences).colwise().sum();
414 
415  // Subtract the max log probability for numerical stability and then exponentiate
416  logits = Eigen::exp(logits.array() - logits.maxCoeff());
417 
418  // Return the mean log probability
419  Scalar sum(0);
420  for (int i = 0; i < logits.size(); ++i) {
421  sum += logits[i];
422  }
423  return sum / Scalar(logits.size());
424  }
