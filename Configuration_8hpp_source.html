53  struct [[nodiscard]] Configuration {
54  // Rules:
55  // 1) Default config file should define a value for every node.
56  // 2) Platform config overrides default config values.
57  // 3) Per-robot config overrides default and platform config values. This file need only override the values
58  // that need to be overriden.
59  // 4) Per-robot-name config overrides default, platform and per-robot config values. This file need only
60  // override the values that need to be overriden.
61  // 5) Per-binary config overrides per-robot, platform and default config values. This
62  // file need only override the values that need to be overriden.
63  //
64  // Per-robot, per-robot-name, per-platform and per-binary files need not exist.
65  // Per-robot, per-robot-name, per-platform and per-binary files can add new nodes to the file, but this is
66  // probably unwise.
67  //
68  // We have to merge the YAML trees to account for situations where a sub-node is not defined in a higher
69  // priority tree.
70 
71  fs::path file_name{};
72  std::string hostname{};
73  std::string binary{};
74  std::string platform{};
75  YAML::Node config{};

85  /// @brief Constructor without config node given. The correct config file has to be deduced from the params
86  Configuration(const std::string& file_name,
87  const std::string& hostname,
88  const std::string& binary,
89  const std::string& platform)
90  : file_name(file_name), hostname(hostname), binary(binary), platform(platform) {
91  bool loaded = false;
92 
93  // Load the default config file.
94  if (fs::exists(fs::path("config") / file_name)) {
95  config = YAML::LoadFile(fs::path("config") / file_name);
96  loaded = true;
97  }
98 
99  // If the same file exists in this platform's per-platform config directory then load and merge
100  if (fs::exists(fs::path("config") / platform / file_name) && !platform.empty()) {
101  if (loaded) {
102  config = merge_yaml_nodes(config, YAML::LoadFile(fs::path("config") / platform / file_name));
103  }
104 
105  else {
106  config = YAML::LoadFile(fs::path("config") / platform / file_name);
107  loaded = true;
108  }
109  }
110 
111  // If the same file exists in this robots per-robot config directory then load and merge.
112  if ((platform != hostname) && fs::exists(fs::path("config") / hostname / file_name)) {
113  if (loaded) {
114  config = merge_yaml_nodes(config, YAML::LoadFile(fs::path("config") / hostname / file_name));
115  }
116 
117  else {
118  config = YAML::LoadFile(fs::path("config") / hostname / file_name);
119  loaded = true;
120  }
121  }
122 
123  // If the same file exists in this robot's per-robot-name config directory then load and merge.
124  auto robot_name = utility::platform::get_robot_alias(hostname);
125  if (!robot_name.empty()) {
126  if (fs::exists(fs::path("config") / robot_name / file_name)) {
127  if (loaded) {
128  config = merge_yaml_nodes(config, YAML::LoadFile(fs::path("config") / robot_name / file_name));
129  }
130 
131  else {
132  config = YAML::LoadFile(fs::path("config") / robot_name / file_name);
133  loaded = true;
134  }
135  }
136  }
137 
138  // If the same file exists in this binary's per-binary config directory then load and merge.
139  if (fs::exists(fs::path("config") / binary / file_name)) {
140  if (loaded) {
141  config = merge_yaml_nodes(config, YAML::LoadFile(fs::path("config") / binary / file_name));
142  }
143 
144  else {
145  config = YAML::LoadFile(fs::path("config") / binary / file_name);
146  }
147  }
148  }

150  [[nodiscard]] static YAML::Node merge_yaml_nodes(const YAML::Node& base, const YAML::Node& override) {
151  YAML::Node ret(base);
152 
153  for (const auto& item : override) {
154  const std::string& key = item.first.as<std::string>();
155 
156  // If the key doesn't exist in the base set then add it and move on.
157  if (!base[key]) {
158  ret[key] = item.second;
159  }
160 
161  // If the key does exist then we must go deeper.
162  else {
163  // The type dictates whether we need a recursive call or not.
164  switch (item.second.Type()) {
165  // Just a raw value (int, double, etc)
166  case YAML::NodeType::Scalar: {
167  ret[key] = item.second;
168  break;
169  }
170 
171  // Essentially a vector.
172  case YAML::NodeType::Sequence: {
173  ret[key] = item.second;
174  break;
175  }
176 
177  // Recurse.
178  case YAML::NodeType::Map: {
179  ret[key] = merge_yaml_nodes(base[key], override[key]);
180  break;
181  }
182 
183  // Its nothing, so overwrite (unset) the value.
184  // Is this really the intended behaviour?
185  case YAML::NodeType::Null: [[fallthrough]];
186  case YAML::NodeType::Undefined: [[fallthrough]];
187  default: {
188  NUClear::log<NUClear::LogLevel::WARN>("Unsetting key",
189  "'" + key + "'",
190  "in YAML file. Is this what you intended?");
191  ret[key] = item.second;
192  break;
193  }
194  }
195  }
196  }
197 
198  return ret;
199  }

201  static inline std::string get_platform(const std::string& hostname) {
202  // It is assumed that all hostnames are in the format <platform name><robot number>,
203  // such that the regular expression
204  // [a-z]+[0-9]+?
205  // will match all hostnames
206  std::regex re("([a-z]+)([0-9]+)?");
207  std::smatch match;
208 
209  if (std::regex_match(hostname, match, re)) {
210  // match[0] will be the full string
211  // match[1] the first match (platform name)
212  // match[2] the second match (robot number)
213  return match[1].str();
214  }
215 
216  // If platform cannot be found, return empty
217  return "";
218  }

310  /// @brief Sets up the NUClear Reaction for Configuration
311  /// @details Sets up a FileWatch on the file_name which is passed in
312  /// The config files are FileWatched in order from least specific to most specific:
313  /// 1. default, 2. platform, 3. per-robot, 4. per-robot-name, 5. binary
314  /// The later, more specific configs (if they exist) supersede the less specific ones
315  /// If the default config file doesn't exist, we make one
316  /// The flags used during binding tell FileWatch that the config files have been changed or
317  /// renamed, which tells FileWatch they should be updated
318  /// @see FileWatch
319  /// @tparam DSL Magic NUClear type. Ignore for the purpose of understanding this function
320  /// @param reaction The reaction we are binding which will watch the config path(s)
321  /// @param file_name The file_name of the desired config file
322  template <typename DSL>
323  static void bind(const std::shared_ptr<threading::Reaction>& reaction, const fs::path& file_name) {
324  auto flags = ::extension::FileWatch::RENAMED | ::extension::FileWatch::CHANGED;
325 
326  // Get hostname so we can find the correct per-robot config directory.
327  const std::string hostname = utility::support::get_hostname();
328  const std::string platform(::extension::Configuration::get_platform(hostname));
329  const std::string robot_name = utility::platform::get_robot_alias(hostname);
330 
331  // Check if there is a default config. If there isn't, try to make one
332  const fs::path default_config = fs::path("config") / file_name;
333  if (!fs::exists(default_config)) {
334  NUClear::log<NUClear::LogLevel::WARN>("Configuration file '" + default_config.string()
335  + "' does not exist. Creating it.");
336 
337  // Check for a directory.
338  if (fs::is_directory(default_config)) {
339  fs::create_directory(default_config);
340  }
341 
342  else {
343  std::ofstream ofs(default_config.string());
344  if (!ofs.is_open()) {
345  throw std::runtime_error("Failed creating file '" + file_name.string() + "'.");
346  }
347  ofs.close();
348  }
349  }
350 
351  // Bind our default config file path
352  DSLProxy<::extension::FileWatch>::bind<DSL>(reaction, default_config, flags);
353 
354  // Bind our robot name specific config file if it exists
355  const fs::path name_config = fs::path("config") / robot_name / file_name;
356  if (!robot_name.empty() && fs::exists(name_config)) {
357  DSLProxy<::extension::FileWatch>::bind<DSL>(reaction, name_config, flags);
358  }
359 
360  // Bind our robot specific config file if it exists
361  const fs::path robot_config = fs::path("config") / hostname / file_name;
362  if (fs::exists(robot_config)) {
363  DSLProxy<::extension::FileWatch>::bind<DSL>(reaction, robot_config, flags);
364  }
365 
366  // Bind our robot specific config file if it exists
367  const fs::path platform_config = fs::path("config") / platform / file_name;
368  if ((platform != hostname) && fs::exists(platform_config) && !platform.empty()) {
369  DSLProxy<::extension::FileWatch>::bind<DSL>(reaction, platform_config, flags);
370  }
371 
372  // If there were command line arguments, we can get the binary name, and check for a binary config
373  // If not, we don't bother checking for a binary config to bind
374  const auto binary_name = get_first_command_line_arg();
375  if (!binary_name.empty()) {
376  fs::path binary_config = fs::path("config") / binary_name / file_name;
377  // Bind our binary specific config file if it exists
378  if (fs::exists(binary_config)) {
379  DSLProxy<::extension::FileWatch>::bind<DSL>(reaction, binary_config, flags);
380  }
381  }
382  }

384  /// @brief Checks if the Configuration Reaction's parameters are valid
385  /// @details Parameters are not valid if the FileWatch object doesn't exist or it isn't a yaml file.
386  /// If this is the installation phase (denoted by the NO_OP FileWatch event), and this is not the default
387  /// configuration file, then the Reaction will also not run. This is to prevent Configuration Reactors
388  /// running multiple times during installation for one module.
389  /// @throws std::runtime_error if there is a YAML parsing error
390  /// @tparam DSL Magic NUClear type. Ignore for the purpose of understanding this function
391  /// @param t The associated Configuration Reaction
392  /// @return False is the reaction is not to be run, otherwise true
393  template <typename DSL>
394  [[nodiscard]] static inline bool precondition(threading::ReactionTask& t) {
395  ::extension::FileWatch watch = DSLProxy<::extension::FileWatch>::get<DSL>(t);
396  // Check if the watch is valid and the file is a yaml file
397  if (!watch || fs::path(watch.path).extension() != ".yaml") {
398  return false;
399  }
400 
401  // Get hostname, platform and binary name to check if this is not a default configuration file
402  const std::string hostname = utility::support::get_hostname();
403  const std::string platform(::extension::Configuration::get_platform(hostname));
404  const auto binary_name = get_first_command_line_arg();
405  const std::string robot_name = utility::platform::get_robot_alias(hostname);
406 
407  // Get the components of the path
408  auto c = utility::strutil::split(watch.path, '/');
409 
410  // Remove the yaml file name from the vector since we don't care what it is
411  c.pop_back();
412 
413  // Returns true if the string exists in the vector
414  auto str_exists = [&](const std::string& key) {
415  if (key.empty()) {
416  return false;
417  }
418  return std::find(c.begin(), c.end(), key) != c.end();
419  };
420 
421  // Remove all parts of the path before and including config since we don't care about them
422  while (str_exists("config")) {
423  c.erase(c.begin());
424  }
425 
426  // If it's the installation phase, and the path contains anything indicating it is not default config,
427  // then don't let the reaction run
428  bool is_not_default = robot_name.empty()
429  ? (str_exists(hostname) || str_exists(platform) || str_exists(binary_name))
430  : (str_exists(hostname) || str_exists(platform) || str_exists(binary_name)
431  || str_exists(robot_name));
432  if (watch.events == ::extension::FileWatch::Event::NO_OP && is_not_default) {
433  return false;
434  }
435  // Satisfied all the checks, the reaction can run
436  return true;
437  }

439  /// @brief Runs just before the Configuration callback to prepare the Reaction's parameters
440  /// @details The Configuration is constructed and returned, parsing the YAML in the process. Checks are done
441  /// in the precondition function, preventing invalid or unwanted files.
442  /// @throws std::runtime_error if there is a YAML parsing error
443  /// @tparam DSL Magic NUClear type. Ignore for the purpose of understanding this function
444  /// @param t The associated Configuration Reaction
445  /// @return A Configuration object is returned.
446  template <typename DSL>
447  [[nodiscard]] static std::shared_ptr<::extension::Configuration> get(threading::ReactionTask& t) {
448  // Get the file watch event
449  ::extension::FileWatch watch = DSLProxy<::extension::FileWatch>::get<DSL>(t);
450 
451  // Return our yaml file
452  try {
453  // Get hostname so we can find the correct per-robot config directory.
454  const std::string hostname = utility::support::get_hostname();
455  const std::string platform(::extension::Configuration::get_platform(hostname));
456  const auto binary_name = get_first_command_line_arg();
457  std::string robot_name = utility::platform::get_robot_alias(hostname);
458  robot_name = robot_name.empty() ? hostname : robot_name;
459 
460  // Get relative path to config file.
461  const auto components = utility::strutil::split(watch.path, '/');
462  fs::path relative_path{};
463  bool flag = false;
464  for (const auto& component : components) {
465  // Ignore the hostname/binary name if they are present.
466  if (flag && (component != hostname) && (component != binary_name) && (component != platform)
467  && (component != robot_name)) {
468  relative_path = relative_path / component;
469  }
470 
471  // Want paths relative to the config folder.
472  if (component == "config") {
473  flag = true;
474  }
475  }
476  return std::make_shared<::extension::Configuration>(relative_path, hostname, binary_name, platform);
477  }
478  catch (const YAML::ParserException& e) {
479  throw std::runtime_error(watch.path.string() + " " + std::string(e.what()));
480  }
481  }
482  };
