35  // PING Instruction that checks whether the Packet has arrived to a device with the same ID as Packet
36  // ID READ Instruction to read data from the Device WRITE Instruction to write data on the
37  // Device REG_WRITE Instruction that registers the Instruction Packet to a standby status; Packet is later
38  // executed
39  // through the Action command
40  // ACTION Instruction that executes the Packet that was registered beforehand using Reg Write
41  // FACTORY_RESET Instruction that resets the Control Table to its initial factory default settings
42  // REBOOT Instruction to reboot the Device
43  // STATUS_RETURN Return Instruction for the Instruction Packet
44  // SYNC_READ For multiple devices, Instruction to read data from the same Address with the same length
45  // at once
46  // SYNC_WRITE For multiple devices, Instruction to write data on the same Address with the same length
47  // at once
48  // BULK_READ For multiple devices, Instruction to read data from different Addresses with different
49  // lengths at once
50  // BULK_WRITE For multiple devices, Instruction to write data on different Addresses with different
51  // lengths at once
52 
53  enum Instruction : uint8_t {
54  PING = 0x01,
55  READ = 0x02,
56  WRITE = 0x03,
57  REG_WRITE = 0x04,
58  ACTION = 0x05,
59  FACTORY_RESET = 0x06,
60  REBOOT = 0x08,
61  STATUS_RETURN = 0x55,
62  SYNC_READ = 0x82,
63  SYNC_WRITE = 0x83,
64  BULK_READ = 0x92,
65  BULK_WRITE = 0x93
66  };
