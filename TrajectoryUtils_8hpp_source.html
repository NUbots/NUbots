190  /**
191  * Return initialized trajectories for trunk/foot ik cartesian with empty splines

193  inline void trajectories_init(Trajectories& traj) {
194  if (traj.size() != 0) {
195  traj.reset();

197  else {
198  traj.add(TrajectoryTypes::IS_DOUBLE_SUPPORT);
199  traj.add(TrajectoryTypes::IS_LEFT_SUPPORT_FOOT);

208  traj.add(TrajectoryTypes::FOOT_POS_Z);

215  /**
216  * Compute from given spline container trajectory Cartesian trunk and foot position/velocity/acceleration
217  * and assign it to given vector
218  */
219  [[nodiscard]] inline Vector3fQuadruple trajectories_trunk_foot_pos(const float& t, const Trajectories& traj) {
220  // Compute Cartesian positions
221  const auto trunk_pos = Eigen::Vector3f(traj.get(TrajectoryTypes::TRUNK_POS_X).pos(t),
222  traj.get(TrajectoryTypes::TRUNK_POS_Y).pos(t),
223  traj.get(TrajectoryTypes::TRUNK_POS_Z).pos(t));
224  const auto trunkAxis = Eigen::Vector3f(traj.get(TrajectoryTypes::TRUNK_AXIS_X).pos(t),
225  traj.get(TrajectoryTypes::TRUNK_AXIS_Y).pos(t),
226  traj.get(TrajectoryTypes::TRUNK_AXIS_Z).pos(t));
227  const auto footPos = Eigen::Vector3f(traj.get(TrajectoryTypes::FOOT_POS_X).pos(t),
228  traj.get(TrajectoryTypes::FOOT_POS_Y).pos(t),
229  traj.get(TrajectoryTypes::FOOT_POS_Z).pos(t));
230  const auto footAxis = Eigen::Vector3f(traj.get(TrajectoryTypes::FOOT_AXIS_X).pos(t),
231  traj.get(TrajectoryTypes::FOOT_AXIS_Y).pos(t),
232  traj.get(TrajectoryTypes::FOOT_AXIS_Z).pos(t));
233  return {trunk_pos, trunkAxis, footPos, footAxis};
234  }

270  /**
271  * @brief Computes the support foot state, evaluating double support foot and whether is_left_supportFoot is true
272  * at time t
273  * @param t Time at which the state should be evaluated
274  * @param traj The set of trajectories which are being evaluated
275  * @return std::pair<bool, bool> {is_double_supportFoot, is_left_supportFoot}, as evaluated at time t
276  */
277  [[nodiscard]] constexpr inline std::pair<bool, bool> trajectories_support_foot_state(const float& t,
278  const Trajectories& traj) {
279  // Compute support foot state
280  return {traj.get(TrajectoryTypes::IS_DOUBLE_SUPPORT).pos(t) >= 0.5f,
281  traj.get(TrajectoryTypes::IS_LEFT_SUPPORT_FOOT).pos(t) >= 0.5f};
282  }
