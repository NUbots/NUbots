45  /*! @brief Calculates the leg joints for a given input ankle position.
46  The robot coordinate system has origin a distance DISTANCE_FROM_BODY_TO_HIP_JOINT above the midpoint
47  of the hips. Robot coordinate system: x is out of the front of the robot y is left, from right shoulder
48  to left z is upward, from feet to head Input ankle coordinate system: x is forward, from heel to toe y is
49  left, z is normal to the plane of the foot
50  @param target The target 4x4 basis matrix for the ankle
51  @param isLeft Request for left leg motors or right leg motors?
52  @param RobotKinematicsModel The class containing the leg model of the robot.
53  */
54  template <typename Scalar>
55  [[nodiscard]] std::vector<std::pair<ServoID, Scalar>> calculateLegJoints(
56  const message::actuation::KinematicsModel& model,
57  const Eigen::Transform<Scalar, 3, Eigen::Isometry>& target_,
58  const LimbID& limb) {
59  const Scalar LENGTH_BETWEEN_LEGS = model.leg.LENGTH_BETWEEN_LEGS;
60  const Scalar DISTANCE_FROM_BODY_TO_HIP_JOINT = model.leg.HIP_OFFSET_Z;
61  const Scalar HIP_OFFSET_X = model.leg.HIP_OFFSET_X;
62  const Scalar UPPER_LEG_LENGTH = model.leg.UPPER_LEG_LENGTH;
63  const Scalar LOWER_LEG_LENGTH = model.leg.LOWER_LEG_LENGTH;
64 
65  std::vector<std::pair<ServoID, Scalar>> positions;
66 
67  Scalar hipYaw = 0;
68  Scalar hipRoll = 0;
69  Scalar hipPitch = 0;
70  Scalar knee = 0;
71  Scalar anklePitch = 0;
72  Scalar ankleRoll = 0;
73 
74  // Correct for input referencing the bottom of the foot
75  Eigen::Transform<Scalar, 3, Eigen::Isometry> target(target_);
76  target = target.translate(Eigen::Matrix<Scalar, 3, 1>(0.0, 0.0, model.leg.FOOT_HEIGHT));
77 
78  // Tci = transformation (not necessarily homogeneous) from input coordinates to calculation coordinates
79  // TODO remove this. It was due to wrong convention use
80  Eigen::Matrix<Scalar, 4, 4> Tci;
81  // clang-format off
82  Tci << 0.0, 1.0, 0.0, 0.0,
83  1.0, 0.0, 0.0, 0.0,
84  0.0, 0.0, -1.0, 0.0,
85  0.0, 0.0, 0.0, 1.0;
86  // clang-format on
87  // Rotate input position from standard robot coords to foot coords
88  Eigen::Matrix<Scalar, 3, 1> translation = (Tci * target.matrix().template rightCols<1>()).template head<3>();
89  target = Tci * target * Tci.transpose();
90  target.translation() = translation;
91 
92  // swap legs if needed
93  if (limb != LimbID::LEFT_LEG) {
94  target(0, 1) = -target(0, 1);
95  target(0, 2) = -target(0, 2);
96  target(1, 0) = -target(1, 0);
97  target(2, 0) = -target(2, 0);
98  target.translation().x() = -target.translation().x();
99  }
100 
101  const Eigen::Matrix<Scalar, 3, 1> ankleX = target.matrix().template leftCols<1>().template head<3>();
102  const Eigen::Matrix<Scalar, 3, 1> ankleY = target.matrix().template middleCols<1>(1).template head<3>();
103  const Eigen::Matrix<Scalar, 3, 1> anklePos = target.translation();
104 
105  const Eigen::Matrix<Scalar, 3, 1> hipOffset(LENGTH_BETWEEN_LEGS * 0.5,
106  HIP_OFFSET_X,
107  DISTANCE_FROM_BODY_TO_HIP_JOINT);
108 
109  Eigen::Matrix<Scalar, 3, 1> targetLeg = anklePos - hipOffset;
110 
111  const Scalar maxLegLength = UPPER_LEG_LENGTH + LOWER_LEG_LENGTH;
112  Scalar length = targetLeg.norm();
113  if (length > maxLegLength) {
114  targetLeg = targetLeg * maxLegLength / length;
115  length = targetLeg.norm();
116  }
117  const Scalar sqrLength = length * length;
118  const Scalar sqrUpperLeg = UPPER_LEG_LENGTH * UPPER_LEG_LENGTH;
119  const Scalar sqrLowerLeg = LOWER_LEG_LENGTH * LOWER_LEG_LENGTH;
120 
121  const Scalar cosKnee = (sqrUpperLeg + sqrLowerLeg - sqrLength) / (2.0 * UPPER_LEG_LENGTH * LOWER_LEG_LENGTH);
122  knee = std::acos(std::max(std::min(cosKnee, Scalar(1.0)), Scalar(-1.0)));
123 
124  const Scalar cosLowerLeg = (sqrLowerLeg + sqrLength - sqrUpperLeg) / (2.0 * LOWER_LEG_LENGTH * length);
125  const Scalar lowerLeg = std::acos(std::max(std::min(cosLowerLeg, Scalar(1.0)), Scalar(-1.0)));
126 
127  const Scalar phi2 = std::acos(targetLeg.dot(ankleY) / length);
128 
129  anklePitch = lowerLeg + phi2 - M_PI_2;
130 
131  const Eigen::Matrix<Scalar, 3, 1> unitTargetLeg = targetLeg / length;
132 
133  Eigen::Matrix<Scalar, 3, 1> hipX = ankleY.cross(unitTargetLeg);
134  Scalar hipXLength = hipX.norm();
135  if (hipXLength > 0) {
136  hipX /= hipXLength;
137  }
138  else {
139  NUClear::log<NUClear::DEBUG>(
140  "InverseKinematics::calculateLegJoints : targetLeg and ankleY parallel. This is unhandled at "
141  "the moment. requested pose = \n",
142  target.matrix());
143  return positions;
144  }
145  // Will be unit as ankleY and hipX are normal and unit
146  const Eigen::Matrix<Scalar, 3, 1> legPlaneTangent = ankleY.cross(hipX);
147 
148  ankleRoll = std::atan2(ankleX.dot(legPlaneTangent), ankleX.dot(hipX));
149 
150  const bool isAnkleAboveWaist = unitTargetLeg.dot(Eigen::Matrix<Scalar, 3, 1>::UnitZ()) < 0;
151 
152  const Scalar cosZandHipX = Eigen::Matrix<Scalar, 3, 1>::UnitZ().dot(hipX);
153  const bool hipRollPositive = cosZandHipX <= 0;
154  Eigen::Matrix<Scalar, 3, 1> legPlaneGlobalZ =
155  (isAnkleAboveWaist ? -1 : 1) * (Eigen::Matrix<Scalar, 3, 1>::UnitZ() - (cosZandHipX * hipX));
156  const Scalar legPlaneGlobalZLength = legPlaneGlobalZ.norm();
157  if (legPlaneGlobalZLength > 0) {
158  legPlaneGlobalZ /= legPlaneGlobalZLength;
159  }
160 
161  const Scalar cosHipRoll = legPlaneGlobalZ.dot(Eigen::Matrix<Scalar, 3, 1>::UnitZ());
162  hipRoll = (hipRollPositive ? 1 : -1) * std::acos(std::max(std::min(cosHipRoll, Scalar(1)), Scalar(-1)));
163 
164  const Scalar phi4 = M_PI - knee - lowerLeg;
165  // Superposition values:
166  const Scalar sinPIminusPhi2 = std::sin(M_PI - phi2);
167  const Eigen::Matrix<Scalar, 3, 1> unitUpperLeg =
168  unitTargetLeg * (std::sin(phi2 - phi4) / sinPIminusPhi2) + ankleY * (std::sin(phi4) / sinPIminusPhi2);
169  const bool isHipPitchPositive = hipX.dot(unitUpperLeg.cross(legPlaneGlobalZ)) >= 0;
170 
171  hipPitch = (isHipPitchPositive ? 1 : -1)
172  * std::acos(std::max(std::min(legPlaneGlobalZ.dot(unitUpperLeg), Scalar(1)), Scalar(-1)));
173 
174  // If leg is above waist then hipX is pointing in the wrong direction in the xy plane
175  Eigen::Matrix<Scalar, 3, 1> hipXProjected = (isAnkleAboveWaist ? -1 : 1) * hipX;
176  hipXProjected.z() = 0;
177  hipXProjected.normalize();
178  const bool isHipYawPositive = hipXProjected.dot(Eigen::Matrix<Scalar, 3, 1>::UnitY()) >= 0;
179  hipYaw = (isHipYawPositive ? 1 : -1)
180  * std::acos(std::max(std::min(hipXProjected.dot(Eigen::Matrix<Scalar, 3, 1>::UnitX()), Scalar(1)),
181  Scalar(-1)));
182 
183  if (limb == LimbID::LEFT_LEG) {
184  positions.push_back(std::make_pair(ServoID::L_HIP_YAW, -hipYaw));
185  positions.push_back(std::make_pair(ServoID::L_HIP_ROLL, hipRoll));
186  positions.push_back(std::make_pair(ServoID::L_HIP_PITCH, -hipPitch));
187  positions.push_back(std::make_pair(ServoID::L_KNEE, M_PI - knee));
188  positions.push_back(std::make_pair(ServoID::L_ANKLE_PITCH, -anklePitch));
189  positions.push_back(std::make_pair(ServoID::L_ANKLE_ROLL, ankleRoll));
190  }
191  else {
192  positions.push_back(std::make_pair(ServoID::R_HIP_YAW, (model.leg.LEFT_TO_RIGHT_HIP_YAW) * -hipYaw));
193  positions.push_back(std::make_pair(ServoID::R_HIP_ROLL, (model.leg.LEFT_TO_RIGHT_HIP_ROLL) * hipRoll));
194  positions.push_back(std::make_pair(ServoID::R_HIP_PITCH, (model.leg.LEFT_TO_RIGHT_HIP_PITCH) * -hipPitch));
195  positions.push_back(std::make_pair(ServoID::R_KNEE, (model.leg.LEFT_TO_RIGHT_KNEE) * (M_PI - knee)));
196  positions.push_back(
197  std::make_pair(ServoID::R_ANKLE_PITCH, (model.leg.LEFT_TO_RIGHT_ANKLE_PITCH) * -anklePitch));
198  positions.push_back(
199  std::make_pair(ServoID::R_ANKLE_ROLL, (model.leg.LEFT_TO_RIGHT_ANKLE_ROLL) * ankleRoll));
200  }
201 
202  return positions;
203  }

205  template <typename Scalar>
206  [[nodiscard]] std::vector<std::pair<ServoID, Scalar>> calculateLegJoints(
207  const message::actuation::KinematicsModel& model,
208  const Eigen::Transform<Scalar, 3, Eigen::Isometry>& leftTarget,
209  const Eigen::Transform<Scalar, 3, Eigen::Isometry>& rightTarget) {
210  auto joints = calculateLegJoints<Scalar>(model, leftTarget, LimbID::LEFT_LEG);
211  auto joints2 = calculateLegJoints<Scalar>(model, rightTarget, LimbID::RIGHT_LEG);
212  joints.insert(joints.end(), joints2.begin(), joints2.end());
213  return joints;
214  }

216  template <typename Scalar>
217  [[nodiscard]] std::vector<std::pair<ServoID, Scalar>> calculateHeadJoints(
218  const Eigen::Matrix<Scalar, 3, 1>& cameraUnitVector) {
219  return {std::make_pair(ServoID::HEAD_YAW, std::atan2(cameraUnitVector.y(), cameraUnitVector.x())),
220  std::make_pair(ServoID::HEAD_PITCH,
221  std::atan2(-cameraUnitVector.z(),
222  std::sqrt(cameraUnitVector.x() * cameraUnitVector.x()
223  + cameraUnitVector.y() * cameraUnitVector.y())))};
224  }
