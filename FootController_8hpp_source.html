64  /// @brief Called by the powerplant to build and setup the FootController reactor.

67  private:
68  /// @brief Stores configuration values
69  struct Config {
70  /// @brief Foot controller mode
71  std::string mode = "IK";
72 
73  /// @brief Map between ServoID and ServoState
74  std::map<utility::input::ServoID, ServoState> servo_states = {
75  {utility::input::ServoID::L_HIP_YAW, ServoState()},
76  {utility::input::ServoID::L_HIP_ROLL, ServoState()},
77  {utility::input::ServoID::L_HIP_PITCH, ServoState()},
78  {utility::input::ServoID::L_KNEE, ServoState()},
79  {utility::input::ServoID::L_ANKLE_PITCH, ServoState()},
80  {utility::input::ServoID::L_ANKLE_ROLL, ServoState()},
81  {utility::input::ServoID::R_HIP_YAW, ServoState()},
82  {utility::input::ServoID::R_HIP_ROLL, ServoState()},
83  {utility::input::ServoID::R_HIP_PITCH, ServoState()},
84  {utility::input::ServoID::R_KNEE, ServoState()},
85  {utility::input::ServoID::R_ANKLE_PITCH, ServoState()},
86  {utility::input::ServoID::R_ANKLE_ROLL, ServoState()},
87  };
88  /// @brief Startup gain before setting the desired gains
89  double startup_gain = 0;
90  /// @brief Delay before setting the desired gains
91  std::chrono::seconds set_gain_delay = std::chrono::seconds(0);
92  /// @brief Desired gains for each servo
93  std::map<std::string, double> desired_gains{};
94 
95  /// @brief Proportional gain for torso orientation roll correction
96  double roll_p_gain = 0.0;
97  /// @brief Proportional gain for torso orientation pitch correction
98  double pitch_p_gain = 0.0;
99  /// @brief Integral gain for torso orientation roll correction
100  double roll_i_gain = 0.0;
101  /// @brief Integral gain for torso orientation pitch correction
102  double pitch_i_gain = 0.0;
103  /// @brief Antiwindup max I error
104  double max_i_error = 0.0;
105  /// @brief Derivative gain for torso orientation roll correction
106  double roll_d_gain = 0.0;
107  /// @brief Derivative gain for torso orientation pitch correction
108  double pitch_d_gain = 0.0;
109  /// @brief Maximum allowed pitch error before disabling correction
110  double max_pitch_error = 0.0;
111  /// @brief Maximum allowed roll error before disabling correction
112  double max_roll_error = 0.0;
113  } cfg;
114 
115  /// @brief Accumulates the integral of the roll error over time for use in PID control.
116  double integral_roll_error = 0;
117  /// @brief Accumulates the integral of the pitch error over time for use in PID control.
118  double integral_pitch_error = 0;
119 
120  /// @brief Stores the timestamp of the last PID control update, used to compute time deltas for integral and
121  /// derivative calculations.
122  NUClear::clock::time_point last_update_time{};

124  /// @brief Foot controller logic
125  template <typename FootControlTask, typename IKTask>
126  void control_foot(const FootControlTask& foot_control_task,
127  IKTask& ik_task,
128  const Sensors& sensors,
129  LimbID limb_id) {
130 
131  ik_task->time = foot_control_task.time;
132  ik_task->Htf = foot_control_task.Htf;
133 
134  if (cfg.mode == "IK") {
135  for (auto id : utility::input::LimbID::servos_for_limb(limb_id)) {
136  ik_task->servos[id] = ServoState(cfg.servo_states[id].gain, TORQUE_ENABLED);
137  // Get servo from sensors for graphing information
138  auto it = std::find_if(sensors.servo.begin(),
139  sensors.servo.end(),
140  [id](const message::input::Sensors::Servo& servo) {
141  return servo.id == static_cast<uint32_t>(id);
142  });
143  auto servo = *it;
144  std::string name = static_cast<std::string>(id);
145  emit(graph("Servo Present Position/" + name, servo.present_position));
146  emit(graph("Servo Goal Position/" + name, servo.goal_position));
147  emit(graph("Servo Error/" + name, servo.present_position - servo.goal_position));
148  }
149  }
150  else {
151  throw std::runtime_error("Invalid mode");
152  }
153 
154  // Add correction to the torso orientation if enabled
155  if (foot_control_task.correction_enabled) {
156  // Hwt quaternion
157  Eigen::Quaterniond Hwt_quat(sensors.Htw.inverse().linear());
158 
159  // Get fused roll and pitch
160  Eigen::Vector3d rpy = mat_to_rpy_intrinsic(Hwt_quat.toRotationMatrix());
161  double fused_roll = rpy.x();
162  double fused_pitch = rpy.y();
163 
164  if (log_level <= DEBUG) {
165  // Graph the correction being applied
166  emit(graph("Balance/Actual_Roll", fused_roll));
167  emit(graph("Balance/Actual_Pitch", fused_pitch));
168  }
169 
170  // Get the desired roll and pitch
171  Eigen::Quaterniond Hft_quat(ik_task->Htf.inverse().linear());
172  rpy = mat_to_rpy_intrinsic(Hft_quat.toRotationMatrix());
173  double desired_roll = rpy.x();
174  double desired_pitch = rpy.y();
175 
176  if (log_level <= DEBUG) {
177  // Graph desired orientation (before PID correction)
178  emit(graph("Balance/Desired_Roll_Original", desired_roll));
179  emit(graph("Balance/Desired_Pitch_Original", desired_pitch));
180  }
181 
182  // Compute the error between the desired torso orientation and the actual torso orientation
183  auto roll_error = desired_roll - fused_roll;
184  auto pitch_error = desired_pitch - fused_pitch;
185 
186  if (log_level <= DEBUG) {
187  // Graph the errors
188  emit(graph("Balance/Roll_Error", roll_error));
189  emit(graph("Balance/Pitch_Error", pitch_error));
190  }
191 
192  auto dt =
193  std::chrono::duration_cast<std::chrono::duration<double>>(NUClear::clock::now() - last_update_time)
194  .count();
195  last_update_time = NUClear::clock::now();
196 
197  // P control
198  auto clamped_roll_error = utility::math::clamp(-cfg.max_roll_error, roll_error, cfg.max_roll_error);
199  auto clamped_pitch_error = utility::math::clamp(-cfg.max_pitch_error, pitch_error, cfg.max_pitch_error);
200  desired_roll += cfg.roll_p_gain * clamped_roll_error;
201  desired_pitch += cfg.pitch_p_gain * clamped_pitch_error;
202 
203  if (log_level <= DEBUG) {
204  if (roll_error > cfg.max_roll_error || roll_error < -cfg.max_roll_error) {
205  // If the error is too large, we are probably falling over and should pause applying control.
206  log<DEBUG>(fmt::format("Balance correction disabled due to large ROLL error: roll_error = {}",
207  roll_error));
208  log<DEBUG>(fmt::format("Clamping roll error to: {}", clamped_roll_error));
209  }
210 
211  if (pitch_error > cfg.max_pitch_error || pitch_error < -cfg.max_pitch_error) {
212  // If the error is too large, we are probably falling over and should pause applying control.
213  log<DEBUG>(fmt::format("Balance correction disabled due to large error: pitch_error = {}",
214  pitch_error));
215  log<DEBUG>(fmt::format("Clamping pitch error to: {}", clamped_pitch_error));
216  }
217  }
218 
219  // Graph P Correction
220  if (log_level <= DEBUG) {
221  emit(graph("Balance/P_Roll_Correction", cfg.roll_p_gain * roll_error));
222  emit(graph("Balance/P_Pitch_Correction", cfg.pitch_p_gain * pitch_error));
223  }
224 
225  // I control
226  integral_roll_error += roll_error * dt;
227  integral_pitch_error += pitch_error * dt;
228 
229  // Graph the integral correction
230  if (log_level <= DEBUG) {
231  emit(graph("Balance/I_Roll_Correction", cfg.roll_i_gain * integral_roll_error));
232  emit(graph("Balance/I_Pitch_Correction", cfg.pitch_i_gain * integral_pitch_error));
233  }
234 
235  // Anti windup
236  integral_roll_error = std::max(std::min(integral_roll_error, cfg.max_i_error), -cfg.max_i_error);
237  integral_pitch_error = std::max(std::min(integral_pitch_error, cfg.max_i_error), -cfg.max_i_error);
238 
239  desired_roll += cfg.roll_i_gain * integral_roll_error;
240  desired_pitch += cfg.pitch_i_gain * integral_pitch_error;
241 
242  // D control using gyroscope values
243  auto roll_rate = sensors.gyroscope.x();
244  auto pitch_rate = sensors.gyroscope.y();
245  desired_roll -= cfg.roll_d_gain * roll_rate;
246  desired_pitch -= cfg.pitch_d_gain * pitch_rate;
247 
248  if (log_level <= DEBUG) {
249  // Graph D corrections and rates
250  emit(graph("Balance/D_Roll_Correction", -cfg.roll_d_gain * roll_rate));
251  emit(graph("Balance/D_Pitch_Correction", -cfg.pitch_d_gain * pitch_rate));
252  emit(graph("Balance/Roll_Rate", roll_rate));
253  emit(graph("Balance/Pitch_Rate", pitch_rate));
254 
255  // Graph final corrected desired values
256  emit(graph("Balance/Desired_Roll_Final", desired_roll));
257  emit(graph("Balance/Desired_Pitch_Final", desired_pitch));
258  }
259 
260  double desired_yaw = mat_to_rpy_intrinsic(Hft_quat.toRotationMatrix()).z();
261 
262  // Compute desired orientation: yaw * fused_roll_pitch
263  Eigen::Matrix3d desired_Rft = Eigen::AngleAxisd(desired_yaw, Eigen::Vector3d::UnitZ())
264  * rpy_intrinsic_to_mat(Eigen::Vector3d(desired_roll, desired_pitch, 0.0));
265  Eigen::Isometry3d Htf_corrected = foot_control_task.Htf;
266  Htf_corrected.linear() = desired_Rft.transpose();
267 
268  ik_task->Htf = Htf_corrected;
269  }
270  }
271  };
