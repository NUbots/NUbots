98  /// @brief Foot controller logic
99  template <typename FootControlTask, typename IKTask>
100  void control_foot(const FootControlTask& foot_control_task,
101  IKTask& ik_task,
102  const Sensors& sensors,
103  LimbID limb_id) {
104 
105  ik_task->time = foot_control_task.time;
106  ik_task->Htf = foot_control_task.Htf;
107 
108  if (cfg.mode == "IK") {
109  for (auto id : utility::input::LimbID::servos_for_limb(limb_id)) {
110  ik_task->servos[id] = ServoState(cfg.servo_states[id].gain, TORQUE_ENABLED);
111  // Get servo from sensors for graphing information
112  auto it = std::find_if(sensors.servo.begin(),
113  sensors.servo.end(),
114  [id](const message::input::Sensors::Servo& servo) {
115  return servo.id == static_cast<uint32_t>(id);
116  });
117  auto servo = *it;
118  std::string name = static_cast<std::string>(id);
119  emit(graph("Servo Present Position/" + name, servo.present_position));
120  emit(graph("Servo Goal Position/" + name, servo.goal_position));
121  emit(graph("Servo Error/" + name, servo.present_position - servo.goal_position));
122  }
123  }
124  else {
125  throw std::runtime_error("Invalid mode");
126  }
127  }
128  };
