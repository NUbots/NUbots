68  struct Apply {
69  Apply(std::ostream& out) : out(out) {}
70 
71  template <typename T>
72  std::ostream& operator<<(const T& el) {
73 
74  // Print out the start of our escape code
75  out << "\033[";
76 
77  // Print out our first code
78  out << int(First);
79 
80  // Print out our remaining codes
81  for (int code : std::initializer_list<int>({int(Codes)...})) {
82  out << ";" << code;
83  }
84 
85  // Close our code
86  out << "m";
87 
88  // Print our message and reset
89  return out << el << "\033[0m";
90  }
91 
92  std::ostream& out;
93  };
94  };

96  template <enum ANSICode... Codes>
97  typename ANSISGR<Codes...>::Apply operator<<(std::ostream& out, const ANSISGR<Codes...> /*codes*/) {
98  return typename ANSISGR<Codes...>::Apply(out);
99  }
