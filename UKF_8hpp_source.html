38  template <typename Scalar, template <typename> class FilterModel>
39  class UKF {
40 
41  public:
42  // The model
43  using Model = FilterModel<Scalar>;
44  Model model;
45 
46  // Dimension types for vectors and square matrices
47  using StateVec = Eigen::Matrix<Scalar, Model::size, 1>;
48  using StateMat = Eigen::Matrix<Scalar, Model::size, Model::size>;
49 
50  static constexpr Scalar ALPHA_DEFAULT = Scalar(0.1);
51  static constexpr Scalar KAPPA_DEFAULT = Scalar(0.0);
52  static constexpr Scalar BETA_DEFAULT = Scalar(2.0);
53 
54  private:
55  // The number of sigma points
56  static constexpr unsigned int NUM_SIGMA_POINTS = (Model::size * 2) + 1;
57 
58  using SigmaVec = Eigen::Matrix<Scalar, NUM_SIGMA_POINTS, 1>;
59  using SigmaMat = Eigen::Matrix<Scalar, Model::size, NUM_SIGMA_POINTS>;
60  using SigmaSquareMat = Eigen::Matrix<Scalar, NUM_SIGMA_POINTS, NUM_SIGMA_POINTS>;
61 
62  // Our estimate and covariance
63  StateVec mean;
64  StateMat covariance;
65 
66  // Our sigma points for UKF
67  StateVec sigma_mean;
68  SigmaMat sigma_points;
69 
70  SigmaMat centred_sigma_points; // X in Steves kalman theory
71  SigmaVec d;
72  SigmaSquareMat covariance_update; // C in Steves kalman theory
73 
74  // The mean and covariance weights
75  SigmaVec mean_weights;
76  SigmaVec covariance_weights;
77 
78 
79  // UKF variables
80  Scalar covariance_sigma_weight;

82  /**
83  * @brief Generate new sigma points given our mean and covariance.
84  */
85  template <typename T, int S>
86  [[nodiscard]] static Eigen::ComputationInfo generate_sigma_points(const Eigen::Matrix<T, S, 1>& mean,
87  const Eigen::Matrix<T, S, S>& covariance,
88  const T& sigma_weight,
89  SigmaMat& points) {
90 
91  // Our first row is always the mean
92  points.col(0) = mean;
93 
94  // Get our Cholesky decomposition
95  // Impose positive semi-definiteness on the covariance matrix
96  Eigen::LLT<StateMat> cholesky(sigma_weight
97  * covariance.unaryExpr([](const Scalar& c) { return std::abs(c); }));
98 
99  if (cholesky.info() == Eigen::Success) {
100  // Put our values in either end of the matrix
101  StateMat chol = cholesky.matrixU().toDenseMatrix();
102  for (unsigned int i = 1; i < Model::size + 1; ++i) {
103  points.col(i) = (mean + chol.col(i - 1));
104  points.col(i + Model::size) = (mean - chol.col(i - 1));
105  }
106  }
107 
108  return cholesky.info();
109  }

111  /**
112  * @brief Calculate the the mean given a set of sigma points.
113  *
114  * @param mean the mean to set
115  * @param sigma_points the sigma points to calculate the mean from
116  */
117  template <typename T1, typename T2, int S>
118  // model size, num sigma points
119  static Eigen::Matrix<T1, S, 1> mean_from_sigmas(const Eigen::Matrix<T1, S, NUM_SIGMA_POINTS>& sigma_points,
120  const Eigen::Matrix<T2, NUM_SIGMA_POINTS, 1>& weights) {
121  return sigma_points * weights.template cast<T1>();
122  }
123 
124  template <typename T1, typename T2, int S>
125  static Eigen::Matrix<T1, S, S> covariance_from_sigmas(
126  const Eigen::Matrix<T1, S, NUM_SIGMA_POINTS>& sigma_points,
127  const Eigen::Matrix<T1, S, 1>& mean,
128  const Eigen::Matrix<T2, NUM_SIGMA_POINTS, 1>& weights) {
129 
130  Eigen::Matrix<T1, S, NUM_SIGMA_POINTS> mean_centred = sigma_points.colwise() - mean;
131  return mean_centred * weights.template cast<T1>().asDiagonal() * mean_centred.transpose();
132  }
133 
134  template <typename T1, typename T2, int S>
135  static Eigen::Matrix<T1, S, S> covariance_from_sigmas(
136  const Eigen::Matrix<T1, S, NUM_SIGMA_POINTS>& sigma_points,
137  const T1& mean,
138  const Eigen::Matrix<T2, NUM_SIGMA_POINTS, 1>& weights) {
139 
140  Eigen::Matrix<T1, S, NUM_SIGMA_POINTS> mean_centred = (sigma_points.array() - mean).matrix();
141  return mean_centred * weights.template cast<T1>().asDiagonal() * mean_centred.transpose();
142  }

144  public:
145  UKF(StateVec initial_mean = StateVec::Zero(),
146  StateMat initial_covariance = StateMat::Identity() * 0.1,
147  Scalar alpha = ALPHA_DEFAULT,
148  Scalar kappa = KAPPA_DEFAULT,
149  Scalar beta = BETA_DEFAULT)
150  : model()
151  , mean(initial_mean)
152  , covariance(initial_covariance)
153  , sigma_mean(StateVec::Zero())
154  , sigma_points(SigmaMat::Zero())
155  , centred_sigma_points(SigmaMat::Zero())
156  , d(SigmaVec::Zero())
157  , covariance_update(SigmaSquareMat::Identity())
158  , mean_weights(SigmaVec::Zero())
159  , covariance_weights(SigmaVec::Zero())
160  , covariance_sigma_weight(0.0) {
161 
162  reset(initial_mean, initial_covariance, alpha, kappa, beta);
163  }
164 
165  Eigen::ComputationInfo reset(StateVec initial_mean,
166  StateMat initial_covariance,
167  Scalar alpha = ALPHA_DEFAULT,
168  Scalar kappa = KAPPA_DEFAULT,
169  Scalar beta = BETA_DEFAULT) {
170 
171  Scalar lambda = alpha * alpha * (Model::size + kappa) - Model::size;
172 
173  covariance_sigma_weight = Model::size + lambda;
174 
175  mean_weights.fill(1.0 / (2.0 * covariance_sigma_weight));
176  mean_weights[0] = lambda / covariance_sigma_weight;
177 
178  covariance_weights.fill(1.0 / (2.0 * covariance_sigma_weight));
179  covariance_weights[0] = lambda / covariance_sigma_weight + (1.0 - (alpha * alpha) + beta);
180 
181  return set_state(initial_mean, initial_covariance);
182  }
183 
184  Eigen::ComputationInfo set_state(StateVec initial_mean, StateMat initial_covariance) {
185  mean = initial_mean;
186  covariance = initial_covariance;
187 
188  // Calculate our sigma points
189  sigma_mean = mean;
190  Eigen::ComputationInfo cholesky_info =
191  generate_sigma_points(mean, covariance, covariance_sigma_weight, sigma_points);
192 
193  if (cholesky_info == Eigen::Success) {
194  // Reset our state for more measurements
195  covariance_update = covariance_weights.asDiagonal();
196  d.setZero();
197  centred_sigma_points = sigma_points.colwise() - sigma_mean;
198  }
199 
200  return cholesky_info;
201  }

203  template <typename... Args>
204  Eigen::ComputationInfo time(const Scalar& dt, const Args&... params) {
205  // Generate our sigma points
206  Eigen::ComputationInfo cholesky_info =
207  generate_sigma_points(mean, covariance, covariance_sigma_weight, sigma_points);
208 
209  if (cholesky_info == Eigen::Success) {
210  // Write the propagated version of the sigma point
211  for (unsigned int i = 0; i < NUM_SIGMA_POINTS; ++i) {
212  sigma_points.col(i) = model.time(sigma_points.col(i), dt, params...);
213  }
214 
215  // Calculate the new mean and covariance values.
216  mean = mean_from_sigmas(sigma_points, mean_weights);
217  mean = model.limit(mean);
218  covariance = covariance_from_sigmas(sigma_points, mean, covariance_weights);
219  covariance += model.noise(dt);
220 
221  // Re calculate our sigma points
222  sigma_mean = mean;
223  cholesky_info = generate_sigma_points(mean, covariance, covariance_sigma_weight, sigma_points);
224 
225  if (cholesky_info == Eigen::Success) {
226  // Reset our state for more measurements
227  covariance_update = covariance_weights.asDiagonal();
228  d.setZero();
229  centred_sigma_points = sigma_points.colwise() - sigma_mean;
230  }
231  }
232  return cholesky_info;
233  }

236  /**
237  * Perform a measurement update using the given measurement and covariance
238  */
239  template <typename MeasurementScalar, int S, int... VArgs, int... MArgs, typename... Args>
240  utility::support::LazyEvaluation<MeasurementScalar> measure(
241  const Eigen::Matrix<MeasurementScalar, S, 1, VArgs...>& measurement,
242  const Eigen::Matrix<MeasurementScalar, S, S, MArgs...>& measurement_variance,
243  const Args&... params) {
244 
245  // Allocate room for our predictions
246  Eigen::Matrix<MeasurementScalar, S, NUM_SIGMA_POINTS> predictions;
247 
248  // First step is to calculate the predicted measurement for each sigma point.
249  for (unsigned int i = 0; i < NUM_SIGMA_POINTS; ++i) {
250  predictions.col(i) = model.predict(sigma_points.col(i), params...);
251  }
252 
253  // Now calculate the mean of these measurement sigmas.
254  Eigen::Matrix<MeasurementScalar, S, 1> predicted_mean = mean_from_sigmas(predictions, mean_weights);
255  Eigen::Matrix<MeasurementScalar, S, NUM_SIGMA_POINTS> centred = predictions.colwise() - predicted_mean;
256 
257  // Create a state update in our measurement units
258  Eigen::Matrix<MeasurementScalar, NUM_SIGMA_POINTS, NUM_SIGMA_POINTS> update =
259  covariance_update.template cast<MeasurementScalar>();
260  update = update.transpose() * centred.transpose()
261  * (measurement_variance + centred * update * centred.transpose())
262  .llt()
263  .solve(Eigen::Matrix<MeasurementScalar, S, S>::Identity())
264  * centred * update;
265  covariance_update -= update.template cast<Scalar>();
266 
267  Eigen::Matrix<MeasurementScalar, S, 1> innovation = model.difference(measurement, predicted_mean);
268  d += (centred.transpose()
269  * measurement_variance.llt().solve(Eigen::Matrix<MeasurementScalar, S, S>::Identity()) * innovation)
270  .template cast<Scalar>();
271 
272  // Update our mean and covariance
273  mean = sigma_mean + centred_sigma_points * covariance_update * d;
274  mean = model.limit(mean);
275  covariance = centred_sigma_points * covariance_update * centred_sigma_points.transpose();
276 
277  // Calculate and return the likelihood of the prior mean and covariance given the new measurement
278  // (i.e. the prior probability density of the measurement):
279  return utility::support::LazyEvaluation<MeasurementScalar>(
280  [predictions, predicted_mean, measurement_variance, innovation, cw = this->covariance_weights] {
281  Eigen::Matrix<MeasurementScalar, S, S> predicted_covariance =
282  covariance_from_sigmas(predictions, predicted_mean, cw);
283 
284  Eigen::Matrix<MeasurementScalar, S, S> innovation_variance =
285  predicted_covariance + measurement_variance;
286 
287  MeasurementScalar likelihood_exponent =
288  (innovation.transpose()
289  * innovation_variance.llt().solve(Eigen::Matrix<MeasurementScalar, S, S>::Identity())
290  * innovation)
291  .x();
292 
293  MeasurementScalar loglikelihood =
294  0.5
295  * (std::log(innovation_variance.determinant()) + std::abs(likelihood_exponent)
296  + innovation.size() * std::log(2 * M_PI));
297 
298  return -loglikelihood;
299  });
300  }
