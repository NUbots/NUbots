76  /**
77  * @brief Calculates polynomial coefficients that approximate the inverse distortion
78  *
79  * @details
80  * These coefficients are based on math from the paper
81  * An Exact Formula for Calculating Inverse Radial Lens Distortions
82  * https://www.ncbi.nlm.nih.gov/pmc/articles/PMC4934233/pdf/sensors-16-00807.pdf
83  * These terms have been stripped back to only include k1 and k2 and only uses the first 4 terms
84  * In general for most cases this provides an accuracy of around 0.2 pixels which is sufficient.
85  * If more accuracy is required in the future or more input parameters are used they can be adjusted here.
86  *
87  * @tparam T the scalar type used for calculations and storage (normally one of float or double)
88  *
89  * @param lens the paramters that describe the lens that we are using to project
90  *
91  * @return the inverse coefficients that go from an undistorted image to a distorted one
92  */
93  template <typename T, typename Lens>
94  inline Eigen::Matrix<T, 4, 1> inverse_coefficients(const Lens& lens) {
95  const auto& k = lens.k;
96  return Eigen::Matrix<T, 4, 1>(
97  -k[0],
98  T(3.0) * (k[0] * k[0]) - k[1],
99  T(-12.0) * (k[0] * k[0]) * k[0] + T(8.0) * k[0] * k[1],
100  T(55.0) * (k[0] * k[0]) * (k[0] * k[0]) - T(55.0) * (k[0] * k[0]) * k[1] + T(5.0) * (k[1] * k[1]));
101  }

103  /**
104  * @brief Undistorts radial distortion using the provided distortion coefficients
105  *
106  * @details
107  * Given a radial distance from the optical centre, this applies a polynomial distortion model in order to
108  * approximate an ideal lens. After the radial distance has gone through this function it will approximate the
109  * equivilant radius in an ideal lens projection (depending on which base lens projection you are using).
110  *
111  * @tparam T the scalar type used for calculations and storage (normally one of float or double)
112  *
113  * @param r the radial distance from the optical centre
114  * @param lens the paramters that describe the lens that we are using to project
115  *
116  * @return the undistorted radial distance from the optical centre
117  */
118  template <typename T, typename Lens>
119  inline T distort(const T& r, const Lens& lens) {
120  // Uses the math from the paper
121  // An Exact Formula for Calculating Inverse Radial Lens Distortions
122  // https://www.ncbi.nlm.nih.gov/pmc/articles/PMC4934233/pdf/sensors-16-00807.pdf
123  // These terms have been stripped back to only include k1 and k2 and only uses the first 4 terms
124  // if more are needed in the future go and get them from the original paper
125  // TODO(thouliston) if performance ever becomes an issue, this can be precomputed for the same k values
126  const Eigen::Matrix<T, 4, 1> ik = inverse_coefficients<T>(lens);
127  return r
128  * (1.0 //
129  + ik[0] * (r * r) //
130  + ik[1] * ((r * r) * (r * r)) //
131  + ik[2] * ((r * r) * (r * r)) * (r * r) //
132  + ik[3] * ((r * r) * (r * r)) * ((r * r) * (r * r)) //
133  );
134  }

136  /**
137  * @brief Undistorts radial distortion using the provided distortion coefficients
138  *
139  * @details
140  * Given a radial distance from the optical centre, this applies a polynomial distortion model in order to
141  * approximate an ideal lens. After the radial distance has gone through this function it will approximate the
142  * equivalent radius in an ideal lens projection (depending on which base lens projection you are using).
143  *
144  * @tparam T the scalar type used for calculations and storage (normally one of float or double)
145  *
146  * @param r the radial distance from the optical centre
147  * @param lens the paramters that describe the lens that we are using to project
148  *
149  * @return the undistorted radial distance from the optical centre
150  */
151  template <typename T, typename Lens>
152  inline T undistort(const T& r, const Lens& lens) {
153  const auto& k = lens.k;
154  // These parenthesis are important as they allow the compiler to optimise further
155  return r * (1.0 + k[0] * (r * r) + k[1] * (r * r) * (r * r));
156  }

158  /**
159  * @brief Projects a unit vector into a pixel coordinate while working out which lens model to use via the lens
160  * parameters.
161  *
162  * @details
163  * This function expects a unit vector in camera space. For this camera space is defined as a coordinate system
164  * with the x axis going down the viewing direction of the camera, y is to the left of the image, and z is up in
165  * the resulting image. The pixel coordinate that results will have (0,0) at the top left of the image, with x
166  * to the right and y down.
167  *
168  * @tparam T the scalar type used for calculations and storage (normally one of float or double)
169  *
170  * @param p the unit vector to project
171  * @param lens the paramters that describe the lens that we are using to project
172  *
173  * @return a pixel coordinate that this vector projects into
174  */
175  template <typename T, int options, int max_rows_at_compile_time, int max_cols_at_compile_time, typename Lens>
176  Eigen::Matrix<T, 2, 1> project(
177  const Eigen::Matrix<T, 3, 1, options, max_rows_at_compile_time, max_cols_at_compile_time>& ray,
178  const Lens& lens,
179  const Eigen::Matrix<T, 2, 1>& dimensions) {
180 
181  const T f = lens.focal_length;
182  const T theta = std::acos(ray.x());
183  const T rsin_theta = T(1) / std::sqrt(T(1) - ray.x() * ray.x());
184  T r_u;
185  switch (lens.projection.value) {
186  case Lens::Projection::RECTILINEAR: r_u = rectilinear::r(theta, f); break;
187  case Lens::Projection::EQUISOLID: r_u = equisolid::r(theta, f); break;
188  case Lens::Projection::EQUIDISTANT: r_u = equidistant::r(theta, f); break;
189  default: throw std::runtime_error("Cannot project: Unknown lens type"); break;
190  }
191  // If the dimensions are not width-normalised then we need to normalise r_u and then un-normalise r_d
192  const T r_d = distort(r_u, lens);
193 
194  // Work out our pixel coordinates as a 0 centred image with x to the left and y up (screen space)
195  // Sometimes x is greater than one due to floating point error, this almost certainly means that we are
196  // facing directly forward
197  Eigen::Matrix<T, 2, 1> screen =
198  ray.x() >= T(1) ? Eigen::Matrix<T, 2, 1>::Zero()
199  : Eigen::Matrix<T, 2, 1>(r_d * ray.y() * rsin_theta, r_d * ray.z() * rsin_theta);
200 
201  // Apply our offset to move into image space (0 at top left, x to the right, y down)
202  // Then apply the offset to the centre of our lens
203  return (dimensions * T(0.5)) - screen - lens.centre.template cast<T>();
204  }

206  /**
207  * @brief Unprojects a pixel coordinate into a unit vector working out which lens model to use via the lens
208  * parameters.
209  *
210  * @details
211  * This function expects a pixel coordinate having (0,0) at the top left of the image, with x to the right and
212  * y down. It will then convert this into a unit vector in camera space. For this camera space is defined as a
213  * coordinate system with the x axis going down the viewing direction of the camera, y is to the left of the
214  * image, and z is up.
215  *
216  * @tparam T the scalar type used for calculations and storage (normally one of float or double)
217  *
218  * @param px the pixel coordinate to unproject
219  * @param lens the parameters that describe the lens that we are using to unproject
220  * @param dimensions the dimensions of the image
221  *
222  * @return the unit vector that this pixel represents in camera space
223  */
224  template <typename T, typename Lens>
225  Eigen::Matrix<T, 3, 1> unproject(const Eigen::Matrix<T, 2, 1>& px,
226  const Lens& lens,
227  const Eigen::Matrix<T, 2, 1>& dimensions) {
228 
229  Eigen::Matrix<T, 2, 1> screen = (dimensions * T(0.5)) - px - lens.centre.template cast<T>();
230 
231  // Perform the unprojection math
232  const T f = lens.focal_length;
233  const T r_d = screen.norm();
234  if (r_d == T(0)) {
235  return Eigen::Matrix<T, 3, 1>::UnitX();
236  }
237  // If the dimensions are not width-normalised then we need to normalise r_d and then un-normalise r_u
238  const T r_u = undistort(r_d, lens);
239  T theta;
240  switch (lens.projection.value) {
241  case Lens::Projection::RECTILINEAR: theta = rectilinear::theta(r_u, f); break;
242  case Lens::Projection::EQUISOLID: theta = equisolid::theta(r_u, f); break;
243  case Lens::Projection::EQUIDISTANT: theta = equidistant::theta(r_u, f); break;
244  default: throw std::runtime_error("Cannot project: Unknown lens type"); break;
245  }
246 
247  const T sin_theta = std::sin(theta);
248 
249  return Eigen::Matrix<T, 3, 1>(std::cos(theta), sin_theta * screen.x() / r_d, sin_theta * screen.y() / r_d);
250  }
