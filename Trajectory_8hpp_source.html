111  /// @brief Build splines between waypoints for the specified dimension
112  void build_splines_for_dimension(TrajectoryDimension dimension) {
113  auto& dimensions_splines = splines[dimension];
114  auto& dimensions_waypoints = waypoints[dimension];
115  auto& dimensions_timepoints = timepoints[dimension];
116 
117  // If there are less than 2 waypoints, skip building the splines
118  if (dimensions_waypoints.size() < 2) {
119  return;
120  }
121 
122  dimensions_splines.clear();
123  for (size_t i = 0; i < dimensions_waypoints.size() - 1; ++i) {
124  Scalar segment_duration = dimensions_timepoints[i + 1] - dimensions_timepoints[i];
125  dimensions_splines.emplace_back(dimensions_waypoints[i], dimensions_waypoints[i + 1], segment_duration);
126  }
127  }

129  /// @brief Evaluate a specific dimension at the given time
130  Scalar eval(TrajectoryDimension dimension, Scalar time) const {
131  // Determine which segment the time falls into
132  size_t segment_idx = 0;
133  while (segment_idx < timepoints[dimension].size() && time > timepoints[dimension][segment_idx + 1]) {
134  segment_idx++;
135  }
136 
137  if (segment_idx < timepoints[dimension].size()) {
138  Scalar segment_start_time = timepoints[dimension][segment_idx];
139  return splines[dimension][segment_idx].position(time - segment_start_time);
140  }
141  else {
142  // If time is beyond the end of the trajectory, return the end value.
143  return splines[dimension].back().position(timepoints[dimension].back());
144  }
145  }
