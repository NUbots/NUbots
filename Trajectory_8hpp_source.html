66  template <typename Scalar>
67  class Trajectory {
68  public:
69  // Add waypoint for the specified dimension
70  void add_waypoint(TrajectoryDimension dimension,
71  Scalar timepoint,
72  const Scalar position,
73  const Scalar velocity = 0,
74  const Scalar acceleration = 0) {
75  waypoints[dimension].push_back(Eigen::Matrix<Scalar, 3, 1>(position, velocity, acceleration));
76  timepoints[dimension].push_back(timepoint);
77  build_splines_for_dimension(dimension);
78  }
79 
80  // Add waypoint for all dimensions
81  void add_waypoint(const Waypoint<Scalar>& waypoint) {
82  add_waypoint(X, waypoint.time_point, waypoint.position(0), waypoint.velocity(0), waypoint.acceleration(0));
83  add_waypoint(Y, waypoint.time_point, waypoint.position(1), waypoint.velocity(1), waypoint.acceleration(1));
84  add_waypoint(Z, waypoint.time_point, waypoint.position(2), waypoint.velocity(2), waypoint.acceleration(2));
85  add_waypoint(ROLL,
86  waypoint.time_point,
87  waypoint.orientation(0),
88  waypoint.angular_velocity(0),
89  waypoint.angular_acceleration(0));
90  add_waypoint(PITCH,
91  waypoint.time_point,
92  waypoint.orientation(1),
93  waypoint.angular_velocity(1),
94  waypoint.angular_acceleration(1));
95  add_waypoint(YAW,
96  waypoint.time_point,
97  waypoint.orientation(2),
98  waypoint.angular_velocity(2),
99  waypoint.angular_acceleration(2));
100  }
101 
102  /// @brief Get position (x,y,z) at a given time
103  Eigen::Matrix<Scalar, 3, 1> position(Scalar time) const {
104  return Eigen::Matrix<Scalar, 3, 1>(eval(X, time), eval(Y, time), eval(Z, time));
105  }
106 
107  /// @brief Get orientation (roll,pitch,yaw) at a given time
108  Eigen::Matrix<Scalar, 3, 1> orientation(Scalar time) const {
109  return Eigen::Matrix<Scalar, 3, 1>(eval(ROLL, time), eval(PITCH, time), eval(YAW, time));
110  }
111 
112  /// @brief Get pose at a given time
113  Eigen::Transform<Scalar, 3, Eigen::Isometry> pose(Scalar time) const {
114  Eigen::Transform<Scalar, 3, Eigen::Isometry> pose;
115  pose.translation() = position(time);
116  pose.linear() = Eigen::AngleAxis<Scalar>(orientation(time)(2), Eigen::Matrix<Scalar, 3, 1>::UnitZ())
117  * Eigen::AngleAxis<Scalar>(orientation(time)(1), Eigen::Matrix<Scalar, 3, 1>::UnitY())
118  * Eigen::AngleAxis<Scalar>(orientation(time)(0), Eigen::Matrix<Scalar, 3, 1>::UnitX())
119  .toRotationMatrix();
120  return pose;
121  }

123  /// @brief Clear all splines, waypoints, and timepoints for all dimensions
124  void clear() {
125  for (auto& spline : splines) {
126  spline.clear();
127  }
128  for (auto& waypoint : waypoints) {
129  waypoint.clear();
130  }
131  for (auto& timepoint : timepoints) {
132  timepoint.clear();
133  }
134  }
135 
136  private:
137  /// @brief Build splines between waypoints for the specified dimension
138  void build_splines_for_dimension(TrajectoryDimension dimension) {
139  auto& dimensions_splines = splines[dimension];
140  auto& dimensions_waypoints = waypoints[dimension];
141  auto& dimensions_timepoints = timepoints[dimension];
142 
143  // If there are less than 2 waypoints, skip building the splines
144  if (dimensions_waypoints.size() < 2) {
145  return;
146  }
147 
148  dimensions_splines.clear();
149  for (size_t i = 0; i < dimensions_waypoints.size() - 1; ++i) {
150  Scalar segment_duration = dimensions_timepoints[i + 1] - dimensions_timepoints[i];
151  dimensions_splines.emplace_back(dimensions_waypoints[i], dimensions_waypoints[i + 1], segment_duration);
152  }
153  }
154 
155  /// @brief Evaluate a specific dimension at the given time
156  Scalar eval(TrajectoryDimension dimension, Scalar time) const {
157  // Determine which segment the time falls into
158  size_t segment_idx = 0;
159  while (segment_idx < timepoints[dimension].size() && time > timepoints[dimension][segment_idx + 1]) {
160  segment_idx++;
161  }
162 
163  if (segment_idx < timepoints[dimension].size()) {
164  Scalar segment_start_time = timepoints[dimension][segment_idx];
165  return splines[dimension][segment_idx].position(time - segment_start_time);
166  }
167  else {
168  // If time is beyond the end of the trajectory, return the end value.
169  return splines[dimension].back().position(timepoints[dimension].back());
170  }
171  }

173  /// @brief Array of spline vectors for x, y, z, roll, pitch, and yaw.
174  std::array<std::vector<QuinticSpline<Scalar>>, 6> splines;
175 
176  /// @brief Array of waypoint vectors for x, y, z, roll, pitch, and yaw.
177  std::array<std::vector<Eigen::Matrix<Scalar, 3, 1>>, 6> waypoints;
178 
179  /// @brief Array of timepoints for x, y, z, roll, pitch, and yaw.
180  std::array<std::vector<Scalar>, 6> timepoints;
181  };
