112  struct GameControllerPacket {
113  std::array<char, 4> header; // header to identify the structure
114  uint16_t version; // version of the data structure
115  uint8_t packet_number; // number incremented with each packet sent (with wraparound)
116  uint8_t players_per_team; // the number of players on a team
117  GameType game_type; // type of the game (GAME_ROUNDROBIN, GAME_PLAYOFF, GAME_DROPIN)
118  State state; // state of the game (STATE_READY, STATE_PLAYING, etc)
119  bool first_half; // 1 = game in first half, 0 otherwise
120  uint8_t kick_off_team; // the team number of the next team to kick off or DROPBALL
121  Mode mode; // extra state information - (STATE2_NORMAL, STATE2_PENALTYSHOOT, etc)
122  std::array<uint8_t, 4> secondary_state_info; // Extra info on the secondary state
123  TeamColour drop_in_team; // number of team that caused last drop in
124  int16_t drop_in_time; // number of seconds passed since the last drop in. -1 (0xffff) before first dropin
125  uint16_t secs_remaining; // estimate of number of seconds remaining in the half
126  uint16_t secondary_time; // number of seconds shown as secondary time (remaining ready, until free ball, etc)
127  std::array<Team, 2> teams;
128  };
