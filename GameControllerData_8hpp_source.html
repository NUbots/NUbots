104  struct GameControllerPacket {
105  std::array<char, 4> header; // header to identify the structure
106  uint16_t version; // version of the data structure
107  uint8_t packetNumber; // number incremented with each packet sent (with wraparound)
108  uint8_t playersPerTeam; // the number of players on a team
109  GameType gameType; // type of the game (GAME_ROUNDROBIN, GAME_PLAYOFF, GAME_DROPIN)
110  State state; // state of the game (STATE_READY, STATE_PLAYING, etc)
111  bool firstHalf; // 1 = game in first half, 0 otherwise
112  uint8_t kickOffTeam; // the team number of the next team to kick off or DROPBALL
113  Mode mode; // extra state information - (STATE2_NORMAL, STATE2_PENALTYSHOOT, etc)
114  std::array<uint8_t, 4> secondaryStateInfo; // Extra info on the secondary state
115  TeamColour dropInTeam; // number of team that caused last drop in
116  int16_t dropInTime; // number of seconds passed since the last drop in. -1 (0xffff) before first dropin
117  uint16_t secsRemaining; // estimate of number of seconds remaining in the half
118  uint16_t secondaryTime; // number of seconds shown as secondary time (remaining ready, until free ball, etc)
119  std::array<Team, 2> teams;
120  };
