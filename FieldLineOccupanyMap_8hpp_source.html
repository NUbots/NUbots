49  inline OccupancyMap<double> setup_fieldline_distance_map(const FieldDescription& fd, double grid_size) {
50  // Field line distance map (encodes the minimum distance to a field line)
51  OccupancyMap<double> fieldline_distance_map{};
52 
53  // Resize map to the field dimensions
54  int map_width = (fd.dimensions.field_width + 2 * fd.dimensions.border_strip_min_width) / grid_size;
55  int map_length = (fd.dimensions.field_length + 2 * fd.dimensions.border_strip_min_width) / grid_size;
56  fieldline_distance_map.resize(map_width, map_length);
57 
58  // Calculate line width in grid cells
59  int line_width = fd.dimensions.line_width / grid_size;
60 
61  // Add outer field lines
62  int field_x0 = (fd.dimensions.border_strip_min_width) / grid_size;
63  int field_y0 = (fd.dimensions.border_strip_min_width) / grid_size;
64  int field_width = (fd.dimensions.field_width) / grid_size;
65  int field_length = (fd.dimensions.field_length) / grid_size;
66  fieldline_distance_map.add_rectangle(field_x0, field_y0, field_length, field_width, line_width);
67 
68  // Add left goal area box
69  int left_goal_box_x0 = (fd.dimensions.border_strip_min_width) / grid_size;
70  int left_goal_box_y0 =
71  (fd.dimensions.border_strip_min_width + (fd.dimensions.field_width - fd.dimensions.goal_area_width) / 2)
72  / grid_size;
73  int left_goal_box_width = (fd.dimensions.goal_area_length) / grid_size;
74  int left_goal_box_length = (fd.dimensions.goal_area_width) / grid_size;
75  fieldline_distance_map.add_rectangle(left_goal_box_x0,
76  left_goal_box_y0,
77  left_goal_box_width,
78  left_goal_box_length,
79  line_width);
80 
81  // Add right goal area box
82  int right_goal_box_x0 =
83  (fd.dimensions.border_strip_min_width + fd.dimensions.field_length - fd.dimensions.goal_area_length)
84  / grid_size;
85  int right_goal_box_y0 = left_goal_box_y0;
86  int right_goal_box_width = left_goal_box_width;
87  int right_goal_box_length = left_goal_box_length;
88  fieldline_distance_map.add_rectangle(right_goal_box_x0,
89  right_goal_box_y0,
90  right_goal_box_width,
91  right_goal_box_length,
92  line_width);
93 
94  // Add left penalty area box
95  if (fd.dimensions.penalty_area_length != 0 && fd.dimensions.penalty_area_width != 0) {
96  int left_penalty_box_x0 = (fd.dimensions.border_strip_min_width) / grid_size;
97  int left_penalty_box_y0 = (fd.dimensions.border_strip_min_width
98  + (fd.dimensions.field_width - fd.dimensions.penalty_area_width) / 2)
99  / grid_size;
100  int left_penalty_box_width = (fd.dimensions.penalty_area_width) / grid_size;
101  int left_penalty_box_length = (fd.dimensions.penalty_area_length) / grid_size;
102  fieldline_distance_map.add_rectangle(left_penalty_box_x0,
103  left_penalty_box_y0,
104  left_penalty_box_length,
105  left_penalty_box_width,
106  line_width);
107 
108  // Add right penalty area box
109  int right_penalty_box_x0 =
110  (fd.dimensions.border_strip_min_width + fd.dimensions.field_length - fd.dimensions.penalty_area_length)
111  / grid_size;
112  int right_penalty_box_y0 = left_penalty_box_y0;
113  int right_penalty_box_width = left_penalty_box_width;
114  int right_penalty_box_length = left_penalty_box_length;
115  fieldline_distance_map.add_rectangle(right_penalty_box_x0,
116  right_penalty_box_y0,
117  right_penalty_box_length,
118  right_penalty_box_width,
119  line_width);
120  }
121 
122  // Add centre line
123  int centre_line_x0 = (fd.dimensions.border_strip_min_width + fd.dimensions.field_length / 2) / grid_size;
124  int centre_line_y0 = (fd.dimensions.border_strip_min_width) / grid_size;
125  int centre_line_width = (fd.dimensions.field_width) / grid_size;
126  int centre_line_length = (fd.dimensions.line_width) / grid_size;
127  fieldline_distance_map.add_rectangle(centre_line_x0,
128  centre_line_y0,
129  centre_line_length,
130  centre_line_width,
131  line_width);
132 
133 
134  if (fd.dimensions.penalty_mark_distance != 0) {
135  // Add left penalty cross in centre of penalty area
136  int left_penalty_cross_x0 =
137  (fd.dimensions.border_strip_min_width + fd.dimensions.penalty_mark_distance) / grid_size;
138  int left_penalty_cross_y0 =
139  (fd.dimensions.border_strip_min_width + fd.dimensions.field_width / 2) / grid_size;
140  int left_penalty_cross_width = 0.1 / grid_size;
141  fieldline_distance_map.add_cross(left_penalty_cross_x0,
142  left_penalty_cross_y0,
143  left_penalty_cross_width,
144  line_width);
145 
146  // Add right penalty cross in centre of penalty area
147  int right_penalty_cross_x0 = (fd.dimensions.border_strip_min_width + fd.dimensions.field_length
148  - fd.dimensions.penalty_mark_distance)
149  / grid_size;
150  int right_penalty_cross_y0 = left_penalty_cross_y0;
151  int right_penalty_cross_width = 0.1 / grid_size;
152  fieldline_distance_map.add_cross(right_penalty_cross_x0,
153  right_penalty_cross_y0,
154  right_penalty_cross_width,
155  line_width);
156 
157  // Add centre cross in centre of field
158  int centre_cross_x0 =
159  (fd.dimensions.border_strip_min_width + fd.dimensions.field_length / 2) / grid_size + line_width / 2;
160  int centre_cross_y0 = left_penalty_cross_y0;
161  int centre_cross_width = 0.1 / grid_size;
162  fieldline_distance_map.add_cross(centre_cross_x0, centre_cross_y0, centre_cross_width, line_width);
163  }
164 
165 
166  // Add centre circle
167  int centre_circle_x0 = (fd.dimensions.border_strip_min_width + fd.dimensions.field_length / 2) / grid_size;
168  int centre_circle_y0 = (fd.dimensions.border_strip_min_width + fd.dimensions.field_width / 2) / grid_size;
169  int centre_circle_r = (fd.dimensions.center_circle_diameter / 2) / grid_size;
170  fieldline_distance_map.add_circle(centre_circle_x0, centre_circle_y0, centre_circle_r, line_width);
171 
172  // Precompute the distance map
173  fieldline_distance_map.create_distance_map(grid_size);
174  return fieldline_distance_map;
175  }

177  inline std::vector<Landmark> setup_field_landmarks(const FieldDescription& fd) {
178  /// List of landmarks (field intersections rLFf) in field space
179  std::vector<Landmark> landmarks{};
180 
181  // Half dimensions for easier calculation
182  double half_length = fd.dimensions.field_length / 2;
183  double half_width = fd.dimensions.field_width / 2;
184 
185  // Corners of the field
186  landmarks.push_back({Eigen::Vector3d(-half_length, half_width, 0),
187  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
188  landmarks.push_back({Eigen::Vector3d(-half_length, -half_width, 0),
189  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
190  landmarks.push_back({Eigen::Vector3d(half_length, half_width, 0),
191  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
192  landmarks.push_back({Eigen::Vector3d(half_length, -half_width, 0),
193  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
194 
195  // Mid-points of each sideline
196  landmarks.push_back(
197  {Eigen::Vector3d(0, half_width, 0), message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
198  landmarks.push_back(
199  {Eigen::Vector3d(0, -half_width, 0), message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
200 
201  // X intersection at the center
202  landmarks.push_back(
203  {Eigen::Vector3d(0, 0, 0), message::vision::FieldIntersection::IntersectionType::X_INTERSECTION});
204  landmarks.push_back({Eigen::Vector3d(0, fd.dimensions.center_circle_diameter / 2, 0),
205  message::vision::FieldIntersection::IntersectionType::X_INTERSECTION});
206  landmarks.push_back({Eigen::Vector3d(0, -fd.dimensions.center_circle_diameter / 2, 0),
207  message::vision::FieldIntersection::IntersectionType::X_INTERSECTION});
208 
209  if (fd.dimensions.penalty_area_length != 0 && fd.dimensions.penalty_area_width != 0) {
210  // T intersections from the penalty areas
211  landmarks.push_back({Eigen::Vector3d(half_length, fd.dimensions.penalty_area_width / 2, 0),
212  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
213  landmarks.push_back({Eigen::Vector3d(half_length, -fd.dimensions.penalty_area_width / 2, 0),
214  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
215  landmarks.push_back({Eigen::Vector3d(-half_length, fd.dimensions.penalty_area_width / 2, 0),
216  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
217  landmarks.push_back({Eigen::Vector3d(-half_length, -fd.dimensions.penalty_area_width / 2, 0),
218  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
219  }
220 
221  // T intersections from the goal areas
222  landmarks.push_back({Eigen::Vector3d(half_length, fd.dimensions.goal_area_width / 2, 0),
223  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
224  landmarks.push_back({Eigen::Vector3d(half_length, -fd.dimensions.goal_area_width / 2, 0),
225  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
226  landmarks.push_back({Eigen::Vector3d(-half_length, fd.dimensions.goal_area_width / 2, 0),
227  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
228  landmarks.push_back({Eigen::Vector3d(-half_length, -fd.dimensions.goal_area_width / 2, 0),
229  message::vision::FieldIntersection::IntersectionType::T_INTERSECTION});
230 
231  // L intersections from the penalty areas
232  landmarks.push_back(
233  {Eigen::Vector3d(half_length - fd.dimensions.penalty_area_length, fd.dimensions.penalty_area_width / 2, 0),
234  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
235  landmarks.push_back(
236  {Eigen::Vector3d(half_length - fd.dimensions.penalty_area_length, -fd.dimensions.penalty_area_width / 2, 0),
237  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
238  landmarks.push_back(
239  {Eigen::Vector3d(-half_length + fd.dimensions.penalty_area_length, fd.dimensions.penalty_area_width / 2, 0),
240  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
241  landmarks.push_back({Eigen::Vector3d(-half_length + fd.dimensions.penalty_area_length,
242  -fd.dimensions.penalty_area_width / 2,
243  0),
244  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
245 
246  // L intersections from the goal areas
247  landmarks.push_back(
248  {Eigen::Vector3d(half_length - fd.dimensions.goal_area_length, fd.dimensions.goal_area_width / 2, 0),
249  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
250  landmarks.push_back(
251  {Eigen::Vector3d(half_length - fd.dimensions.goal_area_length, -fd.dimensions.goal_area_width / 2, 0),
252  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
253  landmarks.push_back(
254  {Eigen::Vector3d(-half_length + fd.dimensions.goal_area_length, fd.dimensions.goal_area_width / 2, 0),
255  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
256  landmarks.push_back(
257  {Eigen::Vector3d(-half_length + fd.dimensions.goal_area_length, -fd.dimensions.goal_area_width / 2, 0),
258  message::vision::FieldIntersection::IntersectionType::L_INTERSECTION});
259 
260  // X intersections from penalty spots
261  landmarks.push_back({Eigen::Vector3d(half_length - fd.dimensions.penalty_mark_distance, 0, 0),
262  message::vision::FieldIntersection::IntersectionType::X_INTERSECTION});
263  landmarks.push_back({Eigen::Vector3d(-half_length + fd.dimensions.penalty_mark_distance, 0, 0),
264  message::vision::FieldIntersection::IntersectionType::X_INTERSECTION});
265  landmarks.push_back(
266  {Eigen::Vector3d(0, 0, 0), message::vision::FieldIntersection::IntersectionType::X_INTERSECTION});
267  return landmarks;
268  }
