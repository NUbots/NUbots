44  static uint64_t xorShift() {
45  static thread_local uint64_t s[2] = {uint64_t(rand()), uint64_t(rand())};
46 
47  uint64_t s1 = s[0];
48  const uint64_t s0 = s[1];
49  s[0] = s0;
50  s1 ^= s1 << 23;
51  return (s[1] = (s1 ^ s0 ^ (s1 >> 17) ^ (s0 >> 26))) + s0;
52  }

54  template <typename Iterator, typename... Args>
55  static bool regenerateRandomModel(Model& model,
56  const std::array<Iterator, Model::REQUIRED_POINTS + 1>& iterators,
57  Args... args) {
58 
59  std::array<DataPoint, Model::REQUIRED_POINTS> points;
60  for (uint i = 0; i < Model::REQUIRED_POINTS; ++i) {
61  uint range = std::distance(iterators[i], iterators[i + 1]);
62  points[i] = *std::next(iterators[i], xorShift() % range);
63  }
64  return model.regenerate(points, std::forward<Args>(args)...);
65  }

67  /**
68  * Finds an individual ransac model that fits the data
69  *
70  * @return A pair containing an iterator to the start of the remaining set, and the best fitting model
71  */
72  template <typename Iterator, typename... Args>
73  static std::pair<bool, RansacResult<Iterator, Model>> fitModel(
74  std::array<Iterator, Model::REQUIRED_POINTS + 1>& iterators,
75  uint minimumPointsForConsensus,
76  uint maximumIterationsPerFitting,
77  double consensusErrorThreshold,
78  Args... args) {
79 
80  // Check we have enough points in each part
81  for (uint i = 0; i < Model::REQUIRED_POINTS; ++i) {
82  if (std::distance(iterators[i], iterators[i + 1]) < 1) {
83  return std::make_pair(false, RansacResult<Iterator, Model>());
84  }
85  }
86 
87  uint largestConsensus = 0;
88  double bestError = std::numeric_limits<double>::max();
89  Model bestModel;
90  Model model;
91 
92  for (uint i = 0; i < maximumIterationsPerFitting; ++i) {
93 
94  uint consensusSize = 0;
95  double error = 0.0;
96 
97  // Make our model have new set of points
98  if (!regenerateRandomModel(model, iterators, std::forward<Args>(args)...)) {
99  continue;
100  }
101 
102  // Look through all our points and see if they fall within our error bounds
103  for (auto it = iterators.front(); it != iterators.back(); ++it) {
104  if (model.calculateError(*it) < consensusErrorThreshold) {
105  ++consensusSize;
106  error += consensusErrorThreshold;
107  }
108  }
109 
110  // If this model has the largest consensus so far
111  if (consensusSize > largestConsensus or (consensusSize == largestConsensus and error < bestError)) {
112  largestConsensus = consensusSize;
113  bestError = error;
114  bestModel = std::move(model);
115  }
116  }
117 
118  // If the best model we found is good enough
119  if (largestConsensus >= minimumPointsForConsensus) {
120 
121  // If we can, refine the model using the points in the consensus
122  bestModel.refineModel(iterators.front(), iterators.back(), consensusErrorThreshold);
123 
124  // Split off the valid points in each part to the start of it
125  std::array<uint, Model::REQUIRED_POINTS> offsets;
126  for (uint i = 0; i < Model::REQUIRED_POINTS; ++i) {
127 
128  // Split the points from this list off to the start
129  auto newStart = std::partition(iterators[i],
130  iterators[i + 1],
131  [consensusErrorThreshold, bestModel](const DataPoint& point) {
132  return consensusErrorThreshold > bestModel.calculateError(
133  std::forward<const DataPoint&>(point));
134  });
135  offsets[i] = std::distance(newStart, iterators[i + 1]);
136  }
137 
138  // Now we split off all our new points into a global partition
139  auto newFirst = std::stable_partition(iterators.front(),
140  iterators.back(),
141  [consensusErrorThreshold, bestModel](const DataPoint& point) {
142  return consensusErrorThreshold > bestModel.calculateError(
143  std::forward<const DataPoint&>(point));
144  });
145 
146  // We now start with this new first
147  auto start = iterators.front();
148  iterators.front() = newFirst;
149 
150  // Put in our new iterator points
151  for (uint i = 0; i < Model::REQUIRED_POINTS; ++i) {
152  // Move each iterator along
153  iterators[i + 1] = std::next(iterators[i], offsets[i]);
154  }
155 
156  // Return the result
157  return std::make_pair(true, RansacResult<Iterator, Model>(std::move(bestModel), start, newFirst));
158  }
159  else {
160  // We couldn't find a good enough set, return a flag that we failed
161  return std::make_pair(false, RansacResult<Iterator, Model>());
162  }
163  }

165  template <typename Iterator, typename... Args>
166  static std::vector<RansacResult<Iterator, Model>> fitModels(
167  std::array<Iterator, Model::REQUIRED_POINTS + 1> iterators,
168  uint minimumPointsForConsensus,
169  uint maximumIterationsPerFitting,
170  uint maximumFittedModels,
171  double consensusErrorThreshold,
172  Args... args) {
173 
174  std::vector<RansacResult<Iterator, Model>> results;
175  results.reserve(maximumFittedModels);
176 
177  while (results.size() < maximumFittedModels) {
178  bool valid;
179  RansacResult<Iterator, Model> result;
180  std::tie(valid, result) = fitModel(iterators,
181  minimumPointsForConsensus,
182  maximumIterationsPerFitting,
183  consensusErrorThreshold,
184  std::forward<Args>(args)...);
185 
186  // If we have more datapoints left then add this one and continue
187  if (valid) {
188  results.push_back(std::move(result));
189  }
190  else {
191  return results;
192  }
193  }
194 
195  return results;
196  }
197  };
