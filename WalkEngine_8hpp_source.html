49  struct WalkingParameter {
50  // Full walk cycle frequency
51  //(in Hz, > 0)
52  float freq = 0.0f;
53  // Length of double support phase in half cycle
54  //(ratio, [0:1])
55  float double_support_ratio = 0.0f;
56  // Lateral distance between the feet center
57  //(in m, >= 0)
58  float foot_distance = 0.0f;
59  // Maximum flying foot height
60  //(in m, >= 0)
61  float foot_rise = 0.0f;
62  // Pause of Z movement on highest point
63  //(single support cycle ratio, [0,1])
64  float foot_z_pause = 0.0f;
65  // Let the foot's downward trajectory end above the ground
66  // this is helpful if the support leg bends
67  //(in m, >= 0)
68  float foot_put_down_z_offset = 0.0f;
69  // Phase time for moving the foot from Z offset to ground,
70  // also used for X and Y since they should not move after contact to the ground
71  //(phase between apex and single support end [0:1])
72  float foot_put_down_phase = 0.0f;
73  // Phase of flying foot apex
74  //(single support cycle phase, [0:1])
75  float foot_apex_phase = 0.0f;
76  // Foot X/Y overshoot in ratio of step length
77  //(ratio, >= 0)
78  float foot_overshoot_ratio = 0.0f;
79  // Foot X/Y overshoot phase
80  //(single support cycle phase, [footApexPhase:1]
81  float foot_overshoot_phase = 0.0f;
82  // Height of the trunk from ground
83  //(in m, > 0)
84  float trunk_height = 0.0f;
85  // Trunk pitch orientation
86  //(in rad)
87  float trunk_pitch = 0.0f;
88  // Phase offset of trunk oscillation
89  //(half cycle phase, [0:1])
90  float trunk_phase = 0.0f;
91  // Trunk forward offset
92  //(in m)
93  float trunk_x_offset = 0.0f;
94  // Trunk lateral offset
95  //(in m)
96  float trunk_y_offset = 0.0f;
97  // Trunk lateral oscillation amplitude ratio
98  //(ratio, >= 0)
99  float trunk_swing = 0.0f;
100  // Trunk swing pause length in phase at apex
101  //(half cycle ratio, [0:1])
102  float trunk_pause = 0.0f;
103  // Trunk forward offset proportional to forward step
104  //(in 1)
105  float trunk_x_offset_p_coef_forward = 0.0f;
106  // Trunk forward offset proportional to rotation step
107  //(in m/rad)
108  float trunk_x_offset_p_coef_turn = 0.0f;
109  // Trunk pitch orientation proportional to forward step
110  //(in rad/m)
111  float trunk_pitch_p_coef_forward = 0.0f;
112  // Trunk pitch orientation proportional to rotation step
113  //(in 1)
114  float trunk_pitch_p_coef_turn = 0.0f;
115  float trunk_y_only_in_double_support = 0.0f;
116  float kick_length = 0.0f;
117  float kick_phase = 0.0f;
118  float foot_put_down_roll_offset = 0.0f;
119  float kick_vel = 0.0f;
120  float pause_duration = 0.0f;
121  float first_step_swing_factor = 0.0f;
122  };

126  /**
127  * QuinticWalkEngine
128  *
129  * Holonomic and open loop walk
130  * generator based on footstep control
131  * and quintic splines in cartesian space.
132  * Expressed all target state in cartesian
133  * space with respect to current support foot
134  */
135  class QuinticWalkEngine {
136  public:
137  QuinticWalkEngine();
138 
139  /**
140  * Return current walk phase
141  * between 0 and 1
142  */
143  [[nodiscard]] float get_phase() const {
144  return phase;
145  }
146 
147  /**
148  * Return current time between
149  * 0 and half period for
150  * trajectories evaluation
151  */
152  [[nodiscard]] float get_trajs_time() const {
153  return phase < 0.5f ? phase / params.freq : (phase - 0.5f) / params.freq;
154  }
155 
156  /**
157  * Get the footstep object.
158  */
159  [[nodiscard]] inline Footstep get_footstep() const {
160  return foot_step;
161  }
162 
163  /**
164  * Return if true if left is current support foot
165  */
166  [[nodiscard]] bool is_left_support() const {
167  return foot_step.is_left_support();
168  }
169 
170  /**
171  * Return true if both feet are currently on the ground
172  */
173  [[nodiscard]] bool is_double_support() const {
174  // returns true if the value of the "is_float_support" spline is currently higher than 0.5
175  // the spline should only have values of 0 or 1
176  return trajs.get(TrajectoryTypes::IS_DOUBLE_SUPPORT).pos(get_trajs_time()) >= 0.5f;
177  }
