34  /**
35  * Footstep

37  * Manage humanoid footstep generation and state
38  */
39  class Footstep {
40  public:
41  Footstep() = delete;
42  /**
43  * Initialization with lateral foot distance and support foot
44  */
45  Footstep(const float& foot_distance, const bool& is_left_support_foot = true);

47  /**
48  * Set the lateral foot distance parameters
49  */
50  constexpr void set_foot_distance(const float& foot_distance_) {
51  foot_distance = foot_distance_;
52  }
53  [[nodiscard]] constexpr float getFootDistance() const {
54  return foot_distance;
55  }
56 
57  /**
58  * Reset to neutral position the current step (not the integrated odometry)
59  */
60  void reset(const bool& is_left_support_foot);
61 
62  // Reset odometry
63  void resetInWorld(const bool& is_left_support_foot);
64 
65  /**
66  * Current support foot
67  */
68  [[nodiscard]] constexpr bool is_left_support() const {
69  return is_left_support_foot;
70  }
71 
72  /**
73  * Starting position of current flying foot in support foot frame
74  */
75  [[nodiscard]] const Eigen::Vector3f& get_last() const {
76  return support_to_last;
77  }
78 
79  /**
80  * Target pose of current flying foot in support foot frame
81  */
82  [[nodiscard]] const Eigen::Vector3f& get_next() const {
83  return support_to_next;
84  }
85 
86  /**
87  * Returns the odometry change of the current step.
88  */
89  [[nodiscard]] const Eigen::Vector3f& getOdom() const;
90 
91  /**
92  * Left and right, current or next pose of foot in world initial frame
93  */
94  [[nodiscard]] const Eigen::Vector3f& getLeft() const {
95  return left_in_world;
96  }
97  [[nodiscard]] const Eigen::Vector3f& getRight() const {
98  return right_in_world;
99  }
100 
101  /**
102  * Set the target pose of current support foot during next support phase and update support foot.
103  * The target foot pose diff is given with respect to next support foot pose (current flying foot
104  * target).
105  */
106  void stepFromSupport(const Eigen::Vector3f& diff);
107 
108  /**
109  * Set target pose of current support foot using diff orders.
110  * Zero vector means in place walking.
111  * Special handle of lateral and turn step to avoid foot collision.
112  */
113  void step_from_orders(const Eigen::Vector3f& diff);
114 
115  private:
116  /**
117  * Static lateral distance between the feet
118  */
119  float foot_distance;
120 
121  /**
122  * Current support foot
123  * (left or right)
124  */
125  bool is_left_support_foot;
126 
127  /**
128  * Pose diff [dx, dy, dtheta] from support foot to flying foot last and next position
129  */
130  Eigen::Vector3f support_to_last = Eigen::Vector3f::Zero();
131  Eigen::Vector3f support_to_next = Eigen::Vector3f::Zero();
132 
133  /**
134  * Pose integration of left and right foot in initial frame.
135  * Set at "future" state taking into account next expected fot pose.
136  */
137  Eigen::Vector3f left_in_world = Eigen::Vector3f::Zero();
138  Eigen::Vector3f right_in_world = Eigen::Vector3f::Zero();
139  };
