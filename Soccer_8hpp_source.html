50  /// @brief Smart enum for the robot's position
51  struct Position {
52  enum Value { STRIKER, GOALIE, DEFENDER, DYNAMIC };
53  Value value = Value::STRIKER;
54 
55  Position() = default;
56  Position(Value value) : value(value) {}
57  Position(std::string const& str) {
58  // clang-format off
59  if (str == "STRIKER") { value = Value::STRIKER; }
60  else if (str == "GOALIE") { value = Value::GOALIE; }
61  else if (str == "DEFENDER") { value = Value::DEFENDER; }
62  else if (str == "DYNAMIC") { value = Value::DYNAMIC; }
63  else { throw std::runtime_error("Invalid robot position"); }
64  // clang-format on
65  }
66 
67  operator int() const {
68  return value;
69  }
70 
71  operator std::string() const {
72  switch (value) {
73  case Value::STRIKER: return "Striker";
74  case Value::GOALIE: return "Goalie";
75  case Value::DEFENDER: return "Defender";
76  case Value::DYNAMIC: return "Dynamic";
77  default: return "Unknown";
78  }
79  }
80  };
