28  template <typename Scalar, int NumStates, int NumInputs, int NumMeasurements>
29  class KalmanFilter {
30  public:
31  // Dimension types for vectors and matrices
32  using StateVec = Eigen::Matrix<Scalar, NumStates, 1>;
33  using StateMat = Eigen::Matrix<Scalar, NumStates, NumStates>;
34  using InputVec = Eigen::Matrix<Scalar, NumInputs, 1>;
35  using InputMat = Eigen::Matrix<Scalar, NumInputs, NumInputs>;
36  using MeasVec = Eigen::Matrix<Scalar, NumMeasurements, 1>;
37  using MeasMat = Eigen::Matrix<Scalar, NumMeasurements, NumMeasurements>;
38  using GainMat = Eigen::Matrix<Scalar, NumStates, NumMeasurements>;
39  using MeasModelMat = Eigen::Matrix<Scalar, NumMeasurements, NumStates>;
40  using InputModelMat = Eigen::Matrix<Scalar, NumStates, NumInputs>;

62  /**
63  * @brief Update the filter parameters.
64  *
65  * @param[in] state_transition_matrix
66  * @param[in] input_model_matrix
67  * @param[in] measurement_model_matrix
68  * @param[in] process_noise_covariance

71  void update(StateMat state_transition_matrix,
72  InputModelMat input_model_matrix,
73  MeasModelMat measurement_model_matrix,
74  StateMat process_noise_covariance,
75  MeasMat measurement_noise_covariance) {
76  A = state_transition_matrix;
77  B = input_model_matrix;
78  C = measurement_model_matrix;
79  Q = process_noise_covariance;
80  R = measurement_noise_covariance;
81  }

83  /**
84  * @brief Resets the filter to a new initial state and covariance.
85  *
86  * @param[in] initial_mean
87  * @param[in] initial_covariance
88  * @return Eigen::ComputationInfo indicating success or failure of the reset.
89  */
90  Eigen::ComputationInfo reset(StateVec initial_mean, StateMat initial_covariance) {
91  state = initial_mean;
92  covariance = initial_covariance;
93  return Eigen::Success;
94  }
95 
96  /**
97  * @brief Performs a time update on the state and covariance using the system model and control input.
98  *
99  * @param[in] control_input
100  * @param[in] dt
101  */
102  void time(const InputVec& control_input, const Scalar& dt) {
103  Eigen::Matrix<Scalar, NumStates, NumStates> Ad;
104  Eigen::Matrix<Scalar, NumStates, NumInputs> Bd;
105  zero_order_hold(Ad, Bd, dt);
106  state = Ad * state + Bd * control_input;
107  covariance = Ad * covariance * Ad.transpose() + Q;
108  }
109 
110  /**
111  * @brief Performs a measurement update on the state and covariance using the measurement model and measurement.
112  *
113  * @param[in] measurement
114  */
115  void measure(const MeasVec& measurement) {
116  // Calculate the optimal Kalman gain
117  GainMat K = covariance * C.transpose() * (C * covariance * C.transpose() + R).inverse();
118  // Update the state estimate
119  state += K * (measurement - C * state);
120  // Update the covariance estimate
121  covariance = (StateMat::Identity() - K * C) * covariance * (StateMat::Identity() - K * C).transpose()
122  + K * R * K.transpose();
123  }
124 
125  /// @brief Get the current state estimate
126  const StateVec& get_state() const {
127  return state;
128  }
129 
130  /// @brief Get the current covariance
131  const StateMat& get_covariance() const {
132  return covariance;
133  }
134 
135  private:
136  /**
137  * @brief Discretises the system model using a zero-order hold method.
138  *
139  * @param[out] Ad Discretised state transition matrix.
140  * @param[out] Bd Discretised input model matrix.
141  * @param[in] dt Timestep to discretise with.
142  */
143  void zero_order_hold(Eigen::Matrix<Scalar, NumStates, NumStates>& Ad,
144  Eigen::Matrix<Scalar, NumStates, NumInputs>& Bd,
145  const Scalar& dt) {
146  Ad = Eigen::Matrix<Scalar, NumStates, NumStates>::Identity() + dt * A;
147  Bd = dt * B;
148  }
149 
150  /// @brief Current estimate of the system state
151  StateVec state = StateVec::Zero();
152 
153  /// @brief Current covariance of the system state
154  StateMat covariance = StateMat::Identity();
155

157  StateMat A = StateMat::Identity();

163  MeasModelMat C = MeasModelMat::Zero();
164 
165  /// @brief Process noise covariance matrix
166  StateMat Q = StateMat::Identity();
167 
168  /// @brief Measurement noise covariance matrix
169  MeasMat R = MeasMat::Identity();
170  };
