36  template <typename Scalar, int NumStates, int NumInputs, int NumMeasurements>
37  class KalmanFilter {
38  public:
39  // Dimension types for vectors and matrices
40  using StateVec = Eigen::Matrix<Scalar, NumStates, 1>;
41  using StateMat = Eigen::Matrix<Scalar, NumStates, NumStates>;
42  using InputVec = Eigen::Matrix<Scalar, NumInputs, 1>;
43  using InputMat = Eigen::Matrix<Scalar, NumInputs, NumInputs>;
44  using MeasVec = Eigen::Matrix<Scalar, NumMeasurements, 1>;
45  using MeasMat = Eigen::Matrix<Scalar, NumMeasurements, NumMeasurements>;
46  using GainMat = Eigen::Matrix<Scalar, NumStates, NumMeasurements>;
47  using MeasModelMat = Eigen::Matrix<Scalar, NumMeasurements, NumStates>;
48  using InputModelMat = Eigen::Matrix<Scalar, NumStates, NumInputs>;

51  KalmanFilter() = default;

53  /// @brief Construct a KalmanFilter with the given parameters
54  KalmanFilter(StateVec initial_state,
55  StateMat initial_covariance,

58  MeasModelMat measurement_model_matrix,

70  /**
71  * @brief Update the filter parameters.
72  *
73  * @param[in] state_transition_matrix
74  * @param[in] input_model_matrix
75  * @param[in] measurement_model_matrix
76  * @param[in] process_noise_covariance
77  * @param[in] measurement_noise_covariance
78  */
79  void update(StateMat state_transition_matrix,
80  InputModelMat input_model_matrix,
81  MeasModelMat measurement_model_matrix,
82  StateMat process_noise_covariance,
83  MeasMat measurement_noise_covariance) {
84  A = state_transition_matrix;
85  B = input_model_matrix;
86  C = measurement_model_matrix;
87  Q = process_noise_covariance;
88  R = measurement_noise_covariance;
89  }
90 
91  /**
92  * @brief Resets the filter to a new initial state and covariance.
93  *
94  * @param[in] initial_mean
95  * @param[in] initial_covariance
96  * @return Eigen::ComputationInfo indicating success or failure of the reset.
97  */
98  Eigen::ComputationInfo reset(StateVec initial_mean, StateMat initial_covariance) {
99  state = initial_mean;
100  covariance = initial_covariance;
101  return Eigen::Success;
102  }
103 
104  /**
105  * @brief Performs a time update on the state and covariance using the system model and control input.
106  *
107  * @param[in] control_input
108  * @param[in] dt
109  */
110  void time(const InputVec& control_input, const Scalar& dt) {
111  Eigen::Matrix<Scalar, NumStates, NumStates> Ad;
112  Eigen::Matrix<Scalar, NumStates, NumInputs> Bd;
113  zero_order_hold(Ad, Bd, dt);
114  state = Ad * state + Bd * control_input;
115  covariance = Ad * covariance * Ad.transpose() + Q;
116  }
117 
118  /**
119  * @brief Performs a measurement update on the state and covariance using the measurement model and measurement.
120  *
121  * @param[in] measurement
122  */
123  void measure(const MeasVec& measurement) {
124  // Calculate the optimal Kalman gain
125  GainMat K = covariance * C.transpose() * (C * covariance * C.transpose() + R).inverse();
126  // Update the state estimate
127  state += K * (measurement - C * state);
128  // Update the covariance estimate
129  covariance = (StateMat::Identity() - K * C) * covariance * (StateMat::Identity() - K * C).transpose()
130  + K * R * K.transpose();
131  }
132 
133  /// @brief Get the current state estimate
134  const StateVec& get_state() const {
135  return state;
136  }
137 
138  /// @brief Set the current state estimate
139  void set_state(const StateVec& new_state) {
140  state = new_state;
141  }
142 
143  /// @brief Get the current covariance
144  const StateMat& get_covariance() const {
145  return covariance;
146  }

148  private:
149  /**
150  * @brief Discretises the system model using a zero-order hold method.
151  *
152  * @param[out] Ad Discretised state transition matrix.
153  * @param[out] Bd Discretised input model matrix.
154  * @param[in] dt Timestep to discretise with.
155  */
156  void zero_order_hold(Eigen::Matrix<Scalar, NumStates, NumStates>& Ad,
157  Eigen::Matrix<Scalar, NumStates, NumInputs>& Bd,
158  const Scalar& dt) {
159  Ad = Eigen::Matrix<Scalar, NumStates, NumStates>::Identity() + dt * A;
160  Bd = dt * B;
161  }
162 
163  /// @brief Current estimate of the system state
164  StateVec state = StateVec::Zero();
165 
166  /// @brief Current covariance of the system state
167  StateMat covariance = StateMat::Identity();
168 
169  /// @brief State transition matrix
170  StateMat A = StateMat::Identity();
171 
172  /// @brief Input model matrix
173  InputModelMat B = InputModelMat::Zero();
174 
175  /// @brief Measurement model matrix
176  MeasModelMat C = MeasModelMat::Zero();
177 
178  /// @brief Process noise covariance matrix
179  StateMat Q = StateMat::Identity();
180 
181  /// @brief Measurement noise covariance matrix
182  MeasMat R = MeasMat::Identity();
183  };
