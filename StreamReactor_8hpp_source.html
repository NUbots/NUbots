45  /**
46  * This reactor implements the common pattern of connecting to a stream device such as a serial port or a TCP
47  * connection. It then will process bytes one at a time as they come in by calling the parser function provided
48  * by the CRTP T type.
49  *
50  * @tparam T The CRTP type (the type that inherits from this class)
51  * @tparam Parser The parser function that will be called to process the bytes as they come in
52  * The type is expected to have operator()(uint8_t) so either a functor or a lambda can be used
53  * @tparam TimeoutTicks The number of units to wait before considering the device to be disconnected or
54  * non-responsive
55  * @tparam TimeoutPeriod The units that TimeoutTicks is measured in
56  */
57  template <typename T, typename Parser, int TimeoutTicks = 1, typename TimeoutPeriod = std::chrono::seconds>
58  class StreamReactor : public NUClear::Reactor {
59  public:
60  /// @brief Called by the powerplant to build and setup the StreamReactor.
61  explicit StreamReactor(std::unique_ptr<NUClear::Environment> environment) : Reactor(std::move(environment)) {
62 
63  /**************************
64  * USER TRIGGER HOOKS *
65  **************************/
66  on<ConnectTCP>().then("Initiate a TCP Connection", [this](const ConnectTCP& c) {
67  // Setup the connection details
68  remote = c;
69 
70  emit(std::make_unique<Reconnect>());
71  });
72 
73  on<ConnectSerial>().then("Initiate a Serial Connection", [this](const ConnectSerial& c) {
74  // Setup the connection details
75  remote = c;
76 
77  emit(std::make_unique<Reconnect>());
78  });
79 
80  on<Disconnect>().then("Disconnecting", [this] {
81  // Allow pre-disconnection settings to be applied
82  emit(std::make_unique<PreDisconnect>(false));
83  emit(std::make_unique<Do<PreDisconnect>>(false));
84  });
85 
86  on<Reconnect>().then("Reconnecting", [this](const Reconnect& rc) {
87  // Log the reason we are reconnecting if it's not an empty string.
88  // empty string means we are just starting up
89  if (!rc.reason.empty()) {
90  log<WARN>(rc.reason);
91  }
92 
93  if (connection_valid()) {
94  // Initiate the reconnection
95  emit(std::make_unique<PreDisconnect>(true));
96  emit(std::make_unique<Do<PreDisconnect>>(true));
97  }
98  else {
99  // Initiate the connection
100  emit(std::make_unique<PreConnect>());
101  emit(std::make_unique<Do<PreConnect>>());
102  }
103  });
104 
105  on<TransmitData>().then("Transmit data to device", [this](const TransmitData& t) {
106  // If there is not already a connection established, then calling reconnect is an error
107  if (std::holds_alternative<std::monostate>(connection) || fd < 0) {
108  throw std::runtime_error("Can't transmit to no one.");
109  }
110 
111  // Write data to the device
112  size_t bytes_written = 0;
113  while (bytes_written < t.data.size()) {
114  ssize_t written = ::write(fd, t.data.data() + bytes_written, t.data.size() - bytes_written);
115  if (written < 0) {
116  throw std::system_error(errno,
117  std::system_category(),
118  fmt::format("Failed to transmit data to device at {} with error {}",
119  device_description(),
120  errno));
121  }
122  bytes_written += written;
123  }
124  if (log_level <= TRACE) {
125  std::stringstream debug_string{};
126  for (const auto& byte : t.data) {
127  debug_string << std::isprint(byte) ? std::string(1, char(byte))
128  : fmt::format("\\{:#04x} ", byte);
129  }
130  log<TRACE>("Wrote:", debug_string.str());
131  }
132  });
133 
134 
135  /**************************
136  * INTERNAL TRIGGER CHAIN *
137  **************************/
138  on<Do<PreConnect>>().then([this] { emit(std::make_unique<DoConnect>()); });
139 
140  on<DoConnect>().then([this] {
141  // Increase the connection attempt number to stop any existing configuration loops
142  ++connection_attempt;
143 
144  try {
145  // Initiate the connection and get the file descriptor for the IO handle
146  if (std::holds_alternative<ConnectTCP>(remote)) {
147  const auto& r = std::get<ConnectTCP>(remote);
148  connection = TCPConnection(r.host, r.port);
149 
150  auto& tcp = std::get<TCPConnection>(connection);
151  tcp.fd = utility::network::connect(tcp.host, tcp.port);
152  fd = tcp.fd.get();
153  }
154  else if (std::holds_alternative<ConnectSerial>(remote)) {
155  const auto& r = std::get<ConnectSerial>(remote);
156  connection = SerialConnection(r.device, r.baud_rate);
157 
158  auto& serial = std::get<SerialConnection>(connection);
159  serial.uart = utility::io::uart(serial.device, serial.baud_rate);
160  fd = serial.uart.native_handle();
161  }
162  else {
163  remote = std::monostate();
164  connection = std::monostate();
165  fd = -1;
166  throw std::runtime_error("No valid connection settings found.");
167  }
168 
169  // Something must have gone wrong and we are not connected
170  if (!connection_valid()) {
171  throw std::runtime_error("Failed to connect to device");
172  }
173 
174  // Recreate the IO reaction
175  auto flags = IO::READ | IO::CLOSE | IO::ERROR;
176  io_handle = on<IO>(fd, flags).then("Read Stream", [this](const IO::Event& event) {
177  if ((event.events & IO::READ) != 0) {
178  process(event.fd);
179  }
180  if ((event.events & IO::ERROR) != 0) {
181  // Unbind the handle, so it does not read stream in later iterations
182  io_handle.unbind();
183  // Start a new connection
184  emit(std::make_unique<Reconnect>("An invalid state occurred"));
185  }
186  if ((event.events & IO::CLOSE) != 0) {
187  // Unbind the handle, so it does not read stream in later iterations
188  io_handle.unbind();
189  // Start a new connection
190  emit(std::make_unique<Reconnect>("The device hung up"));
191  }
192  if ((event.events & ~(IO::READ | IO::ERROR | IO::CLOSE)) != 0) {
193  log<ERROR>(
194  utility::strutil::dedent(fmt::format("Unknown IO event on {}valid connection: {}",
195  connection_valid() ? "" : "in",
196  event.events)));
197  }
198  });
199 
200  // Allow post-connection settings to be applied
201  emit(std::make_unique<PostConnect>(connection_attempt));
202 
203  log<INFO>(fmt::format("Successfully connected to the device at {}", device_description()));
204  }
205  catch (const std::runtime_error& ex) {
206  log<WARN>(fmt::format("Failed to connect to the device at {}\nError: {}",
207  device_description(),
208  ex.what()));
209  }
210  });
211 
212  on<Do<PreDisconnect>>().then([this](const Do<PreDisconnect>& pd) {
213  // Allow post-disconnection settings to be applied
214  emit(std::make_unique<DoDisconnect>(pd.reconnect));
215  });
216 
217  on<DoDisconnect>().then("Disconnect from device", [this](const DoDisconnect& d) {
218  // Unbind the reaction handle if it is bound
219  if (io_handle) {
220  io_handle.unbind();
221  }
222 
223  // Reset the connection part of the variant
224  if (!std::holds_alternative<std::monostate>(connection)) {
225  connection = std::monostate();
226  }
227 
228  // Clear old file descriptor
229  fd = -1;
230 
231  // Allow post-disconnection settings to be applied
232  emit(std::make_unique<PostDisconnect>(d.reconnect));
233  emit(std::make_unique<Do<PostDisconnect>>(d.reconnect));
234  });
235 
236  on<Do<PostDisconnect>>().then([this](const Do<PostDisconnect>& pd) {
237  // If a reconnect has been initiated then kick off the connection attempt now
238  if (pd.reconnect) {
239  emit(std::make_unique<PreConnect>());
240  emit(std::make_unique<Do<PreConnect>>());
241  }
242  });
243 
244  // Watchdog to ensure we are getting data from the device
245  on<Watchdog<StreamReactor<T, Parser>, TimeoutTicks, TimeoutPeriod>, Single>().then([this] {
246  auto s = double(TimeoutTicks) * double(TimeoutPeriod::period::num) / double(TimeoutPeriod::period::den);
247  auto [si, unit] = utility::support::si_time(s);
248  emit(std::make_unique<Reconnect>(fmt::format("No activity from the Device at {} in {}{}. Reconnecting",
249  device_description(),
250  si,
251  unit)));
252  });
253  }

458  /**
459  * @brief Process the data received from the device
460  *
461  * @details This function is called when the IO reaction is triggered. It reads the data from the device and
462  * passes it to the parser. Once the parser finds a packet it emits that packet.
463  */
464  void process(const int& fd) {
465  std::stringstream debug_string;
466  // Read bytes one at a time
467  uint8_t byte = 0;
468  while (::read(fd, &byte, 1) == 1) {
469 
470  if (log_level <= TRACE) {
471  debug_string << std::isprint(byte) ? std::string(1, char(byte)) : fmt::format("\\{:#04x} ", byte);
472  }
473 
474  try {
475  // Pass the byte to the parser and emit the packet if it is complete
476  auto packet = parser(byte);
477  if (packet) {
478  // Service the watchdog and emit the packet
479  emit<Scope::WATCHDOG>(ServiceWatchdog<StreamReactor<T, Parser>>());
480  emit(std::move(packet));
481  }
482  }
483  catch (const std::exception& ex) {
484  log<WARN>(ex.what());
485  }
486  }
487  if (log_level <= TRACE) {
488  log<TRACE>("Read:", debug_string.str());
489  }
490  }

497  std::variant<std::monostate, ConnectSerial, ConnectTCP> remote{};
498  /// The handle attached to the IO reaction
499  ReactionHandle io_handle{};
500  /// The file descriptor for the currently active connection
501  int fd = -1;
502  };
