34  template <typename Scalar>
35  class QuinticSpline {
36  public:
37  /**
38  * @brief Construct a quintic spline with duration and start and end waypoints.
39  * @param start_waypoint Start waypoint (start position, start velocity, start acceleration).
40  * @param end_waypoint End waypoint (end position, end velocity, end acceleration).
41  * @param t Time at which to compute the position.
42  */
43  QuinticSpline(const Eigen::Matrix<Scalar, 3, 1> start_waypoint,
44  const Eigen::Matrix<Scalar, 3, 1> end_waypoint,
45  const Scalar& t) {
46  // Compute the coefficients of the quintic polynomial.
47  if (t <= static_cast<Scalar>(1e-5)) {
48  throw std::logic_error("QuinticSpline invalid duration: " + std::to_string(t));
49  }
50  const Scalar t2 = t * t;
51  const Scalar t3 = t2 * t;
52  const Scalar t4 = t3 * t;
53  const Scalar t5 = t4 * t;
54  coefficients[0] = start_waypoint[0];
55  coefficients[1] = start_waypoint[1];
56  coefficients[2] = start_waypoint[2] / 2;
57  coefficients[3] = -(-end_waypoint[2] * t2 + 3 * start_waypoint[2] * t2 + 8 * end_waypoint[1] * t
58  + 12 * start_waypoint[1] * t - 20 * end_waypoint[0] + 20 * start_waypoint[0])
59  / (2 * t3);
60  coefficients[4] = (-2 * end_waypoint[2] * t2 + 3 * start_waypoint[2] * t2 + 14 * end_waypoint[1] * t
61  + 16 * start_waypoint[1] * t - 30 * end_waypoint[0] + 30 * start_waypoint[0])
62  / (2 * t4);
63  coefficients[5] = -(-end_waypoint[2] * t2 + start_waypoint[2] * t2 + 6 * end_waypoint[1] * t
64  + 6 * start_waypoint[1] * t - 12 * end_waypoint[0] + 12 * start_waypoint[0])
65  / (2 * t5);
66  }
67 
68 
69  /**
70  * @brief Compute the position at a given time.
71  * @param time Time at which to compute the position.
72  * @return Position vector at the given time.
73  */
74  Scalar position(Scalar time) const {
75  Scalar tt = 1;
76  Scalar position = 0;
77  for (size_t i = 0; i < size_t(coefficients.size()); ++i) {
78  position += coefficients[i] * tt;
79  tt *= time;
80  }
81  return position;
82  }
83 
84  /**
85  * @brief Compute the velocity at a given time.
86  * @param t Time at which to compute the velocity.
87  * @return Velocity at the given time.
88  */
89  Scalar velocity(Scalar time) const {
90  Scalar tt = 1;
91  Scalar velocity = 0;
92  for (size_t i = 1; i < size_t(coefficients.size()); ++i) {
93  velocity += coefficients[i] * i * tt;
94  tt *= time;
95  }
96  return velocity;
97  }
98 
99  /**
100  * @brief Compute the acceleration at a given time.
101  * @param t Time at which to compute the acceleration.
102  * @return Acceleration at the given time.
103  */
104  Scalar acceleration(Scalar time) const {
105  Scalar tt = 1;
106  Scalar acceleration = 0;
107  for (size_t i = 2; i < size_t(coefficients.size()); ++i) {
108  acceleration += coefficients[i] * i * (i - 1) * tt;
109  tt *= time;
110  }
111  return acceleration;
112  }
113 
114  private:
115  /// @brief Spline Coefficients
116  Eigen::Matrix<Scalar, 6, 1> coefficients = Eigen::Matrix<Scalar, 6, 1>::Zero();
117  };
