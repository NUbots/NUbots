145  /**
146  * @brief Get the left or right foot pose at the current time in the torso {t} frame.
147  * @param limb Limb ID of foot to get pose of.
148  * @return Swing foot pose at time t.
149  */
150  Iso3 get_foot_pose(const LimbID& limb) const {
151  // Return the desired pose of the specified foot
152  if (limb == LimbID::LEFT_LEG)
153  return phase == LEFT ? get_torso_pose().inverse() : get_torso_pose().inverse() * get_swing_foot_pose();
154  if (limb == LimbID::RIGHT_LEG)
155  return phase == LEFT ? get_torso_pose().inverse() * get_swing_foot_pose() : get_torso_pose().inverse();
156  throw std::runtime_error("Invalid Limb ID");
157  }
158 
159  /**
160  * @brief Get the left or right foot pose at the given time in the torso {t} frame.
161  * @param t Time.
162  * @param limb Limb ID of foot to get pose of.
163  * @return Swing foot pose at time t.
164  */
165  Iso3 get_foot_pose(Scalar t, const LimbID& limb) const {
166  // Return the desired pose of the specified foot
167  if (limb == LimbID::LEFT_LEG)
168  return phase == LEFT ? get_torso_pose(t).inverse()
169  : get_torso_pose(t).inverse() * get_swing_foot_pose(t);
170  if (limb == LimbID::RIGHT_LEG)
171  return phase == LEFT ? get_torso_pose(t).inverse() * get_swing_foot_pose(t)
172  : get_torso_pose(t).inverse();
173  throw std::runtime_error("Invalid Limb ID");
174  }

176  /**
177  * @brief Reset walk engine time and initial poses.
178  */
179  void reset() {
180  // Initialize swing foot pose
181  Hps_start.translation() = Vec3(0.0, get_foot_width_offset(), 0.0);
182  Hps_start.linear().setIdentity();
183 
184  // Initialize torso pose
185  Hpt_start.translation() = p.torso_position_offset + Vec3(0.0, get_foot_width_offset() / 2, p.torso_height);
186  Hpt_start.linear() = Eigen::AngleAxis<Scalar>(p.torso_pitch, Vec3::UnitY()).toRotationMatrix();
187 
188  // Start time at end of step period to avoid taking a step when starting.
189  t = p.step_period;
190 
191  // Generate trajectories for swing foot and torso with zero velocity target
192  generate_walking_trajectories(Vec3::Zero());
193 
194  // Set engine state to stopped
195  engine_state = WalkState::State::STOPPED;
196  }

198  /**
199  * @brief Run an update of the walk engine, updating the time and engine state and trajectories.
200  * @param dt Time step.
201  * @param velocity_target Requested velocity target (dx, dy, dtheta).
202  * @param sensors_planted_foot_phase Planted foot phase from sensors.
203  * @return Engine state.
204  */
205  WalkState::State update(const Scalar& dt,
206  const Vec3& velocity_target,
207  const WalkState::Phase& sensors_planted_foot_phase) {
208  if (velocity_target.isZero() && t < p.step_period) {
209  // Requested velocity target is zero and we haven't finished taking a step, continue stopping
210  engine_state = WalkState::State::STOPPING;
211  }
212  else if (t >= p.step_period
213  && (velocity_target.isZero() || engine_state.value == WalkState::State::STOPPING)) {
214  // Requested velocity target is zero and we have finished taking a step, remain stopped
215  engine_state = WalkState::State::STOPPED;
216  }
217  else if (!velocity_target.isZero() && engine_state.value == WalkState::State::STOPPED) {
218  // Requested velocity target is non-zero and we are stopped, start walking
219  engine_state = WalkState::State::STARTING;
220  t = 0;
221  }
222 
223  // If the sensors have detected either double support or next foot planted, allow switching the planted foot
224  bool can_switch = sensors_planted_foot_phase != phase || !p.only_switch_when_planted;
225  switch (engine_state.value) {
226  case WalkState::State::STARTING:
227  update_time(dt);
228  // Generate starting trajectories for the torso and swing foot
229  generate_starting_trajectories();
230  if (t >= p.step_period) {
231  engine_state = WalkState::State::WALKING;
232  t = 0;
233  }
234  break;
235  case WalkState::State::WALKING:
236  update_time(dt);
237  // If we are at the end of the step and can switch feet, switch the planted foot and reset time
238  if (t >= p.step_period && can_switch) {
239  switch_planted_foot();
240  }
241  generate_walking_trajectories(velocity_target);
242  break;
243  case WalkState::State::STOPPING:
244  update_time(dt);
245  generate_walking_trajectories(velocity_target);
246  // We do not switch the planted foot here because we want to transition to the stopped state
247  break;
248  case WalkState::State::STOPPED:
249  // We do not update the time here because we want to remain in the stopped state
250  generate_walking_trajectories(velocity_target);
251  break;
252  default: NUClear::log<NUClear::WARN>("Unknown state", engine_state.value);
253  }
254 
255 
256  return engine_state;
257  }

323  /**
324  * @brief Generate torso trajectory for the current step.
325  * @param step Step placement in the planted foot frame.
326  * @param velocity_target Requested velocity target (dx, dy, dtheta).
327  */
328  void generate_torso_trajectory(const Vec3& step, const Vec3& velocity_target) {
329  // Create a waypoint variable to use for all waypoints
330  Waypoint<Scalar> wp;
331 
332  // ******************************** Torso Trajectory ********************************
333  torso_trajectory.clear();
334 
335  // Start waypoint: Current torso position
336  wp.time_point = 0.0;
337  wp.position = Hpt_start.translation();
338  wp.velocity = Vec3::Zero();
339  wp.orientation = mat_to_rpy_intrinsic(Hpt_start.rotation());
340  wp.angular_velocity = Vec3::Zero();
341  torso_trajectory.add_waypoint(wp);
342 
343  // Middle waypoint: Shift torso over planted foot
344  wp.time_point = p.torso_sway_ratio * p.step_period;
345  Scalar torso_offset_y = phase == LEFT ? -p.torso_sway_offset.y() : p.torso_sway_offset.y();
346  wp.position = Vec3(p.torso_sway_offset.x(), torso_offset_y, p.torso_height + p.torso_sway_offset.z())
347  + p.torso_position_offset;
348  wp.velocity = Vec3(velocity_target.x(), velocity_target.y(), 0);
349  wp.orientation = Vec3(0.0, p.torso_pitch, velocity_target.z() * p.torso_sway_ratio * p.step_period);
350  wp.angular_velocity = Vec3(0.0, 0.0, velocity_target.z());
351  torso_trajectory.add_waypoint(wp);
352 
353  // End waypoint: Final torso position, ratio of the next step position
354  wp.time_point = p.step_period;
355  wp.position = Vec3(p.torso_final_position_ratio.x() * step.x(),
356  get_foot_width_offset() / 2 + p.torso_final_position_ratio.y() * step.y(),
357  p.torso_height)
358  + p.torso_position_offset;
359  wp.velocity = Vec3::Zero();
360  wp.angular_velocity = Vec3::Zero();
361  wp.orientation = Vec3(0.0, p.torso_pitch, p.torso_final_position_ratio.z() * step.z());
362  torso_trajectory.add_waypoint(wp);
363  }

365  /**
366  * @brief Generate swing foot trajectory for the current step.
367  * @param step Step placement in the planted foot frame.
368  * @param velocity_target Requested velocity target (dx, dy, dtheta).
369  */
370  void generate_swing_foot_trajectory(const Vec3& step, const Vec3& velocity_target) {
371  // Create a waypoint variable to use for all waypoints
372  Waypoint<Scalar> wp;
373 
374  // ******************************** Swing Foot Trajectory ********************************
375  swing_foot_trajectory.clear();
376 
377  // Start waypoint: Current swing foot position
378  wp.time_point = 0.0;
379  wp.position = Hps_start.translation();
380  wp.velocity = Vec3::Zero();
381  wp.orientation = mat_to_rpy_intrinsic(Hps_start.rotation());
382  wp.angular_velocity = Vec3::Zero();
383  swing_foot_trajectory.add_waypoint(wp);
384 
385  // Middle waypoint: Raise foot to step height
386  wp.time_point = p.step_apex_ratio * p.step_period;
387  wp.position = Vec3(0, get_foot_width_offset(), p.step_height);
388  wp.velocity = Vec3(velocity_target.x(), velocity_target.y(), 0);
389  wp.orientation = Vec3(0.0, 0.0, velocity_target.z() * p.step_apex_ratio * p.step_period);
390  wp.angular_velocity = Vec3(0.0, 0.0, velocity_target.z());
391  swing_foot_trajectory.add_waypoint(wp);
392 
393  // End waypoint: Next foot placement on ground
394  wp.time_point = p.step_period;
395  wp.position = Vec3(step.x(), get_foot_width_offset() + step.y(), 0);
396  wp.velocity = Vec3::Zero();
397  wp.orientation = Vec3(0, 0, step.z());
398  wp.angular_velocity = Vec3::Zero();
399  swing_foot_trajectory.add_waypoint(wp);
400  }

402  /**
403  * @brief Generate walking trajectories for the torso and swing foot.
404  * @param velocity_target Requested velocity target (dx, dy, dtheta).
405  */
406  void generate_walking_trajectories(const Vec3& velocity_target) {
407  // Compute the next step placement in the planted foot frame based on the requested velocity target
408  Vec3 step = Vec3::Zero();
409  step.x() = std::max(std::min(velocity_target.x() * p.step_period, p.step_limits.x()), -p.step_limits.x());
410  step.y() = std::max(std::min(velocity_target.y() * p.step_period, p.step_limits.y()), -p.step_limits.y());
411  step.z() = std::max(std::min(velocity_target.z() * p.step_period, p.step_limits.z()), -p.step_limits.z());
412 
413  // Generate torso trajectory
414  generate_torso_trajectory(step, velocity_target);
415 
416  // Generate swing foot trajectory
417  generate_swing_foot_trajectory(step, velocity_target);
418  }

420  /**
421  * @brief Generate starting trajectories for the torso and swing foot.
422  */
423  void generate_starting_trajectories() {
424  // Create a waypoint variable to use for all waypoints
425  Waypoint<Scalar> wp;
426 
427  // ******************************** Swing Foot Trajectory ********************************
428  swing_foot_trajectory.clear();
429 
430  // Start waypoint: Start from the current swing foot position
431  wp.time_point = 0.0;
432  wp.position = Hps_start.translation();
433  wp.velocity = Vec3::Zero();
434  wp.orientation = mat_to_rpy_intrinsic(Hps_start.rotation());
435  wp.angular_velocity = Vec3::Zero();
436  swing_foot_trajectory.add_waypoint(wp);
437 
438  // Start waypoint: End at the current swing foot position
439  wp.time_point = p.step_period;
440  wp.position = Hps_start.translation();
441  wp.velocity = Vec3::Zero();
442  wp.orientation = mat_to_rpy_intrinsic(Hps_start.rotation());
443  wp.angular_velocity = Vec3::Zero();
444  swing_foot_trajectory.add_waypoint(wp);
445 
446  // ******************************** Torso Trajectory ********************************
447  torso_trajectory.clear();
448 
449  // Start waypoint: Start from the current torso position
450  wp.time_point = 0.0;
451  wp.position = Hpt_start.translation();
452  wp.velocity = Vec3::Zero();
453  wp.orientation = mat_to_rpy_intrinsic(Hpt_start.rotation());
454  wp.angular_velocity = Vec3::Zero();
455  torso_trajectory.add_waypoint(wp);
456 
457  // Middle waypoint: Shift torso over planted foot
458  wp.time_point = p.step_period;
459  Scalar torso_offset_y = phase == LEFT ? -p.torso_sway_offset.y() : p.torso_sway_offset.y();
460  wp.position = Vec3(p.torso_sway_offset.x(), torso_offset_y, p.torso_height + p.torso_sway_offset.z())
461  + p.torso_position_offset;
462  wp.velocity = Vec3(0, 0, 0);
463  wp.orientation = Vec3(0.0, p.torso_pitch, 0);
464  wp.angular_velocity = Vec3(0.0, 0.0, 0);
465  torso_trajectory.add_waypoint(wp);
466  }

468  /**
469  * @brief Switch planted foot.
470  * @details This function switches the planted foot and updates the start torso and start swing foot poses.
471  */
472  void switch_planted_foot() {
473  // Transform planted foot to swing foot frame at end of step
474  Hps_start = get_swing_foot_pose(p.step_period).inverse();
475 
476  // Transform torso to end torso frame at end of step, in the new planted foot frame
477  Hpt_start = Hps_start * get_torso_pose(p.step_period);
478 
479  // Switch planted foot indicator
480  phase = (phase == LEFT) ? RIGHT : LEFT;
481 
482  // Reset time
483  t = 0;
484  }

486  /**
487  * @brief Update the time.
488  * @param dt Time step.
489  */
490  void update_time(const Scalar& dt) {
491  // Check for negative time step
492  if (dt <= 0.0f) {
493  NUClear::log<NUClear::WARN>("dt <= 0.0f");
494  return;
495  }
496 
497  // Check for too long dt
498  if (dt > p.step_period) {
499  NUClear::log<NUClear::WARN>("dt > params.p.step_period");
500  return;
501  }
502 
503  // Update the time
504  t += dt;
505 
506  // Clamp time to step period
507  if (t >= p.step_period) {
508  t = p.step_period;
509  }
510  }
