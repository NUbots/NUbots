46  /**
47  * @brief Calibrates a single camera's extrinsic offsets (roll, pitch, yaw) by optimising offset deltas
48  * (del_roll, del_pitch, del_yaw) so that the projected YOLO field-landmark detections (X, L, T
49  * intersections) align with the known ground-truth landmark positions.
50  *
51  * The robot stands itself up and automatically sweeps its head through a yaw/pitch grid (via the PlanLook
52  * LookAround provider, configured by the role-specific PlanLook.yaml) so the detections span the image
53  * without any manual input.
54  *
55  * Assumptions:
56  * - The robot is placed on the centre of the field, looking straight toward the goal.
57  * - The field type (and hence the field dimensions / ground-truth landmark positions) is known via the
58  * FieldDescription.
59  */
60  class ExtrinsicsCalibration : public ::extension::behaviour::BehaviourReactor {
61  private:
62  /// @brief Number of optimisation variables (del_roll, del_pitch, del_yaw)
63  static constexpr unsigned int n_params = 3;
64 
65  /// @brief Stores configuration values
66  struct Config {
67  /// @brief Path to the URDF model used for camera forward kinematics
68  std::string urdf_path{};
69 
70  /// @brief Which camera to calibrate ("Left" or "Right")
71  std::string camera{};
72 
73  /// @brief True if the calibrated camera is the left camera
74  bool is_left_camera = true;
75 
76  /// @brief Minimum number of associated samples required before running the optimisation
77  size_t min_samples = 0;
78 
79  /// @brief Maximum distance [m] in field space for a detection to be associated with a landmark
80  double max_association_distance = 0.0;
81 
82  /// @brief Minimum head yaw/pitch change [rad] since the last captured frame before a new frame is
83  /// gathered. Keeps samples spread across distinct head poses (the head sweeps automatically) instead
84  /// of over-sampling whatever pose the head is lingering in.
85  double min_head_pose_change = 0.0;
86 
87  /// @brief Search bounds (half-width) on each offset delta [rad] (del_roll, del_pitch, del_yaw)
88  Eigen::Vector3d offset_bounds = Eigen::Vector3d::Zero();
89 
90  /// @brief NLopt relative tolerance on the optimisation parameters
91  double xtol_rel = 0.0;
92 
93  /// @brief NLopt relative tolerance on the optimisation function value
94  double ftol_rel = 0.0;
95 
96  /// @brief NLopt maximum number of evaluations
97  size_t maxeval = 0;
98 
99  /// @brief Maximum number of ICP iterations (re-associate then re-optimise) before giving up
100  size_t max_icp_iterations = 0;
101  } cfg;
102 
103  /// @brief A single raw detection, kept so it can be re-associated as the offsets are refined (ICP).
104  struct Observation {
105  /// @brief Detection ray in camera {c} space (offset-independent, recovered from the pixel)
106  Eigen::Vector3d uICc = Eigen::Vector3d::Zero();
107  /// @brief Detection type (X / L / T intersection), used to restrict same-type associations
108  message::vision::FieldIntersection::IntersectionType type =
109  message::vision::FieldIntersection::IntersectionType::UNKNOWN;
110  };
111 
112  /// @brief One captured frame: the offset-free transforms plus every detection in that frame.
113  /// Detections are grouped per frame so the Hungarian association stays one-to-one within a frame.
114  struct Frame {
115  /// @brief World {w} from head-pitch {p} transform for this frame (offset-free)
116  Eigen::Isometry3d Hwp = Eigen::Isometry3d::Identity();
117  /// @brief Field {f} from world {w} transform for this frame (from placement assumption)
118  Eigen::Isometry3d Hfw = Eigen::Isometry3d::Identity();
119  /// @brief Raw detections captured in this frame
120  std::vector<Observation> observations{};
121  };
122 
123  /// @brief A single associated calibration sample (one detection paired with a ground-truth landmark).
124  struct Sample {
125  /// @brief Detection ray in camera {c} space (offset-independent, recovered from the pixel)
126  Eigen::Vector3d uICc = Eigen::Vector3d::Zero();
127  /// @brief World {w} from head-pitch {p} transform at the time of detection (offset-free)
128  Eigen::Isometry3d Hwp = Eigen::Isometry3d::Identity();
129  /// @brief Field {f} from world {w} transform at the time of detection (from placement assumption)
130  Eigen::Isometry3d Hfw = Eigen::Isometry3d::Identity();
131  /// @brief Associated ground-truth landmark position in field {f} space
132  Eigen::Vector3d rLFf = Eigen::Vector3d::Zero();
133  };
134  /// @brief Base (offset-free) head-pitch {p} from camera {c} transform, from URDF forward kinematics
135  Eigen::Isometry3d Hpc_base = Eigen::Isometry3d::Identity();
136 
137  /// @brief Current extrinsic offsets (roll, pitch, yaw) read from the robot's camera config [rad]
138  Eigen::Vector3d current_offsets = Eigen::Vector3d::Zero();
139 
140  /// @brief Head yaw/pitch [rad] of the most recently captured frame, used to gather samples only once the
141  /// head has moved far enough (see Config::min_head_pose_change)
142  double last_head_yaw = 0.0;
143  double last_head_pitch = 0.0;
144 
145  /// @brief True once the calibration has finished and written the offsets. The robot keeps standing and the
146  /// head holds still (rather than exiting and collapsing); the operator stops the binary.
147  bool calibrated = false;
148 
149  /// @brief Ground-truth field landmarks (position in field space + type)
150  std::vector<utility::localisation::Landmark> landmarks{};
151 
152  /// @brief Accumulated raw frames (re-associated each ICP iteration)
153  std::vector<Frame> frames{};
154 
155  /// @brief Total number of raw detections collected so far across all frames
156  size_t collected_detections = 0;
157 
158  /// @brief Associated samples for the current ICP iteration (rebuilt by associate(), used by the optimiser)
159  std::vector<Sample> samples{};
160 
161  /// @brief Path to the robot's camera config file that will be (re)written
162  std::string camera_config_path{};
163 
164  /**
165  * @brief Build the extrinsic offset rotation matrix from (roll, pitch, yaw), matching the convention
166  * used by the Camera module.
167  * @param offsets The offsets (roll, pitch, yaw) [rad]
168  * @return The offset rotation matrix
169  */
170  Eigen::Matrix3d offset_rotation(const Eigen::Vector3d& offsets) const;
171 
172  /**
173  * @brief Re-project a detection ray onto the field ground plane in field {f} space for a given set of
174  * extrinsic offsets.
175  * @param offsets The candidate extrinsic offsets (roll, pitch, yaw) [rad]
176  * @param sample The sample to re-project
177  * @return The projected landmark position in field {f} space
178  */
179  Eigen::Vector3d project_to_field(const Eigen::Vector3d& offsets, const Sample& sample) const;
180 
181  /**
182  * @brief Recover the offset-independent camera-frame rays for a frame's detections and store the raw
183  * frame for later (re-)association. No landmark association is committed here.
184  * @param field_intersections The field intersection detections
185  * @param Hwp The offset-free world from head-pitch transform for this frame
186  * @param Hfw The field from world transform for this frame
187  */
188  void collect(const message::vision::FieldIntersections& field_intersections,
189  const Eigen::Isometry3d& Hwp,
190  const Eigen::Isometry3d& Hfw);
191 
192  /**
193  * @brief (Re-)associate every collected detection with the known landmarks using the Hungarian algorithm,
194  * evaluated at the given offsets. Rebuilds the `samples` buffer used by the optimiser.
195  * @param offsets The extrinsic offsets (roll, pitch, yaw) [rad] at which to project and associate
196  * @return Per-observation signature (frame-major) of assigned landmark indices, -1 where unassociated.
197  * Used to detect ICP convergence (associations no longer changing).
198  */
199  std::vector<int> associate(const Eigen::Vector3d& offsets);
200 
201  /**
202  * @brief Run the ICP loop: alternate (associate at the current best offsets) and (optimise the offsets
203  * with those associations held fixed), re-centring on the refined offsets each iteration, until the
204  * associations stop changing or the iteration cap is reached. Writes the result and exits on success.
205  */
206  void run_calibration();
207 
208  /**
209  * @brief Total squared re-projection cost over all accumulated samples for a given set of offsets.
210  * @param offsets The candidate extrinsic offsets (roll, pitch, yaw) [rad]
211  * @return The summed squared distance between each projected detection and its associated landmark
212  */
213  double total_cost(const Eigen::Vector3d& offsets) const;
214 
215  /**
216  * @brief NLopt objective. The optimisation variables are the offset deltas (del_roll, del_pitch,
217  * del_yaw); the cost is evaluated at (current_offsets + delta).
218  * @param x The offset deltas (del_roll, del_pitch, del_yaw) [rad]
219  * @param grad Unused (BOBYQA is derivative-free)
220  * @param data Pointer to the owning ExtrinsicsCalibration instance
221  * @return The total re-projection cost
222  */
223  static double objective(const std::vector<double>& x, std::vector<double>& grad, void* data);
224 
225  /**
226  * @brief Run the BOBYQA optimisation over the accumulated samples to find the best offsets.
227  * @return Tuple <optimal offsets (roll, pitch, yaw), final cost, optimisation result code>
228  */
229  std::tuple<Eigen::Vector3d, double, nlopt::result> optimise();
230 
231  /**
232  * @brief Write the optimised offsets back to the robot's camera config file, formatted as
233  * "<degrees> * pi / 180" expression strings.
234  * @param offsets The optimised offsets (roll, pitch, yaw) [rad]
235  */
236  void write_offsets(const Eigen::Vector3d& offsets);
237 
238  public:
239  /// @brief Called by the powerplant to build and setup the ExtrinsicsCalibration reactor.
240  explicit ExtrinsicsCalibration(std::unique_ptr<NUClear::Environment> environment);
241  };
