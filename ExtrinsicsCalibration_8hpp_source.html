44  /**
45  * @brief Calibrates a single camera's extrinsic offsets (roll, pitch, yaw) by optimising offset deltas
46  * (del_roll, del_pitch, del_yaw) so that the projected YOLO field-landmark detections (X, L, T
47  * intersections) align with the known ground-truth landmark positions.
48  *
49  * Assumptions:
50  * - The robot is static, placed on the centre of the field, looking straight toward the goal.
51  * - The field type (and hence the field dimensions / ground-truth landmark positions) is known via the
52  * FieldDescription.
53  */
54  class ExtrinsicsCalibration : public NUClear::Reactor {
55  private:
56  /// @brief Number of optimisation variables (del_roll, del_pitch, del_yaw)
57  static constexpr unsigned int n_params = 3;
58 
59  /// @brief Stores configuration values
60  struct Config {
61  /// @brief Path to the URDF model used for camera forward kinematics
62  std::string urdf_path{};
63 
64  /// @brief Which camera to calibrate ("Left" or "Right")
65  std::string camera{};
66 
67  /// @brief True if the calibrated camera is the left camera
68  bool is_left_camera = true;
69 
70  /// @brief Minimum number of associated samples required before running the optimisation
71  size_t min_samples = 0;
72 
73  /// @brief Maximum distance [m] in field space for a detection to be associated with a landmark
74  double max_association_distance = 0.0;
75 
76  /// @brief Search bounds (half-width) on each offset delta [rad] (del_roll, del_pitch, del_yaw)
77  Eigen::Vector3d offset_bounds = Eigen::Vector3d::Zero();
78 
79  /// @brief NLopt relative tolerance on the optimisation parameters
80  double xtol_rel = 0.0;
81 
82  /// @brief NLopt relative tolerance on the optimisation function value
83  double ftol_rel = 0.0;
84 
85  /// @brief NLopt maximum number of evaluations
86  size_t maxeval = 0;
87  } cfg;
88 
89  /// @brief A single associated calibration sample.
90  struct Sample {
91  /// @brief Detection ray in camera {c} space (offset-independent, recovered from the pixel)
92  Eigen::Vector3d uICc = Eigen::Vector3d::Zero();
93  /// @brief World {w} from head-pitch {p} transform at the time of detection (offset-free)
94  Eigen::Isometry3d Hwp = Eigen::Isometry3d::Identity();
95  /// @brief Field {f} from world {w} transform at the time of detection (from placement assumption)
96  Eigen::Isometry3d Hfw = Eigen::Isometry3d::Identity();
97  /// @brief Associated ground-truth landmark position in field {f} space
98  Eigen::Vector3d rLFf = Eigen::Vector3d::Zero();
99  };
100  /// @brief Base (offset-free) head-pitch {p} from camera {c} transform, from URDF forward kinematics
101  Eigen::Isometry3d Hpc_base = Eigen::Isometry3d::Identity();
102 
103  /// @brief Current extrinsic offsets (roll, pitch, yaw) read from the robot's camera config [rad]
104  Eigen::Vector3d current_offsets = Eigen::Vector3d::Zero();
105 
106  /// @brief Ground-truth field landmarks (position in field space + type)
107  std::vector<utility::localisation::Landmark> landmarks{};
108 
109  /// @brief Accumulated associated samples used for the optimisation
110  std::vector<Sample> samples{};
111 
112  /// @brief Path to the robot's camera config file that will be (re)written
113  std::string camera_config_path{};
114 
115  /**
116  * @brief Build the extrinsic offset rotation matrix from (roll, pitch, yaw), matching the convention
117  * used by the Camera module.
118  * @param offsets The offsets (roll, pitch, yaw) [rad]
119  * @return The offset rotation matrix
120  */
121  Eigen::Matrix3d offset_rotation(const Eigen::Vector3d& offsets) const;
122 
123  /**
124  * @brief Re-project a detection ray onto the field ground plane in field {f} space for a given set of
125  * extrinsic offsets.
126  * @param offsets The candidate extrinsic offsets (roll, pitch, yaw) [rad]
127  * @param sample The sample to re-project
128  * @return The projected landmark position in field {f} space
129  */
130  Eigen::Vector3d project_to_field(const Eigen::Vector3d& offsets, const Sample& sample) const;
131 
132  /**
133  * @brief Associate the detections in a frame with the known landmarks using the Hungarian algorithm,
134  * evaluated at the current extrinsic offsets, and append the matched samples to the buffer.
135  * @param field_intersections The field intersection detections
136  * @param Hwp The offset-free world from head-pitch transform for this frame
137  * @param Hfw The field from world transform for this frame
138  */
139  void associate_and_store(const message::vision::FieldIntersections& field_intersections,
140  const Eigen::Isometry3d& Hwp,
141  const Eigen::Isometry3d& Hfw);
142 
143  /**
144  * @brief Total squared re-projection cost over all accumulated samples for a given set of offsets.
145  * @param offsets The candidate extrinsic offsets (roll, pitch, yaw) [rad]
146  * @return The summed squared distance between each projected detection and its associated landmark
147  */
148  double total_cost(const Eigen::Vector3d& offsets) const;
149 
150  /**
151  * @brief NLopt objective. The optimisation variables are the offset deltas (del_roll, del_pitch,
152  * del_yaw); the cost is evaluated at (current_offsets + delta).
153  * @param x The offset deltas (del_roll, del_pitch, del_yaw) [rad]
154  * @param grad Unused (BOBYQA is derivative-free)
155  * @param data Pointer to the owning ExtrinsicsCalibration instance
156  * @return The total re-projection cost
157  */
158  static double objective(const std::vector<double>& x, std::vector<double>& grad, void* data);
159 
160  /**
161  * @brief Run the BOBYQA optimisation over the accumulated samples to find the best offsets.
162  * @return Tuple <optimal offsets (roll, pitch, yaw), final cost, optimisation result code>
163  */
164  std::tuple<Eigen::Vector3d, double, nlopt::result> optimise();
165 
166  /**
167  * @brief Write the optimised offsets back to the robot's camera config file, formatted as
168  * "<degrees> * pi / 180" expression strings.
169  * @param offsets The optimised offsets (roll, pitch, yaw) [rad]
170  */
171  void write_offsets(const Eigen::Vector3d& offsets);
172 
173  public:
174  /// @brief Called by the powerplant to build and setup the ExtrinsicsCalibration reactor.
175  explicit ExtrinsicsCalibration(std::unique_ptr<NUClear::Environment> environment);
176  };
