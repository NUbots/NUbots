59  struct StartingSide {
60  enum Value { UNKNOWN = 0, LEFT = 1, RIGHT = 2, EITHER = 3, CUSTOM = 4 };
61  Value value = Value::UNKNOWN;
62 
63  // Constructors
64  StartingSide() = default;
65  StartingSide(int const& v) : value(static_cast<Value>(v)) {}
66  StartingSide(Value const& v) : value(v) {}
67  StartingSide(std::string const& str) {
68  // clang-format off
69  if (str == "LEFT") { value = Value::LEFT; }
70  else if (str == "RIGHT") { value = Value::RIGHT; }
71  else if (str == "EITHER") { value = Value::EITHER; }
72  else if (str == "CUSTOM") { value = Value::CUSTOM; }
73  else {
74  value = Value::UNKNOWN;
75  throw std::runtime_error("String " + str + " did not match any enum for StartingSide");
76  }
77  // clang-format on
78  }
79 
80  // Conversions
81  [[nodiscard]] operator Value() const {
82  return value;
83  }
84  [[nodiscard]] operator std::string() const {
85  switch (value) {
86  case Value::LEFT: return "LEFT";
87  case Value::RIGHT: return "RIGHT";
88  case Value::EITHER: return "EITHER";
89  case Value::CUSTOM: return "CUSTOM";
90  default: throw std::runtime_error("enum Method's value is corrupt, unknown value stored");
91  }
92  }
93  };

102  template <typename Scalar, int n>
103  inline void eigen_to_nlopt(const Eigen::Matrix<Scalar, n, 1>& eigen_vec, std::vector<Scalar>& nlopt_vec) {
104  // Use Eigen::Map to create a view of the std::vector data
105  Eigen::Map<Eigen::Matrix<Scalar, n, 1>> nlopt_eigen_vec(nlopt_vec.data(), n);

107  // Copy the data from the Eigen vector to the std::vector view

127  /**
128  * @brief Type definition for an objective function that takes an Eigen vector as input and returns a scalar value.
129  * @tparam Scalar The scalar type of the Eigen vector and the scalar return value.
130  * @tparam nv The size of the Eigen vector.
131  */
132  template <typename Scalar, int nv>
133  using ObjectiveFunction =
134  std::function<Scalar(const Eigen::Matrix<Scalar, nv, 1>&, Eigen::Matrix<Scalar, nv, 1>&, void*)>;
135 
136  /**
137  * @brief Wrapper function that converts input and output between NLopt and Eigen formats for an objective function.
138  * @param n The size of the input vector.
139  * @param x The input vector in NLopt format.
140  * @param grad The gradient vector in NLopt format.
141  * @param data Pointer to additional data that is passed to the objective function.
142  * @tparam Scalar The scalar type of the Eigen vector and the scalar return value.
143  * @tparam nv The size of the Eigen vector.
144  * @return The scalar value of the objective function.
145  */
146  template <typename Scalar, int nv>
147  inline Scalar eigen_objective_wrapper(unsigned n, const Scalar* x, Scalar* grad, void* data) {
148  ObjectiveFunction<Scalar, nv>& obj_fun = *static_cast<ObjectiveFunction<Scalar, nv>*>(data);
149 
150  // Convert input from NLopt format to Eigen format
151  Eigen::Map<const Eigen::Matrix<Scalar, nv, 1>> eigen_x(x, n);
152  Eigen::Matrix<Scalar, nv, 1> eigen_grad = Eigen::Matrix<Scalar, nv, 1>::Zero();
153 
154  if (grad) {
155  eigen_grad.resize(n);
156  Eigen::Map<Eigen::Matrix<Scalar, nv, 1>>(grad, n) = eigen_grad;
157  }
158 
159  // Call the actual objective function implemented with Eigen
160  Scalar result = obj_fun(eigen_x, eigen_grad, data);
161 
162  if (grad) {
163  // Copy the gradient back to NLopt format
164  Eigen::Map<Eigen::Matrix<Scalar, nv, 1>>(grad, n) = eigen_grad;
165  }
166 
167  return result;
168  }
