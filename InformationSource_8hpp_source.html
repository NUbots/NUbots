36  /**
37  * Provides information about how and why this provider was executed.
38  */
39  struct RunInfo {
40  enum RunReason {
41  /// Something other than the Director caused this reaction to execute
42  OTHER_TRIGGER,
43  /// A new task has been given to this provider
44  NEW_TASK,
45  /// The provider has been started (will happen on on<Started<X>>)
46  STARTED,
47  /// The provider has been stopped (will happen on on<Stopped<X>>)
48  STOPPED,
49  /// A subtask has finished and emitted a Done message
50  SUBTASK_DONE,
51  /// Another task requires a causing from this provider and pushed it to run
52  PUSHED
53  };
54 
55  /// The reason we executed
56  RunReason run_reason;
57  };
58 
59  /**
60  * Provides information about the current state of a provider group
61  */
62  struct GroupInfo {
63  enum RunState {
64  /// The group has not emitted the task
65  NO_TASK,
66  /// The group is running the task
67  RUNNING,
68  /// The group has the task queued
69  QUEUED
70  };
71 
72  /// The current run state of the group
73  RunState run_state;
74 
75  /// Whether the task is done or not, regardless of if this provider group is running it
76  bool done;
77  };

83  /**
84  * This class is an abstract class that allows the static Behaviour DSL to access the relevant state from the
85  * algorithm that is backing it.
86  *
87  * Extend this class to provide whatever information is needed for outputs in the DSL.
88  */
89  class InformationSource {
90  protected:
91  /// This static variable gets set by whatever class is acting as the information source
92  static InformationSource* source;
93 
94  /**
95  * Internal get task data function. Is overridden by the class that is acting as the information source.
96  *
97  * @param reaction_id the reaction id of the reaction that is asking for data
98  *
99  * @return a void shared pointer to the data for this reaction
100  */
101  virtual std::shared_ptr<void> _get_task_data(const uint64_t& reaction_id) = 0;
102 
103  /**
104  * Internal get run information. Is overridden by the class that is acting as the information source.
105  *
106  * @param reaction_id the reaction id of the reaction that is asking for data
107  *
108  * @return a RunInfo struct containing information about the current run
109  */
110  virtual RunInfo _get_run_info(const uint64_t& reaction_id) = 0;
111 
112  /**
113  * Internal get group information. Is overridden by the class that is acting as the information source.
114  *
115  * @param reaction_id the reaction id of the reaction that is asking for data
116  * @param type the type of the provider group group to get information about
117  * @param root_type the secondary type to use if this is a root task
118  *
119  * @return a GroupInfo struct containing information about the current group
120  */
121  virtual GroupInfo _get_group_info(const uint64_t& reaction_id,
122  const std::type_index& type,
123  const std::type_index& root_type) = 0;
124 
125  public:
126  virtual ~InformationSource() = default;
127 
128  /**
129  * Gets the data for the passed providers reaction id from the information source
130  *
131  * @param reaction_id the reaction id of the reaction that is asking for data
132  *
133  * @return a void shared pointer to the data for this reaction
134  */
135  static inline std::shared_ptr<void> get_task_data(const uint64_t& reaction_id) {
136  return source->_get_task_data(reaction_id);
137  }
138 
139  /**
140  * Gets information about why the current task is running from the information source
141  *
142  * @param reaction_id the reaction id of the reaction that is asking for data
143  *
144  * @return a RunInfo struct containing information about the current run
145  */
146  static inline RunInfo get_run_info(const uint64_t& reaction_id) {
147  return source->_get_run_info(reaction_id);
148  }
149 
150  /**
151  * Get the group info object
152  *
153  * @param reaction_id the reaction id of the reaction that is asking for data
154  * @param type the type of the group to get the info for
155  * @param root_type the secondary type to use if this is a root task
156  *
157  * @return the group info object containing information about the requested group
158  */
159  static inline GroupInfo get_group_info(const uint64_t& reaction_id,
160  const std::type_index& type,
161  const std::type_index& root_type) {
162  return source->_get_group_info(reaction_id, type, root_type);
163  }
164  };
165 
166 } // namespace extension::behaviour::information
