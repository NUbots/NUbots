63  /// @brief The current state of the optimisation
64  struct State {
65  enum Value {
66  UNKNOWN = 0,
67  WAITING_FOR_REQUEST = 1,
68  SETTING_UP_TRIAL = 2,
69  RESETTING_TRIAL = 3,
70  EVALUATING = 4,
71  TERMINATING_EARLY = 5,
72  TERMINATING_GRACEFULLY = 6,
73  FINISHED = 7
74  };
75  Value value = Value::UNKNOWN;
76 
77  // Constructors
78  State() = default;
79  State(Value const& v) : value(v) {}
80 
81  // Overload output operator
82  friend std::ostream& operator<<(std::ostream& os, const State& state) {
83  switch (state.value) {
84  case Value::UNKNOWN: return os << "UNKNOWN";
85  case Value::WAITING_FOR_REQUEST: return os << "WAITING_FOR_REQUEST";
86  case Value::SETTING_UP_TRIAL: return os << "SETTING_UP_TRIAL";
87  case Value::RESETTING_TRIAL: return os << "RESETTING_TRIAL";
88  case Value::EVALUATING: return os << "EVALUATING";
89  case Value::TERMINATING_EARLY: return os << "TERMINATING_EARLY";
90  case Value::TERMINATING_GRACEFULLY: return os << "TERMINATING_GRACEFULLY";
91  case Value::FINISHED: return os << "FINISHED";
92  default: throw std::runtime_error("enum State's value is corrupt, unknown value stored");
93  }
94  }
95  };
96 
97  /// @brief The current event that is happening in the optimisation
98  struct Event {
99  enum Value {
100  // From webots
101  RESET_DONE = 0,
102  // From optimiser
103  CHECK_READY = 1,
104  EVALUATE_REQUEST = 2,
105  TERMINATE_EVALUATION = 3,
106  // Internal
107  TRIAL_SETUP_DONE = 4,
108  TERMINATE_EARLY = 5,
109  TRIAL_COMPLETED = 6,
110  FITNESS_SCORES_SENT = 7
111  };
112 
113  /// @brief The event that is happening
114  Value value = Value::RESET_DONE;
115 
116  // Constructors
117  Event() = default;
118  Event(Value const& v) : value(v) {}
119 
120  // Overload output operator
121  friend std::ostream& operator<<(std::ostream& os, const Event& event) {
122  switch (event.value) {
123  case Value::RESET_DONE: return os << "RESET_DONE";
124  case Value::CHECK_READY: return os << "CHECK_READY";
125  case Value::EVALUATE_REQUEST: return os << "EVALUATE_REQUEST";
126  case Value::TERMINATE_EVALUATION: return os << "TERMINATE_EVALUATION";
127  case Value::TRIAL_SETUP_DONE: return os << "TRIAL_SETUP_DONE";
128  case Value::TERMINATE_EARLY: return os << "TERMINATE_EARLY";
129  case Value::TRIAL_COMPLETED: return os << "TRIAL_COMPLETED";
130  case Value::FITNESS_SCORES_SENT: return os << "FITNESS_SCORES_SENT";
131  default: throw std::runtime_error("enum Event's value is corrupt, unknown value stored");
132  }
133  }
134  };
135

138  std::unique_ptr<EvaluatorTask> task;
139 
140  /// @brief The number of the current generation
141  int generation = 0;

143  /// @brief The number of the current individual in the current generation
144  int individual = 0;

151 
152  /// @brief Send the evaluated fitness scores
153  /// @param scores The evaluated fitness scores for the current individual (sway, distance travelled)
154  /// @param constraints A list of constraints for domination calculation. These can be used to encode one
155  /// or more failure conditions. Here, we set the first constraint if the robot has fallen over, and the
156  /// second one if it's swayed too too much. The more negative the constraint value is, the more it has
157  /// violated the constriant, which means it will be dominated by other individuals that have violated
158  /// less.
159  void send_fitness_scores(std::vector<double> scores, std::vector<double> constraints);
160 
161  /// @brief Implementation of the fitness function (i.e. how well the individual did)
162  std::vector<double> calculate_scores();
163 
164  /// @brief Implementation of the constraints function (i.e. how much to penalise for violations)
165  std::vector<double> calculate_constraints();
166 
167  /// @brief Returns the appropriate values for the case where no constraints are violated
168  std::vector<double> constraints_not_violated();
169 
170  /// @brief The current state of the evaluation

172 
173  /// @brief Get the next state to transition to given the current state and an event

176  /// @brief Handle the WAITING_FOR_REQUEST state

185  /// @brief Handle the STANDING state
186  void standing();
