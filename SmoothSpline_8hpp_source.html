37  * SmoothSpline
38  *
39  * Implementation of 5th order polynomial splines trajectory known to minimize jerk
40  */
41  template <typename Scalar>
42  class SmoothSpline : public Spline<Scalar> {
43  public:

81  /**
82  * Recompute splines interpolation model
83  */
84  void computeSplines() {
85  Spline<Scalar>::splines.clear();
86  if (points.size() < 2) {
87  return;
88  }
89 
90  std::sort(points.begin(), points.end(), [](const Point& p1, const Point& p2) -> bool {
91  return p1.time < p2.time;
92  });
93 
94  for (size_t i = 1; i < points.size(); i++) {
95  const Scalar time = points[i].time - points[i - 1].time;
96  if (time > static_cast<Scalar>(0.00001)) {
97  Spline<Scalar>::splines.push_back({polynomFit(time,
98  points[i - 1].position,
99  points[i - 1].velocity,
100  points[i - 1].acceleration,
101  points[i].position,
102  points[i].velocity,
103  points[i].acceleration),
104  points[i - 1].time,
105  points[i].time});
106  }
107  }
108  }
109 
110  protected:
111  /**
112  * Inherit Load Points
113  */
114  void importCallBack() override {
115  const size_t size = Spline<Scalar>::splines.size();
116  if (size == 0) {
117  return;
118  }
119 
120  const Scalar tBegin = Spline<Scalar>::splines.front().min;
121  points.push_back(
122  {tBegin, Spline<Scalar>::pos(tBegin), Spline<Scalar>::vel(tBegin), Spline<Scalar>::acc(tBegin)});
123 
124  for (size_t i = 1; i < size; i++) {
125  const Scalar t1 = Spline<Scalar>::splines[i - 1].max;
126  const Scalar t2 = Spline<Scalar>::splines[i].min;
127  const Scalar pos1 = Spline<Scalar>::pos(t1);
128  const Scalar vel1 = Spline<Scalar>::vel(t1);
129  const Scalar acc1 = Spline<Scalar>::acc(t1);
130  const Scalar pos2 = Spline<Scalar>::pos(t2);
131  const Scalar vel2 = Spline<Scalar>::vel(t2);
132  const Scalar acc2 = Spline<Scalar>::acc(t2);
133 
134  if (std::abs(t2 - t1) < static_cast<Scalar>(0.0001)
135  && std::abs(pos2 - pos1) < static_cast<Scalar>(0.0001)
136  && std::abs(vel2 - vel1) < static_cast<Scalar>(0.0001)
137  && std::abs(acc2 - acc1) < static_cast<Scalar>(0.0001)) {
138  points.push_back({t1, pos1, vel1, acc1});
139  }
140  else {
141  points.push_back({t1, pos1, vel1, acc1});
142  points.push_back({t2, pos2, vel2, acc2});
143  }
144  }
145 
146  const Scalar tEnd = Spline<Scalar>::splines.back().max;
147  points.push_back({tEnd, Spline<Scalar>::pos(tEnd), Spline<Scalar>::vel(tEnd), Spline<Scalar>::acc(tEnd)});
148  }
149 
150  private:
151  /**
152  * Points container
153  */
154  std::vector<Point> points{};

156  /**
157  * Fit a polynom between 0 and t with given pos, vel and acc initial and final conditions
158  */
159  [[nodiscard]] Polynom<Scalar> polynomFit(const Scalar& t,
160  const Scalar& pos1,
161  const Scalar& vel1,
162  const Scalar& acc1,
163  const Scalar& pos2,
164  const Scalar& vel2,
165  const Scalar& acc2) const {
166  if (t <= static_cast<Scalar>(0.00001)) {
167  throw std::logic_error("SmoothSpline invalid spline interval");
168  }
169  const Scalar t2 = t * t;
170  const Scalar t3 = t2 * t;
171  const Scalar t4 = t3 * t;
172  const Scalar t5 = t4 * t;
173  Polynom<Scalar> p{};
174  p.getCoefs().resize(6);
175  p.getCoefs()[0] = pos1;
176  p.getCoefs()[1] = vel1;
177  p.getCoefs()[2] = acc1 / 2;
178  p.getCoefs()[3] =
179  -(-acc2 * t2 + 3 * acc1 * t2 + 8 * vel2 * t + 12 * vel1 * t - 20 * pos2 + 20 * pos1) / (2 * t3);
180  p.getCoefs()[4] =
181  (-2 * acc2 * t2 + 3 * acc1 * t2 + 14 * vel2 * t + 16 * vel1 * t - 30 * pos2 + 30 * pos1) / (2 * t4);
182  p.getCoefs()[5] =
183  -(-acc2 * t2 + acc1 * t2 + 6 * vel2 * t + 6 * vel1 * t - 12 * pos2 + 12 * pos1) / (2 * t5);
184 
185  return p;
186  }
187  };
