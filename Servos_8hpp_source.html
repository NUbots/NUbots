50  /// @brief Creates a reaction that sends a given servo command as a servo target command for the platform module
51  /// to use
52  // TODO(ysims): add capability to be Done when the servo reaches the target position
53  template <typename Servo, ServoID::Value ID>
54  void add_servo_provider() {
55  on<Provide<Servo>, Priority::HIGH>().then([this](const Servo& servo, const RunReason& run_reason) {
56  if (run_reason == RunReason::NEW_TASK) {
57  if (log_level <= DEBUG) {
58  emit(graph("Servo " + std::to_string(ID) + " (Position, Gain, Torque Enabled): ",
59  servo.command.position,
60  servo.command.state.gain,
61  servo.command.state.torque));
62  }
63  // Emit the request to move the servo
64  emit(std::make_unique<ServoTarget>(servo.command.time,
65  ID,
66  servo.command.position,
67  servo.command.state.gain,
68  servo.command.state.torque));
69  }
70  if (servo.command.done_type == ServoCommand::DoneType::TIME) {
71  if (servo.command.time <= NUClear::clock::now()) {
72  emit<Task>(std::make_unique<Done>());
73  }
74  else {
75  emit<Task>(std::make_unique<Wait>(servo.command.time));
76  }
77  }
78  });
79  }

81  /// @brief Creates a reaction that takes a servo wrapper task (eg LeftLeg) and emits a task for each servo.
82  /// Emits Done when all servo tasks are Done. If individual servos in that wrapper task were not provided, then
83  /// it ignores them.
84  /// @tparam Group is a servo wrapper task (eg LeftLeg)
85  /// @tparam Elements is a template pack of Servos that Group uses
86  template <typename Group, typename... Elements>
87  void add_group_provider() {
88  on<Provide<Group>, Needs<Elements>..., Priority::HIGH>().then(
89  [this](const Group& group, const RunReason& run_reason, const Uses<Elements>... elements) {
90  // Check if any subtask is Done
91  if (run_reason == RunReason::SUBTASK_DONE) {
92  // If every servo task is done then emit Done (ignore servos that weren't included in the Task
93  // message)
94  if (((!group.servos.contains(utility::actuation::ServoMap<Elements>::value) || elements.done)
95  && ...)) {
96  emit<Task>(std::make_unique<Done>());
97  return;
98  }
99  // Emit Continue if all the servos are not Done yet
100  emit<Task>(std::make_unique<Continue>());
101  return;
102  }
103 
104  (
105  // Runs an emit for each servo, if that servo was included in the map
106  [&] {
107  // Check if the Task includes this servo
108  if (group.servos.contains(utility::actuation::ServoMap<Elements>::value)) {
109  // Emit a Task for the servo if the map contains a value for it
110  emit<Task>(std::make_unique<Elements>(
111  group.servos.at(utility::actuation::ServoMap<Elements>::value)));
112  }
113  else { // if a servo was not filled in the map for this group, log it
114  log<TRACE>("Requested a Servo Group Task but did not provide values for Servo",
115  ServoID(utility::actuation::ServoMap<Elements>::value));
116  }
117  }(),
118  ...);
119  });
120  }

122  /// @brief Keeps track of which Group in the Sequence in add_sequence_provider should be emitted next
123  /// @details Should be used carefully and may be bug-prone.
124  /// @tparam Sequence Used to have unique counts for each Sequence Provider.
125  template <typename Sequence>
126  struct Count {
127  long unsigned int count = 0;
128  };
129 
130  /// @brief Creates a reaction that takes a vector of servo wrapper tasks (eg LeftLegs) and emits each servo
131  /// wrapper task after the previous one is Done. Emits Done when the last group is Done.
132  /// @tparam Sequence is a vector of servo wrapper tasks (eg LeftLegSequence)
133  /// @tparam Group is a servo wrapper task (eg LeftLeg)
134  template <typename Sequence, typename Group>
135  void add_sequence_provider() {
136  // Message to keep track of which position in the vector is to be emitted
137  // Make an initial count message
138  emit<Scope::INLINE>(std::make_unique<Count<Sequence>>(0));
139 
140  on<Provide<Sequence>, Needs<Group>, With<Count<Sequence>>, Priority::HIGH>().then(
141  [this](const Sequence& sequence, const RunReason& run_reason, const Count<Sequence>& count) {
142  // If the user gave us nothing then we are done
143  if (sequence.frames.empty()) {
144  emit<Task>(std::make_unique<Done>());
145  }
146  // If this is a new task, run the first pack of servos and increment the counter
147  else if (run_reason == RunReason::NEW_TASK) {
148  emit<Task>(std::make_unique<Group>(sequence.frames[0]));
149  emit<Scope::INLINE>(std::make_unique<Count<Sequence>>(1));
150  }
151  // If the subtask is done, we are done if it is the last servo frames, otherwise use the count to
152  // determine the current frame to emit
153  else if (run_reason == RunReason::SUBTASK_DONE) {
154  if (count.count < sequence.frames.size()) {
155  emit<Task>(std::make_unique<Group>(sequence.frames[count.count]));
156  emit<Scope::INLINE>(std::make_unique<Count<Sequence>>(count.count + 1));
157  }
158  else {
159  emit<Task>(std::make_unique<Done>());
160  }
161  }
162  // Any other run reason shouldn't happen
163  else {
164  emit<Task>(std::make_unique<Continue>());
165  }
166  });
167  }

169  public:
170  /// @brief Called by the powerplant to build and setup the Servos reactor.
171  explicit Servos(std::unique_ptr<NUClear::Environment> environment);
172  };
