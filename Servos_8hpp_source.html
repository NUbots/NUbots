47  class Servos : public ::extension::behaviour::BehaviourReactor {
48  private:
49  /// @brief Creates a reaction that sends a given servo command as a servo target command for the platform module
50  /// to use
51  // TODO(ysims): add capability to be Done when the servo reaches the target position
52  template <typename Servo, ServoID::Value ID>
53  void add_servo_provider() {
54  on<Provide<Servo>, Every<90, Per<std::chrono::seconds>>>().then(
55  [this](const Servo& servo, const RunInfo& info) {
56  if (info.run_reason == RunInfo::RunReason::NEW_TASK) {
57  if (log_level <= NUClear::DEBUG) {
58  emit(graph("Servo " + std::to_string(ID) + " (Position, Gain, Torque Enabled): ",
59  servo.command.position,
60  servo.command.state.gain,
61  servo.command.state.torque));
62  }
63  emit(std::make_unique<ServoTarget>(servo.command.time,
64  ID,
65  servo.command.position,
66  servo.command.state.gain,
67  servo.command.state.torque));
68  }
69  // If the time to reach the position is over, then stop requesting the position
70  else if (NUClear::clock::now() >= servo.command.time) {
71  emit<Task>(std::make_unique<Done>());
72  }
73  });
74  }
75 
76  /// @brief Creates a reaction that takes a servo wrapper task (eg LeftLeg) and emits a task for each servo.
77  /// Emits Done when all servo tasks are Done. If individual servos in that wrapper task were not provided, then
78  /// it ignores them.
79  /// @tparam Group is a servo wrapper task (eg LeftLeg)
80  /// @tparam Elements is a template pack of Servos that Group uses
81  template <typename Group, typename... Elements>
82  void add_group_provider() {
83  on<Provide<Group>, Needs<Elements>...>().then(
84  [this](const Group& group, const RunInfo& info, const Uses<Elements>... elements) {
85  // Check if any subtask is Done
86  if (info.run_reason == RunInfo::RunReason::SUBTASK_DONE) {
87  // If every servo task is done then emit Done (ignore servos that weren't included in the Task
88  // message)
89  if (((!group.servos.contains(utility::actuation::ServoMap<Elements>::value) || elements.done)
90  && ...)) {
91  emit<Task>(std::make_unique<Done>());
92  return;
93  }
94  // Emit Idle if all the servos are not Done yet
95  emit<Task>(std::make_unique<Idle>());
96  return;
97  }
98 
99  (
100  // Runs an emit for each servo, if that servo was included in the map
101  [&] {
102  // Check if the Task includes this servo
103  if (group.servos.contains(utility::actuation::ServoMap<Elements>::value)) {
104  // Emit a Task for the servo if the map contains a value for it
105  emit<Task>(std::make_unique<Elements>(
106  group.servos.at(utility::actuation::ServoMap<Elements>::value)));
107  }
108  else { // if a servo was not filled in the map for this group, log it
109  log<NUClear::TRACE>("Requested a Servo Group Task but did not provide values for Servo",
110  ServoID(utility::actuation::ServoMap<Elements>::value));
111  }
112  }(),
113  ...);
114  });
115  }
116 
117  /// @brief Keeps track of which Group in the Sequence in add_sequence_provider should be emitted next
118  /// @details Should be used carefully and may be bug-prone.
119  /// @tparam Sequence Used to have unique counts for each Sequence Provider.
120  template <typename Sequence>
121  struct Count {
122  long unsigned int count = 0;
123  };
124 
125  /// @brief Creates a reaction that takes a vector of servo wrapper tasks (eg LeftLegs) and emits each servo
126  /// wrapper task after the previous one is Done. Emits Done when the last group is Done.
127  /// @tparam Sequence is a vector of servo wrapper tasks (eg LeftLegSequence)
128  /// @tparam Group is a servo wrapper task (eg LeftLeg)
129  template <typename Sequence, typename Group>
130  void add_sequence_provider() {
131  // Message to keep track of which position in the vector is to be emitted
132  // Make an initial count message
133  emit<Scope::DIRECT>(std::make_unique<Count<Sequence>>(0));
134 
135  on<Provide<Sequence>, Needs<Group>, With<Count<Sequence>>>().then(
136  [this](const Sequence& sequence, const RunInfo& info, const Count<Sequence>& count) {
137  // If the user gave us nothing then we are done
138  if (sequence.frames.empty()) {
139  emit<Task>(std::make_unique<Done>());
140  }
141  // If this is a new task, run the first pack of servos and increment the counter
142  else if (info.run_reason == RunInfo::RunReason::NEW_TASK) {
143  emit<Task>(std::make_unique<Group>(sequence.frames[0]));
144  emit<Scope::DIRECT>(std::make_unique<Count<Sequence>>(1));
145  }
146  // If the subtask is done, we are done if it is the last servo frames, otherwise use the count to
147  // determine the current frame to emit
148  else if (info.run_reason == RunInfo::RunReason::SUBTASK_DONE) {
149  if (count.count < sequence.frames.size()) {
150  emit<Task>(std::make_unique<Group>(sequence.frames[count.count]));
151  emit<Scope::DIRECT>(std::make_unique<Count<Sequence>>(count.count + 1));
152  }
153  else {
154  emit<Task>(std::make_unique<Done>());
155  }
156  }
157  // Any other run reason shouldn't happen
158  else {
159  emit<Task>(std::make_unique<Idle>());
160  }
161  });
162  }
163 
164  public:
165  /// @brief Called by the powerplant to build and setup the Servos reactor.
166  explicit Servos(std::unique_ptr<NUClear::Environment> environment);
167  };
