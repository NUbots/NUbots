34  template <typename Scalar>
35  class VirtualLoadSensor {
36  public:
37  VirtualLoadSensor() = default;
38  VirtualLoadSensor(const ::extension::Configuration& config) {
39  // Bayes settings
40  noise_factor = config["filter"]["noise_factor"].as<Scalar>();
41  certainty_threshold = config["filter"]["certainty_threshold"].as<Scalar>();
42  uncertainty_threshold = config["filter"]["uncertainty_threshold"].as<Scalar>();
43 
44 
45  // Add the servos
46  for (const auto& field : config["network"]["input"]["servos"].config) {
47  servos.emplace_back("R_" + field.as<std::string>());
48  servos.emplace_back("L_" + field.as<std::string>());
49  }
50 
51  // Add the fields
52  for (const auto& field : config["network"]["input"]["fields"].config) {
53  if (field.as<std::string>() == "POSITION") {
54  fields.emplace_back(POSITION);
55  }
56  else if (field.as<std::string>() == "LOAD") {
57  fields.emplace_back(LOAD);
58  }
59  else if (field.as<std::string>() == "VELOCITY") {
60  fields.emplace_back(VELOCITY);
61  }
62  }
63 
64  accelerometer = config["network"]["input"]["accelerometer"].as<bool>();
65  gyroscope = config["network"]["input"]["gyroscope"].as<bool>();
66 
67  for (const auto& layer : config["network"]["layers"].config) {
68 
69  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> weights =
70  layer["weights"].as<utility::support::Expression>();
71  Eigen::Matrix<Scalar, Eigen::Dynamic, 1> bias = layer["biases"].as<utility::support::Expression>();
72 
73  layers.emplace_back(weights, bias);
74  }
75  }
76 
77  [[nodiscard]] std::array<bool, 2> updateFeet(const ::message::input::Sensors& sensors) {
78 
79  // Build our input data
80  int index = 0;
81  Eigen::Matrix<Scalar, Eigen::Dynamic, 1> logits((servos.size() * fields.size()) + (accelerometer ? 3 : 0)
82  + (gyroscope ? 3 : 0));
83  for (const auto& servo_id : servos) {
84  const auto& s = sensors.servo[servo_id];
85  for (const auto& f : fields) {
86  switch (f) {
87  case POSITION: logits[index++] = s.present_position; break;
88  case LOAD: logits[index++] = s.load; break;
89  case VELOCITY: logits[index++] = s.present_velocity; break;
90  }
91  }
92  }
93  if (accelerometer) {
94  logits[index++] = sensors.accelerometer.x();
95  logits[index++] = sensors.accelerometer.y();
96  logits[index++] = sensors.accelerometer.z();
97  }
98  if (gyroscope) {
99  logits[index++] = sensors.gyroscope.x();
100  logits[index++] = sensors.gyroscope.y();
101  logits[index++] = sensors.gyroscope.z();
102  }
103 
104  // Run the neural network
105  const auto LAYERS_SIZE = layers.size();
106  for (size_t i = 0; i < LAYERS_SIZE; ++i) {
107 
108  // Weights and bias
109  logits = logits.transpose() * layers[i].first + layers[i].second.transpose();
110 
111  // RELU except last layer
112  if (i + 1 < layers.size()) {
113  logits = (logits.array() > Scalar(0.0)).select(logits, Scalar(0.0));
114  }
115  // Sigmoid final layer
116  else {
117  logits =
118  logits.unaryExpr([](const Scalar& value) { return Scalar(1.0 / (1.0 + std::exp(-value))); });
119  }
120  }
121 
122  // Do the bayes update (1D kalman filter thing)
123  Scalar k = current_noise / (current_noise + noise_factor);
124  state += k * (logits - state);
125  current_noise *= Scalar(1.0) - k;
126  current_noise += Scalar(1.0);
127 
128  for (size_t leg = 0; leg < 2; leg++) {
129  // We have some certainty in our measurement
130  if (state[leg] > certainty_threshold) {
131  output_state[leg] = true;
132  }
133  if (state[leg] < uncertainty_threshold) {
134  output_state[leg] = false;
135  }
136  }
137  return output_state;
138  }
139 
140  Eigen::Matrix<Scalar, 2, 1> state = Eigen::Matrix<Scalar, 2, 1>::Zero();
141 
142 
143  enum Field { POSITION, VELOCITY, LOAD };
144 
145  Scalar noise_factor = Scalar(0.0);
146  Scalar current_noise = Scalar(0.0);
147  Scalar certainty_threshold = Scalar(0.0);
148  Scalar uncertainty_threshold = Scalar(0.0);
149 
150  std::vector<::utility::input::ServoID> servos{};
151  std::vector<Field> fields{};
152 
153  bool accelerometer = false;
154  bool gyroscope = false;
155 
156  std::vector<
157  std::pair<Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>, Eigen::Matrix<Scalar, Eigen::Dynamic, 1>>>
158  layers{};
159 
160  std::array<bool, 2> output_state = {false, false};
161  };
