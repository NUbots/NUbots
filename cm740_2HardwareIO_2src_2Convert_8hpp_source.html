30  struct Convert {
31 
32  /// The value from the CM740 is between 0 and 1023, representing a value between -4g and 4g.
33  /// This means 512 = 0
34  static constexpr double ACCELEROMETER_CONVERSION_FACTOR = (4 * 9.80665) / 512.0;
35 
36  /// The Gyroscope value from the CM740 is between 0 and 1023, representing a value between -500 and
37  /// 500 degrees per second. This means 512 = 0
38  /// 1880.0 is an empirically measured value for this factor, which was calculated by spinning the robot on an
39  /// office chair and comparing how far the robot thought it spun to how much it actually spun.
40  /// Empirical estimates like this which don't match the data sheet are SUSPECT, and should be rechecked
41  /// periodically
42  static constexpr double GYROSCOPE_CONVERSION_FACTOR = (1880.0 * (M_PI / 180.0)) / 512.0;
43 
44  /// The value that comes from the CM740 is measured in decivolts (0.1 of a volt)
45  static constexpr double VOLTAGE_CONVERSION_FACTOR = 0.1;
46 
47  /// The FSR values that are measured by the NUgus' feet are measured in millinewtons
48  static constexpr double FSR_FORCE_CONVERSION_FACTOR = 0.001;
49 
50  /// The gain is a value between 0 and 254, we want a value between 0 and 100
51  static constexpr double GAIN_CONVERSION_FACTOR = 100.0 / 254.0;
52 
53  /// The angle is given as a value between 0 and 4095
54  static constexpr double POSITION_CONVERSION_FACTOR = (2.0 * M_PI) / 4095.0;
55 
56  /// The load is measured as a value between 0 and 2047 where the 10th bit specifies direction and 1024 = 0
57  /// We convert it to a value between -1 and 1 (percentage)
58  static constexpr double LOAD_CONVERSION_FACTOR = 1.0 / 1023.0;
59 
60  /// The torque limit is measured as a value between 0 and 1023. We use it between 0 and 100
61  static constexpr double TORQUE_LIMIT_CONVERSION_FACTOR = 100.0 / 1023.0;
62 
63  /// The temperatures are given in degrees anyway
64  static constexpr double TEMPERATURE_CONVERSION_FACTOR = 1.0;
65 
66  /// The MX64 and MX106 both measure speed between 0 and 1023, with unit of about 0.114rpm
67  /// If the speed is 1023, then the speed is about 116.62 == 0.114 * 1023, per the datasheet
68  static constexpr double SPEED_CONVERSION_FACTOR = (0.114 * 2.0 * M_PI) / 60.0;
69 
70  /// Picks which direction a motor should be measured in (forward or reverse) -- configurable based on
71  /// the specific humanoid being used.
72  static int8_t SERVO_DIRECTION[20];
73 
74  /// Offsets the radian angles of motors to change their 0 position -- configurable based on the specific
75  /// humanoid being used.
76  static double SERVO_OFFSET[20];
77 
78  template <int bit>
79  static bool getBit(const uint16_t& value) {
80  return (value & (1 << bit)) == (1 << bit);
81  }

84  static float gyroscope(const uint16_t& value);
85  static float voltage(const uint8_t& value);

87  static float fsrCentre(const bool& left, const uint8_t& value);
88 
89  static std::tuple<uint8_t, uint8_t, uint8_t> colourLED(const uint16_t& value);
90  static uint16_t colourLEDInverse(const uint8_t& r, const uint8_t& g, const uint8_t& b);
91 
92  static float gain(const uint8_t& value);
93  static uint8_t gainInverse(const float& value);
94 
95  static float servoPosition(const uint8_t& id, const uint16_t& value);
96  static uint16_t servoPositionInverse(const uint8_t& id, const float& value);
97 
98  static float servoSpeed(const uint8_t& id, const uint16_t& value);
99  static uint16_t servoSpeedInverse(const float& value);
100 
101  static float torqueLimit(const uint16_t& value);
102  static uint16_t torqueLimitInverse(const float& value);
103 
104  static float servoLoad(const uint8_t& id, const uint16_t& value);
105 
106  static float temperature(const uint8_t& value);
