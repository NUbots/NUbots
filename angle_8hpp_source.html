39  /**
40  * Takes an angle in radians and normalizes it to be between -pi and pi
41  *
42  * @param value the angle in radians
43  *
44  * @return the angle between -pi and pi
45  */
46  template <typename T>
47  inline T normalizeAngle(const T value) {
48 
49  T angle = std::fmod(value, static_cast<T>(2 * M_PI));
50 
51  if (angle <= -M_PI) {
52  angle += 2 * M_PI;
53  }
54 
55  if (angle > M_PI) {
56  angle -= 2 * M_PI;
57  }
58 
59  return angle;
60  }

72  /**
73  * Finds the angle between two vectors. This method is numerically stable when the vectors are close to parallel.
74  *
75  * @details Based on the implementation found here https://www.plunk.org/~hatch/rightway.html
76  *
77  * @param u A unit vector
78  * @param v A unit vector
79  *
80  * @tparam Derived Type derived from `Eigen::MatrixBase` (i.e. any dense matrix type)
81  * @tparam Scalar Type of each element stored in the dense matrix (must be a floating point type)
82  *
83  * @return The acute angle in range [0,pi] between two vectors.
84  */
85  template <typename DerivedU, typename DerivedV>
86  auto angle_between(const Eigen::MatrixBase<DerivedU>& u, const Eigen::MatrixBase<DerivedV>& v) ->
87  typename std::enable_if_t<
88  std::is_floating_point_v<typename DerivedU::Scalar> && std::is_floating_point_v<typename DerivedV::Scalar>,
89  typename DerivedU::Scalar> {
90  using Scalar = typename DerivedU::Scalar;
91  return u.dot(v) < Scalar(0) ? std::numbers::pi_v<Scalar> - Scalar(2) * std::asin((-v - u).norm() * Scalar(0.5))
92  : Scalar(2) * std::asin((v - u).norm() * Scalar(0.5));
93  }

95  /**
96  * Calculates the difference between two angles between -pi and pi
97  * Method:
98  * http://math.stackexchange.com/questions/1158223/solve-for-x-where-a-sin-x-b-cos-x-c-where-a-b-and-c-are-kno
99  * @param
100  */
101  inline double difference(const double a, const double b) {
102 
103  return M_PI - std::fabs(std::fmod(std::fabs(a - b), (2 * M_PI)) - M_PI);
104  }

106  /**
107  * Calculates the signed angle with the smallest absolute value such
108  * that: normalizeAngle(b + signedDifference(a, b)) == normalizeAngle(a).
109  * Method: http://stackoverflow.com/a/7869457
110  */
111  inline double signedDifference(const double a, const double b) {
112 
113  auto x = a - b;
114 
115  auto m = x - std::floor(x / (2 * M_PI)) * 2 * M_PI;
116 
117  auto d = std::fmod(m + M_PI, (2 * M_PI)) - M_PI;
118 
119  return d;
120  }

122  /**
123  * Compute the oriented distance between the two given angle
124  * in the range -PI/2:PI/2 radian from angleSrc to angleDst
125  * (Better than doing angleDst-angleSrc)
126  */
127  template <typename Scalar>
128  inline double angleDistance(const Scalar& angleSrc, const Scalar& angleDst) {
129  const Scalar source = normalizeAngle(angleSrc);
130  const Scalar dest = normalizeAngle(angleDst);
131 
132  const Scalar max = source > dest ? source : dest;
133  const Scalar min = source > dest ? dest : source;
134 
135  const Scalar dist1 = max - min;
136  const Scalar dist2 = (2.0 * M_PI) - max + min;
137  // Whichever distance is smaller is used, and we make sure that the rotation is in the correct direction
138  // by negating the distance if it's clockwise
139  return dist1 < dist2 ? (source > dest ? -dist1 : dist1) : (source > dest ? dist2 : -dist2);
140  }

146  /*! @brief Solves for x in $a \sin(x) + b \cos(x) = c ; x \in [0,\pi]$
147  */
148  inline float solveLinearTrigEquation(float a, float b, float c) {
149  float norm = std::sqrt(a * a + b * b);
150  if (norm == 0) {
151  throw std::domain_error(
152  "utility::math::angle::solveLinearTrigEquation - std::sqrt(a*a+b*b) == 0 => Any value for x is "
153  "a "
154  "solution");
155  }
156 
157  // Normalise equation
158  float a_ = a / norm;
159  float b_ = b / norm;
160  float c_ = c / norm;
161 
162  if (std::fabs(c_) > 1) {
163  throw std::domain_error("utility::math::angle::solveLinearTrigEquation - no solution |c_|>1");
164  }
165 
166  // Find alpha such that $\sin(\alpha) = a\_$ and $\cos(\alpha) = b\_$, which is possible because $a\_^2
167  // + b\_^2 = 1$
168  float alpha = std::atan2(a_, b_);
169 
170  // Hence the equation becomes $\cos(\alpha)\cos(x)+\sin(\alpha)\sin(x) = cos(x-\alpha) = c\_$
171  return alpha + acos_clamped(c_);
172  }
