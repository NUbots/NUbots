31  /**
32  * Takes an angle in radians and normalizes it to be between -pi and pi
33  *
34  * @param value the angle in radians
35  *
36  * @return the angle between -pi and pi
37  */
38  template <typename T>
39  inline T normalizeAngle(const T value) {
40 
41  T angle = std::fmod(value, static_cast<T>(2 * M_PI));
42 
43  if (angle <= -M_PI) {
44  angle += 2 * M_PI;
45  }
46 
47  if (angle > M_PI) {
48  angle -= 2 * M_PI;
49  }
50 
51  return angle;
52  }

64  /**
65  * Calculates the difference between two angles between -pi and pi
66  * Method:
67  * http://math.stackexchange.com/questions/1158223/solve-for-x-where-a-sin-x-b-cos-x-c-where-a-b-and-c-are-kno
68  * @param
69  */
70  inline double difference(const double a, const double b) {
71 
72  return M_PI - std::fabs(std::fmod(std::fabs(a - b), (2 * M_PI)) - M_PI);
73  }

75  /**
76  * Calculates the signed angle with the smallest absolute value such
77  * that: normalizeAngle(b + signedDifference(a, b)) == normalizeAngle(a).
78  * Method: http://stackoverflow.com/a/7869457
79  */
80  inline double signedDifference(const double a, const double b) {
81 
82  auto x = a - b;
83 
84  auto m = x - std::floor(x / (2 * M_PI)) * 2 * M_PI;
85 
86  auto d = std::fmod(m + M_PI, (2 * M_PI)) - M_PI;
87 
88  return d;
89  }

91  /**
92  * Compute the oriented distance between the two given angle
93  * in the range -PI/2:PI/2 radian from angleSrc to angleDst
94  * (Better than doing angleDst-angleSrc)
95  */
96  template <typename Scalar>
97  inline double angleDistance(const Scalar& angleSrc, const Scalar& angleDst) {
98  const Scalar source = normalizeAngle(angleSrc);
99  const Scalar dest = normalizeAngle(angleDst);
100 
101  const Scalar max = source > dest ? source : dest;
102  const Scalar min = source > dest ? dest : source;
103 
104  const Scalar dist1 = max - min;
105  const Scalar dist2 = (2.0 * M_PI) - max + min;
106  // Whichever distance is smaller is used, and we make sure that the rotation is in the correct direction
107  // by negating the distance if it's clockwise
108  return dist1 < dist2 ? (source > dest ? -dist1 : dist1) : (source > dest ? dist2 : -dist2);
109  }

115  /*! @brief Solves for x in $a \sin(x) + b \cos(x) = c ; x \in [0,\pi]$
116  */
117  inline float solveLinearTrigEquation(float a, float b, float c) {
118  float norm = std::sqrt(a * a + b * b);
119  if (norm == 0) {
120  throw std::domain_error(
121  "utility::math::angle::solveLinearTrigEquation - std::sqrt(a*a+b*b) == 0 => Any value for x is "
122  "a "
123  "solution");
124  }
125 
126  // Normalise equation
127  float a_ = a / norm;
128  float b_ = b / norm;
129  float c_ = c / norm;
130 
131  if (std::fabs(c_) > 1) {
132  throw std::domain_error("utility::math::angle::solveLinearTrigEquation - no solution |c_|>1");
133  }
134 
135  // Find alpha such that $\sin(\alpha) = a\_$ and $\cos(\alpha) = b\_$, which is possible because $a\_^2
136  // + b\_^2 = 1$
137  float alpha = std::atan2(a_, b_);
138 
139  // Hence the equation becomes $\cos(\alpha)\cos(x)+\sin(\alpha)\sin(x) = cos(x-\alpha) = c\_$
140  return alpha + acos_clamped(c_);
141  }
