114  /// @brief Get a reference to the DynamixelDevice with the given ID
115  /// @param id The ID of the device to get
116  /// @return A reference to the DynamixelDevice with the given ID
117  constexpr DynamixelDevice& operator[](const ID& id) {
118  switch (id) {
119  case ID::OPENCR: return OPENCR;
120  case ID::R_SHOULDER_PITCH: return R_SHOULDER_PITCH;

126  case ID::R_HIP_YAW: return R_HIP_YAW;

128  case ID::R_HIP_ROLL: return R_HIP_ROLL;
129  case ID::L_HIP_ROLL: return L_HIP_ROLL;
130  case ID::R_HIP_PITCH: return R_HIP_PITCH;
131  case ID::L_HIP_PITCH: return L_HIP_PITCH;
132  case ID::R_KNEE: return R_KNEE;
133  case ID::L_KNEE: return L_KNEE;

135  case ID::L_ANKLE_PITCH: return L_ANKLE_PITCH;
136  case ID::R_ANKLE_ROLL: return R_ANKLE_ROLL;

138  case ID::HEAD_YAW: return HEAD_YAW;
