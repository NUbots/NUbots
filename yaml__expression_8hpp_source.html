32  /**
33  * Represents a mathematical expression
34  * This could either be represented as a double or as a vector/matrix.
35  */
36  struct Expression {
37 
38  Expression() = default;
39  Expression(const YAML::Node& node) : node(node) {}
40 
41  operator double() {
42  double value = 0.0;
43 
44  try {
45  value = parse_math_string<double>(node.as<std::string>());
46  }
47 
48  catch (const std::invalid_argument& ex) {
49  throw std::invalid_argument(fmt::format("Unable to convert node to arithmetic type.\n{}", ex.what()));
50  }
51 
52  return value;
53  }

55  // Handle fixed-sized matrices.
56  template <typename T, std::enable_if_t<((T::RowsAtCompileTime > 1) && (T::ColsAtCompileTime > 1))>* = nullptr>
57  operator T() const {
58 
59  // value : [[a, b], [c, d]]
60  const size_t rows = node.size();
61  const size_t cols = node[0].size();
62 
63  // Count the columns on every row.
64  for (const auto& row : node) {
65  if (row.size() != cols) {
66  throw std::out_of_range(
67  fmt::format("Inconsistent number of cols in matrix (cols: {} vs {}).", row.size(), cols));
68  }
69  }
70 
71  // Validate row and column sizes.
72  if ((rows != T::RowsAtCompileTime) || (cols != T::ColsAtCompileTime)) {
73  throw std::out_of_range(
74  fmt::format("Rows and columns in YAML matrix do not align with output matrix. "
75  "(rows: {} vs {}, cols: {} vs {}).",
76  rows,
77  T::RowsAtCompileTime,
78  cols,
79  T::ColsAtCompileTime));
80  }
81 
82  T matrix;
83 
84  try {
85  for (size_t row = 0; row < rows; row++) {
86  for (size_t col = 0; col < cols; col++) {
87  matrix(row, col) = parse_math_string<typename T::Scalar>(node[row][col].as<std::string>());
88  }
89  }
90  }
91  catch (const std::invalid_argument& ex) {
92  throw std::invalid_argument(fmt::format("Unable to convert node to arithmetic type.\n{}", ex.what()));
93  }
94 
95  return matrix;
96  }

277  template <typename T, std::size_t N, typename U = T>
278  inline std::array<T, N> resolve_expression(const YAML::Node& config) {
279  std::array<T, N> result;
280  size_t i = 0;
281  for (auto& data : config.as<std::vector<Expression>>()) {
282  result[i++] = U(data);
283  }
284  assert((fmt::format("We expected {} elements in the YAML file but {} were found", N, i), i == N));
285  return result;
286  }
287 
288  template <typename T, typename U = T>
289  inline std::vector<T> resolve_expression(const YAML::Node& config) {
290  std::vector<T> result;
291  for (auto& data : config.as<std::vector<Expression>>()) {
292  result.emplace_back(U(data));
293  }
294  return result;
295  }
