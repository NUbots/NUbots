56  class Index {
57  public:
58  enum class Ordering {
59  /// Don't sort the nbs files, just load the indexes in order
60  NONE,
61  /// Time ordering orders the messages by their timestamp
62  TIME,
63  /// Transmission order splits the messages into subtypes and orders them such that increasing transmission
64  /// times result in increasing message density (increased resolution)
65  /// This ordering is calculated as the following, for each type/subtype combination, the data is treated as
66  /// a maxheap based on time. This data is then interleaved so that for each message type/subtype the amount
67  /// of bytes of data is kept equal as transfer increases. Eventually some of the message types/subtypes will
68  /// be exhausted using this and the remaining ones will transfer. As an example, sensor packets are much
69  /// smaller than GPS packets, so many of those will be in the initial order until they run out and only
70  /// images remain.
71  TRANSMISSION
72  };

79  /**
80  * @brief Construct a new Index object based on the provided list of paths
81  *
82  * @param paths the paths to the nbs files we are getting the index for
83  * @param show_progress if the progress of loading the index should be printed
84  */
85  template <typename T>
86  Index(const T& paths,
87  const Ordering& order = Ordering::TIME,
88  const bool& show_progress = false,
89  const bool& big_index_mode = false) requires
90  utility::type_traits::range_constructible<T, std::filesystem::path> : idx(big_index_mode) {
91 
92  // Get the total size of all the nbs files that we are loading
93  int64_t total_data = std::accumulate(std::begin(paths),
94  std::end(paths),
95  int64_t(0),
96  [](const int64_t& total, const std::filesystem::path& p) {
97  return total + std::filesystem::file_size(p);
98  });
99 
100  int64_t loaded_data = 0;
101  std::optional<utility::support::ProgressBar> load_progress;
102  if (show_progress) {
103  load_progress = utility::support::ProgressBar();
104  }
105 
106  for (auto [i, path] : utility::support::enumerate(paths)) {
107  if (show_progress) {
108  auto file_size = utility::support::si_unit(std::filesystem::file_size(path));
109  load_progress.value().update(
110  loaded_data,
111  total_data,
112  "B",
113  fmt::format("Loading {:.2f}{}B from {}", file_size.first, file_size.second, path.string()));
114  }
115 
116  std::filesystem::path idx_path = path.string() + ".idx";
117 
118  // If our index file does not exist we need to create it
119  if (!std::filesystem::exists(idx_path)) {
120  build_index(path, idx_path, show_progress);
121  }
122 
123  // Load the index file
124  zstr::ifstream input(idx_path);
125  while (input.good()) {
126  IndexItemFile item{};
127  input.read(reinterpret_cast<char*>(&item.item), sizeof(IndexItem));
128  item.fileno = i;
129 
130  if (input.good()) {
131  // Can't bind packed fields, so need to construct the key by copy
132  std::pair<uint64_t, uint32_t> key;
133  key.first = item.item.type;
134  key.second = item.item.subtype;
135  if (!data_types.contains(key)) {
136  data_types[key] = 0;
137  }
138  data_types[key]++;
139  idx.push(item);
140  }
141  }
142 
143  loaded_data += std::filesystem::file_size(path);
144  }
145 
146  if (show_progress) {
147  load_progress.value().update(total_data, total_data, "B", "Loaded NBS files");
148  load_progress.value().close();
149  }
150 
151  // Finalise any memory we used and memory map if required, this is all the index items
152  idx.finalise();
153 
154  // INDEX ORDERING
155  switch (order) {
156  case Ordering::TIME: {
157  // Sort by time
158  std::stable_sort(this->begin(), this->end(), [](const IndexItemFile& a, const IndexItemFile& b) {
159  return a.item.timestamp < b.item.timestamp;
160  });
161  } break;
162  case Ordering::TRANSMISSION: {
163 
164  // Use the timestamp sort the list
165  // Once sorted we can use this data to build a binary search tree later
166  int64_t sort_progress = 0;
167  int64_t total_sort_steps = idx.end() - idx.begin();
168  total_sort_steps = int64_t(double(total_sort_steps) * std::log2(total_sort_steps));
169  int64_t sort_step = total_sort_steps / 10000;
170 
171  std::optional<utility::support::ProgressBar> sort_progress_bar;
172  if (show_progress) {
173  sort_progress_bar = utility::support::ProgressBar();
174  }
175 
176  // Sort the data
177  std::sort(idx.begin(), idx.end(), [&](const IndexItemFile& a, const IndexItemFile& b) {
178  if (show_progress) {
179  if (sort_progress++ % sort_step == 0) {
180  // Ensure that if we go over 100% we keep increasing the limit
181  total_sort_steps = std::max(total_sort_steps, sort_progress + 1);
182 
183  sort_progress_bar.value().update(sizeof(IndexItemFile) * ++sort_progress,
184  sizeof(IndexItemFile) * total_sort_steps,
185  "B",
186  "Sorting");
187  }
188  }
189 
190  return a.item.type != b.item.type ? a.item.type < b.item.type
191  : a.item.subtype != b.item.subtype ? a.item.subtype < b.item.subtype
192  : a.item.timestamp < b.item.timestamp;
193  });
194  if (show_progress) {
195  sort_progress_bar.value().update(sizeof(IndexItemFile) * total_sort_steps,
196  sizeof(IndexItemFile) * total_sort_steps,
197  "B",
198  "Sorting");
199  sort_progress_bar.value().close();
200  }
201 
202  // Split up each of the sections so we can use it for transmission later
203  struct InterleaveTracking {
204  /// The current state of the binary tree partitions
205  std::vector<std::array<IndexItemFile*, 2>> ranges;
206  /// The number of elements we have loaded from the tree
207  int64_t n_loaded{0};
208  /// The number of elements that are available in this type
209  int64_t n_elements{0};
210  /// The number of bytes that have been loaded from this type
211  int64_t size{0};
212 
213  InterleaveTracking(const std::array<IndexItemFile*, 2>& ranges)
214  : ranges({ranges}), n_elements(ranges[1] - ranges[0]) {}
215 
216  inline IndexItemFile& get() {
217  // Current value is the middle of the last range element
218  return *(ranges.back()[0] + ((ranges.back()[1] - ranges.back()[0]) / 2));
219  }
220 
221  inline bool finished() {
222  // We are finished if the current element is the last element
223  return n_loaded + 1 >= n_elements;
224  }
225 
226  /**
227  * Gets the next element that would show up in the BST
228  */
229  inline void advance() {
230 
231  // Advance our counters before we start (we have "loaded" the current element)
232  // Only do this if there is a real element at this point
233  if (ranges.back()[0] != ranges.back()[1]) {
234  this->n_loaded++;
235  this->size += this->get().item.length;
236  }
237 
238  if (n_loaded >= n_elements) {
239  throw std::runtime_error("All elements have already been loaded");
240  }
241 
242  // We start with one element so the initial case of this recalc_point is 0
243  int recalc_point = int(ranges.size()) - 1;
244 
245  while (true) {
246  // We still have elements to check
247  if (recalc_point > 0) {
248  auto& p = ranges[recalc_point - 1];
249  auto& c = ranges[recalc_point];
250 
251  // Check if we are the left child of the parent and should move to its right child
252  // Our parent needs 3 or more elements if we are to move to it's right child
253  // It must have enough for left, self, right
254  if (c[0] == p[0] && p[1] - p[0] > 2) {
255  // Second half does not include the parents element (+1 to the midpoint)
256  c = {p[0] + ((p[1] - p[0]) / 2) + 1, p[1]};
257  break;
258  }
259  // If we have two or less elements we advance up the tree
260  --recalc_point;
261  }
262  // Root element reached, add another layer
263  else {
264  ranges.emplace_back();
265  break;
266  }
267  }
268 
269  // Recalculate from the point after the recalc_point to the end
270  for (int i = recalc_point + 1; i < int(ranges.size()); ++i) {
271  auto& prev = ranges[i - 1];
272  ranges[i] = {prev[0], prev[0] + ((prev[1] - prev[0]) / 2)};
273  }
274 
275  // If we hit an empty element then try again
276  if (ranges.back()[0] == ranges.back()[1]) {
277  advance();
278  }
279  }
280  };
281  std::vector<InterleaveTracking> sections;
282 
283  // Split sections and prepare our interleaving structure
284  for (const auto& type : data_types) {
285  IndexItemFile v{};
286  v.item.type = type.first.first;
287  v.item.subtype = type.first.second;
288  auto range = std::equal_range(idx.begin(),
289  idx.end(),
290  v,
291  [](const IndexItemFile& a, const IndexItemFile& b) {
292  return a.item.type != b.item.type
293  ? a.item.type < b.item.type
294  : a.item.subtype < b.item.subtype;
295  });
296  sections.emplace_back(std::array<IndexItemFile*, 2>{range.first, range.second});
297  }
298 
299  std::optional<utility::support::ProgressBar> interleave_progress_bar;
300  if (show_progress) {
301  interleave_progress_bar = utility::support::ProgressBar();
302  }
303 
304  // Interleave the types based on size back into the original list
305  IndexStorage reordered(big_index_mode);
306  int64_t total_interleave_steps = (idx.end() - idx.begin());
307  int64_t interleave_step = total_interleave_steps / 10000;
308  for (int64_t i = 0; !sections.empty(); ++i) {
309  // Find which of our data types has the smallest size loaded so far
310  auto s = std::min_element(
311  sections.begin(),
312  sections.end(),
313  [&](const InterleaveTracking& a, const InterleaveTracking& b) { return a.size < b.size; });
314 
315  // Store the current item
316  auto& item = s->get();
317  reordered.push(item);
318 
319  // We loaded the last bit of data, erase this type from consideration
320  if (s->finished()) {
321  sections.erase(s);
322  }
323  // Advance to the next item
324  else {
325  s->advance();
326  }
327 
328  if (show_progress) {
329  if (i % interleave_step == 0) {
330  interleave_progress_bar.value().update(sizeof(IndexItemFile) * i,
331  sizeof(IndexItemFile) * total_interleave_steps,
332  "B",
333  "Interleaving");
334  }
335  }
336  }
337  // Send a final update to the progress bar
338  if (show_progress) {
339  interleave_progress_bar.value().update(sizeof(IndexItemFile) * total_interleave_steps,
340  sizeof(IndexItemFile) * total_interleave_steps,
341  "B",
342  "Interleaving");
343  interleave_progress_bar.value().close();
344  }
345  reordered.finalise();
346 
347  idx = std::move(reordered);
348  } break;
349  case Ordering::NONE: break;
350  default: throw std::runtime_error("An invalid ordering was provided");
351  }
352  }
