33  /**
34  * @brief Systematic resampling. See [1] for details on the algorithm (a copy is in
35  * doc/Localisation/On_Resampling_Algorithms_for_Particle_Filters.pdf)
36  *
37  * [1]J. Hol, T. Schon and F. Dustafsson, "On Resampling Algorithms for Particle Filters", in 2006 IEEE Nonlinear
38  * Statistical Signal Processing Workshop, Cambridge. UK, 2006.
39  *
40  * @tparam Scalar The scalar type to used for the particle weights
41  */
42  template <typename Scalar>
43  struct Systematic {
44  /**
45  * @brief Systematic resampling weight generator
46  *
47  * @param count_ Number of particles being resampled
48  */
49  Systematic(const int& count_)
50  : gen((std::random_device()())), dist(Scalar(0.0), Scalar(1.0)), count(count_), rng(dist(gen)) {}
51 
52  /**
53  * @brief Calculate the next resampling weight
54  *
55  * @param i Particle index being resampling
56  */
57  [[nodiscard]] Scalar operator()(const int& i) {
58  return (Scalar(i) + rng) / count;
59  }
60 
61  private:
62  /// @brief Uniform distribution on the interval [0, 1)
63  std::mt19937 gen;
64  std::uniform_real_distribution<Scalar> dist;
65 
66  /// @brief Number of particles being resampled
67  Scalar count;
68  /// @brief Systematic resampling uses a static weight offset
69  Scalar rng;
70  };
71 
72  /**
73  * @brief Call the resampler algorithm using the above resampling weight calculation
74  *
75  * @param count Number of particles being resampled
76  * @param begin Iterator to the first weight to use for resampling
77  * @param end Iterator to the last weight to use for resampling
78  */
79  template <typename Iterator>
80  [[nodiscard]] std::vector<int> systematic(const int& count, Iterator&& begin, Iterator&& end) {
81  return resample<Systematic>(count, std::forward<Iterator>(begin), std::forward<Iterator>(end));
82  }
