36  /**
37  * @brief This struct mimics the expected data structure for a Reboot command.
38  *
39  * @details
40  * This type has it's members arranged in the same way as a raw array of this command would. Because of this
41  * you cannot add or remove members from this type.
42  *
43  * @author Alex Biddulph
44  */
45  struct RebootCommand {
46  explicit RebootCommand(uint8_t id)
47  : magic(0x00FDFFFF)
48  , id(id)
49  , length(3)
50  , instruction(Instruction::REBOOT)
51  , checksum(calculate_checksum(this)) {}
52 
53  /// Magic number that heads up every packet
54  const uint32_t magic;
55  /// The ID of the device that we are communicating with
56  const uint8_t id;
57  /// The total length of the data packet (always 3)
58  const uint16_t length;
59  /// The instruction that we will be executing
60  const uint8_t instruction;
61  /// Our checksum for this command
62  const uint16_t checksum;
63  } __attribute__((packed)); // Make it so that the compiler reads this struct "as is" (no padding bytes)
64  // Check that this struct is not cache alligned
65  static_assert(sizeof(RebootCommand) == 10, "The compiler is adding padding to this struct, Bad compiler!");
66 
67 } // namespace dynamixel::v2
