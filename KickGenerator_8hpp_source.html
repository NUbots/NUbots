46  template <typename Scalar>
47  class KickGenerator {
48  public:
49  /**
50  * @brief Get the swing foot pose at the current time.
51  * @return Trajectory of torso.
52  */
53  Eigen::Transform<Scalar, 3, Eigen::Isometry> get_swing_foot_pose() const {
54  return swingfoot_trajectory.pose(t);
55  }
56 
57  /**
58  * @brief Get the torso pose object at the current time.
59  * @return Pose of torso.
60  */
61  Eigen::Transform<Scalar, 3, Eigen::Isometry> get_torso_pose() const {
62  return torso_trajectory.pose(t);
63  }
64 
65  /**
66  * @brief Get the left or right foot pose at the given time in the torso {t} frame.
67  * @param t Time.
68  * @param limb Limb ID of foot to get pose of.
69  * @return Swing foot pose at time t.
70  */
71  Eigen::Transform<Scalar, 3, Eigen::Isometry> get_foot_pose(const LimbID& limb) const {
72  // Assign the value based on the foot planted
73  Eigen::Transform<Scalar, 3, Eigen::Isometry> Htl =
74  left_foot_is_planted ? get_torso_pose().inverse() : get_torso_pose().inverse() * get_swing_foot_pose();
75  Eigen::Transform<Scalar, 3, Eigen::Isometry> Htr =
76  left_foot_is_planted ? get_torso_pose().inverse() * get_swing_foot_pose() : get_torso_pose().inverse();
77 
78  // Return the desired pose of the specified foot
79  if (limb == LimbID::LEFT_LEG)
80  return Htl;
81  if (limb == LimbID::RIGHT_LEG)
82  return Htr;
83  throw std::runtime_error("Invalid Limb ID");
84  }
85 
86  /**
87  * @brief Reset kick generator.
88  */
89  void reset() {
90  // Reset time
91  t = 0;
92 
93  // Generate trajectories
94  generate_trajectories();
95  }
96 
97  /**
98  * @brief Run an update of the walk engine, updating the time and engine state and trajectories.
99  * @param dt Time step.
100  * @return Engine state.
101  */
102  void update(const Scalar& dt, const LimbID& limb) {
103 
104  // Generate either left or right foot kick trajectory
105  if (limb == LimbID::LEFT_LEG) {
106  left_foot_is_planted = false;
107  }
108  else if (limb == LimbID::RIGHT_LEG) {
109  left_foot_is_planted = true;
110  }
111  else {
112  throw std::runtime_error("Invalid Limb ID");
113  }
114 
115  generate_trajectories();
116 
117  // Check for negative time step
118  if (dt <= 0.0f) {
119  NUClear::log<NUClear::WARN>("dt <= 0.0f");
120  return;
121  }
122 
123  // Check for too long dt
124  if (dt > kick_duration) {
125  NUClear::log<NUClear::WARN>("dt > kick_duration", dt, kick_duration);
126  return;
127  }
128 
129  // Update the phase
130  t += dt;
131 
132  // Clamp the time to the duration of the kick if greater or equal to the duration
133  if (t >= kick_duration) {
134  t = kick_duration;
135  }
136  }
137 
138  /**
139  * @brief Getter for the current time.
140  * @return Current time.
141  */
142  Scalar get_time() const {
143  return t;
144  }
145 
146  /**
147  * @brief Getter for the duration of the kick.
148  * @return Duration of the kick.
149  */
150  Scalar get_duration() const {
151  return kick_duration;
152  }
153 
154  /**
155  * @brief Get whether or not the left foot is planted.
156  * @return True if left foot is planted, false otherwise.
157  */
158  bool is_left_foot_planted() const {
159  return left_foot_is_planted;
160  }

162  /**
163  * @brief Add waypoints to the foot trajectory.
164  */
165  void add_foot_waypoint(const Waypoint<Scalar>& waypoint) {
166  foot_waypoints.push_back(waypoint);
167  }
168 
169  /**
170  * @brief Add waypoints to the torso trajectory.
171  */
172  void add_torso_waypoint(const Waypoint<Scalar>& waypoint) {
173  torso_waypoints.push_back(waypoint);
174  }
175 
176  /**
177  * @brief Clear all waypoints from the foot and torso trajectories.
178  */
179  void clear_waypoints() {
180  foot_waypoints.clear();
181  torso_waypoints.clear();
182  }
183 
184  private:
185  // ******************************** State ********************************
186 
187  /// @brief Duration of complete kick.
188  Scalar kick_duration = 0.0;
189 
190  /// @brief Current time in the step cycle.
191  Scalar t = 0.0;
192 
193  /// @brief Whether the left foot is planted.
194  bool left_foot_is_planted = false;
195 
196  // ******************************** Trajectories ********************************
197 
198  /// @brief Foot waypoints
199  std::vector<Waypoint<Scalar>> foot_waypoints{};
200 
201  /// @brief Torso waypoints
202  std::vector<Waypoint<Scalar>> torso_waypoints{};
203 
204  // 6D piecewise polynomial trajectory for swing foot.
205  Trajectory<Scalar> swingfoot_trajectory{};
206 
207  // 6D piecewise polynomial trajectory for torso.
208  Trajectory<Scalar> torso_trajectory{};
209 
210  /**
211  * @brief Generate swing foot trajectory.
212  * @return Trajectory of swing foot to kick.
213  */
214  void generate_trajectories() {
215  // Clear current trajectories
216  swingfoot_trajectory.clear();
217  torso_trajectory.clear();
218 
219  // Add waypoints to trajectories
220  for (auto waypoint : foot_waypoints) {
221  // Flip y position sign based on which foot is planted
222  waypoint.position.y() *= left_foot_is_planted ? -1.0 : 1.0;
223  swingfoot_trajectory.add_waypoint(waypoint);
224  }
225 
226  for (auto waypoint : torso_waypoints) {
227  // Flip y position sign based on which foot is planted
228  waypoint.position.y() *= left_foot_is_planted ? -1.0 : 1.0;
229  torso_trajectory.add_waypoint(waypoint);
230  }
231 
232  kick_duration = foot_waypoints.back().time_point;
233 
234  // Assert that the first timepoint of both trajectories is 0
235  if (foot_waypoints.front().time_point != 0.0 || torso_waypoints.front().time_point != 0.0) {
236  throw std::runtime_error("The first waypoint of the foot and torso trajectories must start at time 0.");
237  }
238 
239  // Assert that the length of the foot waypoints is the same as the length of the torso waypoints
240  if (foot_waypoints.back().time_point != torso_waypoints.back().time_point) {
241  throw std::runtime_error(
242  "The last waypoint of the foot and torso trajectories must have the same time point.");
243  }
244  }
245  };
