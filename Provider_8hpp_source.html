45  enum class Classification {
46  /// Provide providers are typical providers that run when given a task
47  PROVIDE = int(::extension::behaviour::commands::ProviderClassification::PROVIDE),
48  /// Start providers run once when a provider group first gets a task

56  /**
57  * The data held by a `When` condition in order to monitor when the condition will be met, as well as to
58  * compare it to `Causing` statements to determine if those `Causing` statements would satisfy this Provider
59  */
60  struct WhenCondition {
61  WhenCondition(const std::type_index& type_,
62  std::function<bool(const int&)> validator_,
63  const bool& current_)
64  : type(type_), validator(std::move(validator_)), current(current_) {}
65  /// The type of state that this condition is checking
66  std::type_index type;
67  /// Expression to determine if the passed state is valid
68  std::function<bool(const int&)> validator;
69  /// The current state of the condition when it was last seen by the Director
70  bool current;
71  /// The reaction handle which is monitoring this state for a valid condition
72  NUClear::threading::ReactionHandle handle;
73  };
74 
75  Provider(ProviderGroup& group_,
76  const NUClear::id_t& id_,
77  const Classification& classification_,
78  const std::type_index& type_,
79  std::shared_ptr<NUClear::threading::Reaction> reaction_)
80  : group(group_), id(id_), classification(classification_), type(type_), reaction(std::move(reaction_)) {}
81 
82  /// The provider group this provider belongs to
83  ProviderGroup& group;
84  /// The id for this provider. Will either be reaction->id or a generated one for root providers
85  NUClear::id_t id;
86  /// The classification of this Provider
87  Classification classification;
88  /// The data type this Provider is for
89  std::type_index type;
90  /// The reaction object to run in order to run this Provider
91  std::shared_ptr<NUClear::threading::Reaction> reaction;
92  /// Any when conditions that are required by this Provider
93  std::vector<std::shared_ptr<WhenCondition>> when;
94  /// A list of types and states that are caused by running this Provider
95  std::map<std::type_index, int> causing;
96  /// A list of provider types that this provider needs in order to run
97  std::set<std::type_index> needs;
98  };
