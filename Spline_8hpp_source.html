107  constexpr void exportData(std::ostream& os) const {
108  for (size_t i = 0; i < splines.size(); i++) {
109  os << std::setprecision(17) << splines[i].min << " ";
110  os << std::setprecision(17) << splines[i].max << " ";
111  os << std::setprecision(17) << splines[i].polynom.getCoefs().size() << " ";
112  for (size_t j = 0; j < splines[i].polynom.getCoefs().size(); j++) {
113  os << std::setprecision(17) << splines[i].polynom.getCoefs()[j] << " ";
114  }

119  constexpr void importData(std::istream& is) {
120  bool isFormatError = true;
121  while (is.good()) {
122  isFormatError = true;
123  Scalar min;
124  Scalar max;
125  size_t size = 0;
126  Polynom<Scalar> p;
127  // Load spline interval and degree
128  is >> min;
129  if (!is.good()) {
130  break;
131  }
132  is >> max;
133  if (!is.good()) {
134  break;
135  }
136  is >> size;
137  // Load polynom coefficients
138  p.getCoefs().resize(size);
139  for (size_t i = 0; i < size; i++) {
140  if (!is.good()) {
141  break;
142  }
143  is >> p.getCoefs()[i];
144  }
145  // Save spline part
146  isFormatError = false;
147  splines.push_back({p, min, max});
148  // Exit on line break
149  while (is.peek() == ' ') {
150  if (!is.good()) {
151  break;
152  }
153  is.ignore();
154  }
155  if (is.peek() == '\n') {
156  break;
157  }
158  }
159  if (isFormatError) {
160  throw std::logic_error("Spline import format invalid");
161  }
162  // Call possible post import
163  importCallBack();
164  }

207  private:
208  /**
209  * Return spline interpolation of given value and used given polynom evaluation function (member
210  * function pointer)
211  */
212  [[nodiscard]] constexpr Scalar interpolation(Scalar x,
213  Scalar (Polynom<Scalar>::*func)(const Scalar&) const) const {
214  // Empty case
215  if (splines.empty()) {
216  return static_cast<Scalar>(0);
217  }
218  // Bound asked abscisse into spline range
219  if (x <= splines.front().min) {
220  x = splines.front().min;
221  }
222  if (x >= splines.back().max) {
223  x = splines.back().max;
224  }
225  // Bijection spline search
226  size_t indexLow = 0;
227  size_t indexUp = splines.size() - 1;
228  while (indexLow != indexUp) {
229  size_t index = (indexUp + indexLow) / 2;
230  if (x < splines[index].min) {
231  indexUp = index - 1;
232  }
233  else if (x > splines[index].max) {
234  indexLow = index + 1;
235  }
236  else {
237  indexUp = index;
238  indexLow = index;
239  }
240  }
241  // Compute and return spline value
242  return (splines[indexUp].polynom.*func)(x - splines[indexUp].min);
243  }
244 
245  /**
246  * Return interpolation with x bound between 0 and 1
247  */
248  [[nodiscard]] constexpr Scalar interpolationMod(Scalar x,
249  Scalar (Polynom<Scalar>::*func)(const Scalar&) const) const {
250  if (x < static_cast<Scalar>(0)) {
251  x = static_cast<Scalar>(1) + (x - (static_cast<int>(x) / static_cast<Scalar>(1)));
252  }
253  else if (x > static_cast<Scalar>(1)) {
254  x = (x - ((int) x / 1));
255  }
256  return interpolation(x, func);
257  }
258  };
