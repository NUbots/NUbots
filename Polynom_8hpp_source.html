80  /**
81  * Polynom evaluation, its first, second and third derivative at given x
82  */
83  [[nodiscard]] constexpr Scalar pos(const Scalar& x) const {
84  auto xx = static_cast<Scalar>(1);
85  auto val = static_cast<Scalar>(0);
86  for (size_t i = 0; i < coefs.size(); i++) {
87  val += xx * coefs[i];
88  xx *= x;
89  }
90  return val;
91  }

93  [[nodiscard]] constexpr Scalar vel(const Scalar& x) const {
94  auto xx = static_cast<Scalar>(1);
95  auto val = static_cast<Scalar>(0);
96  for (size_t i = 1; i < coefs.size(); i++) {
97  val += i * xx * coefs[i];
98  xx *= x;
99  }
100  return val;
101  }

103  [[nodiscard]] constexpr Scalar acc(const Scalar& x) const {
104  auto xx = static_cast<Scalar>(1);
105  auto val = static_cast<Scalar>(0);
106  for (size_t i = 2; i < coefs.size(); i++) {
107  val += (i - 1) * i * xx * coefs[i];
108  xx *= x;
109  }
110  return val;
111  }
112 
113  [[nodiscard]] constexpr Scalar jerk(const Scalar& x) const {
114  auto xx = static_cast<Scalar>(1);
115  auto val = static_cast<Scalar>(0);
116  for (size_t i = 3; i < coefs.size(); i++) {
117  val += (i - 2) * (i - 1) * i * xx * coefs[i];
118  xx *= x;
119  }
120  return val;
121  }

125  */

128  coefs[i] *= coef;

132  constexpr void operator+=(const Polynom& p) {
133  while (p.coefs.size() > coefs.size()) {
134  coefs.push_back(static_cast<Scalar>(0));
135  }
136 
137  for (size_t i = 0; i < p.coefs.size(); i++) {
138  coefs[i] += p.coefs[i];
139  }
140  }

142  /**
143  * Update the polynom coefficients by applying delta offset on X abscisse
144  */
145  constexpr void shift(const Scalar& delta) {
146  Polynom<Scalar> n(coefs.size() - 1);
147  n.coefs[0] = coefs[0];
148 
149  for (size_t k = 1; k < coefs.size(); k++) {
150  const Polynom<Scalar> tmp = expandBinomial(delta, k);
151  for (size_t l = 0; l <= k; l++) {
152  n.coefs[l] += coefs[k] * tmp.coefs[l];
153  }
154  }
155 
156  *this = n;
157  }
158 
159  /**
160  * Expand the given formula (x + y)^degree and return the polynom in x whose coefficient are computed
161  * using binomial coefficient
162  */
163  [[nodiscard]] static Polynom<Scalar> expandBinomial(const Scalar& y, const size_t& degree) {
164  Combination combination{};
165  Polynom polynom{};
166 
167  polynom.getCoefs().resize(degree + 1);
168 
169  for (size_t k = 0; k <= degree; k++) {
170  polynom.getCoefs()[k] = combination.binomialCoefficient(k, degree) * std::pow(y, degree - k);
171  }
172 
173  return polynom;
174  }

176  private:

179  */
180  std::vector<Scalar> coefs{};
181  };
