44  static uint64_t xorShift() {
45  static thread_local uint64_t s[2] = {uint64_t(rand()), uint64_t(rand())};
46 
47  uint64_t s1 = s[0];
48  const uint64_t s0 = s[1];
49  s[0] = s0;
50  s1 ^= s1 << 23;
51  return (s[1] = (s1 ^ s0 ^ (s1 >> 17) ^ (s0 >> 26))) + s0;
52  }

54  template <typename Iterator, typename... Args>
55  static bool regenerateRandomModel(Model& model, const Iterator first, const Iterator last, Args... args) {
56 
57  // Get random points between first and last
58  uint range = std::distance(first, last);
59  std::array<DataPoint, Model::REQUIRED_POINTS> points;
60 
61  // Grab a unique random set
62  std::set<uint> indices;
63  while (indices.size() < Model::REQUIRED_POINTS) {
64  indices.insert(xorShift() % range);
65  }
66 
67  uint i = 0;
68  for (auto& index : indices) {
69  points[i++] = *std::next(first, index);
70  }
71  return model.regenerate(points, std::forward<Args>(args)...);
72  }

74  /**
75  * Finds an individual ransac model that fits the data
76  *
77  * @return A pair containing an iterator to the start of the remaining set, and the best fitting model
78  */
79  template <typename Iterator, typename... Args>
80  static std::pair<bool, RansacResult<Iterator, Model>> findModel(Iterator& first,
81  Iterator& last,
82  uint minimumPointsForConsensus,
83  uint maximumIterationsPerFitting,
84  double consensusErrorThreshold,
85  Args... args) {
86 
87  // Check we have enough points
88  if (std::distance(first, last) < int(minimumPointsForConsensus)) {
89  return std::make_pair(false, RansacResult<Iterator, Model>());
90  }
91 
92  uint largestConsensus = 0;
93  double bestError = std::numeric_limits<double>::max();
94  Model bestModel;
95  Model model;
96 
97  for (uint i = 0; i < maximumIterationsPerFitting; ++i) {
98 
99  uint consensusSize = 0;
100  double error = 0.0;
101 
102  // Make our model have new set of points
103  if (!regenerateRandomModel(model, first, last, std::forward<Args>(args)...)) {
104  continue;
105  }
106 
107  for (auto it = first; it != last; ++it) {
108  float this_error = model.calculateError(*it);
109  if (this_error < consensusErrorThreshold) {
110  ++consensusSize;
111  error += this_error;
112  }
113  }
114 
115  // If largest consensus
116  if (consensusSize > largestConsensus or (consensusSize == largestConsensus and error < bestError)) {
117  largestConsensus = consensusSize;
118  bestError = error;
119  bestModel = std::move(model);
120  }
121  }
122 
123  if (largestConsensus >= minimumPointsForConsensus) {
124 
125  bestModel.refineModel(first, last, consensusErrorThreshold);
126 
127  auto newFirst =
128  std::partition(first, last, [consensusErrorThreshold, bestModel](const DataPoint& point) {
129  return consensusErrorThreshold
130  > bestModel.calculateError(std::forward<const DataPoint&>(point));
131  });
132  first = newFirst;
133 
134  return std::make_pair(true, RansacResult<Iterator, Model>(bestModel, first, newFirst));
135  }
136  else {
137  return std::make_pair(false, RansacResult<Iterator, Model>());
138  }
139  }

141  template <typename Iterator, typename... Args>
142  static std::vector<RansacResult<Iterator, Model>> fitModels(Iterator first,
143  Iterator last,
144  uint minimumPointsForConsensus,
145  uint maximumIterationsPerFitting,
146  uint maximumFittedModels,
147  double consensusErrorThreshold,
148  Args... args) {
149 
150  std::vector<RansacResult<Iterator, Model>> results;
151  results.reserve(maximumFittedModels);
152 
153  while (results.size() < maximumFittedModels) {
154  bool valid;
155  RansacResult<Iterator, Model> result;
156  std::tie(valid, result) = findModel(first,
157  last,
158  minimumPointsForConsensus,
159  maximumIterationsPerFitting,
160  consensusErrorThreshold,
161  std::forward<Args>(args)...);
162 
163  // If we have more datapoints left then add this one and continue
164  if (valid) {
165  results.push_back(std::move(result));
166  }
167  else {
168  return results;
169  }
170  }
171 
172  return results;
173  }
