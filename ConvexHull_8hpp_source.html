37  enum Direction {
38  COLLINEAR = 0,
39  CLOCKWISE = 1,
40  COUNTERCLOCKWISE = 2,
41  };

43  /// @brief Finds the order of the ordered triplet (p,q,r)
44  /// Either collinear if they form a line, otherwise clockwise or counterclockwise
45  /// @param p The first ordered point in our space
46  /// @param q The second ordered point in our space
47  /// @param r The third ordered point in our space
48  /// @return zero if the points are collinear, one if clockwise, two if counterclockwise
49  Direction direction(const Eigen::Vector2d& p, const Eigen::Vector2d& q, const Eigen::Vector2d& r) {
50  // Find the cross product of vectors (p1, p0) x (p2, p0)
51  double val = (q(1) - p(1)) * (r(0) - q(0)) - (q(0) - p(0)) * (r(1) - q(1));
52 
53  // If val is zero, return zero for collinear
54  // If val is greater than zero, return one for clockwise
55  // If val is less than zero, return two for counterclockwise
56  return (val == 0) ? Direction::COLLINEAR : ((val > 0) ? Direction::CLOCKWISE : Direction::COUNTERCLOCKWISE);
57  }

59  /// @brief Sorts the points in `indices` by polar angle with the first point in `indices`
60  /// @param indices Indices corresponding to points in `points` that we want to sort
61  /// @param points All points in our space, including points not to be used in sorting
62  void sort_polar(std::vector<int>& indices, const Eigen::Matrix<double, 2, Eigen::Dynamic>& points) {
63  std::sort(indices.begin() + 1, indices.end(), [&](int i, int j) {
64  double dx1 = points(0, i) - points(0, indices[0]);
65  double dy1 = points(1, i) - points(1, indices[0]);
66  double dx2 = points(0, j) - points(0, indices[0]);
67  double dy2 = points(1, j) - points(1, indices[0]);
68 
69  double det = dx1 * dy2 - dy1 * dx2;
70 
71  if (det == 0) {
72  // points p1 and p2 are on the same line from the pivot, check which one is closer
73  double d1 = dx1 * dx1 + dy1 * dy1;
74  double d2 = dx2 * dx2 + dy2 * dy2;
75  return d1 > d2; // keep the point that is farther
76  }
77 
78  return det > 0;
79  });
80  }

82  /// @brief Finds if p0 has a lower y value than p1 - break the tie with lower x if equal
83  /// @param p0 A point, if this point is lower than p1 then return true
84  /// @param p1 The point to compare p0 to
85  /// @return True if p0 has a lower y value than p1, or if equal then if p0 has a lower x value
86  bool lower_point(Eigen::Vector2d p0, Eigen::Vector2d p1) {
87  // Find the boolean comparisons to then use in the final return
88  bool y_less = p0.y() < p1.y();
89  bool y_equal = p0.y() == p1.y();
90  bool x_less = p0.x() < p1.x();
91  // Return if either p0 has lower y, or if equal p0 has lower x
92  return y_less || (y_equal && x_less);
93  }

95  /// @brief Creates a convex hull from the points in `indices` using the Graham Scan algorithm for convex hulls
96  /// @param indices Indices corresponding to points in `points` that we want to use to create a convex hull
97  /// @param points All points in our space, including points not to be used in the convex hull algorithm
98  /// @return Indicies corresponding to points in `points` that make up the convex hull for this set of points
99  std::vector<int> graham_scan(const std::vector<int>& indices,
100  const Eigen::Matrix<double, 2, Eigen::Dynamic>& points) {
101  if (indices.size() < 3) {
102  return indices;
103  }
104 
105  // Copy `indices` so we can modify it
106  std::vector<int> indices_copy = indices;
107 
108  // Find the point with the lowest y-coordinate
109  int p0 = 0;
110  // Start at 1, don't need to compare p0 to itself
111  for (size_t i = 1; i < indices_copy.size(); i++) {
112  // If point i is lower than p0, then i becomes our new p0
113  if (lower_point(points.col(indices_copy[i]), points.col(indices_copy[p0]))) {
114  p0 = i;
115  }
116  }
117 
118  // Swap p0 with the first point
119  std::swap(indices_copy[0], indices_copy[p0]);
120 
121  // Sort the points by polar angle with the first point in indices_copy
122  sort_polar(indices_copy, points);
123 
124  // Initialize the stack
125  std::vector<int> stack;
126  stack.push_back(indices_copy[0]);
127  stack.push_back(indices_copy[1]);
128  stack.push_back(indices_copy[2]);
129 
130  // Process the remaining points
131  for (size_t i = 3; i < indices_copy.size(); i++) {
132  // Keep removing top while the angle formed by points next-to-top, top, and points[i] makes a clockwise turn
133  while (
134  stack.size() > 1
135  && direction(points.col(stack[stack.size() - 2]), points.col(stack.back()), points.col(indices_copy[i]))
136  != Direction::COUNTERCLOCKWISE) {
137  stack.pop_back();
138  }
139  stack.push_back(indices_copy[i]);
140  }
141 
142  // The points in the stack are the vertices of the convex hull
143  return stack;
144  }

146  /// @brief Partition the indices into `m` subsets of equal size
147  /// @param indices Indices of the points to be partitioned, corresponding to indices in a matrix of points
148  /// @param m The number of subsets to partition the indices into
149  /// @return All the subsets, stored as a vector containing each subset
150  std::vector<std::vector<int>> partition_points(const std::vector<int>& indices, int m) {
151  std::vector<std::vector<int>> subsets(m);
152  int n = indices.size();
153  int subset_size = n / m;
154  int remainder = n % m;
155 
156  for (int i = 0; i < m; i++) {
157  int start = i * subset_size + std::min(i, remainder);
158  int end = start + subset_size + (i < remainder ? 1 : 0);
159  subsets[i] = std::vector<int>(indices.begin() + start, indices.begin() + end);
160  }
161 
162  return subsets;
163  }

165  /// @brief From all the convex hulls, find the point that is most to the left or right, based on the comparison op
166  /// @param hulls A vector of sub-hulls, each sub-hull is a vector of indices corresponding to points in `points`
167  /// @param points All the points in our space, including points not to be used in the convex hull algorithm
168  /// @param comp The comparison operator to use, either std::less or std::greater
169  /// @return The index of the extreme point from all of the convex hulls
170  int find_extreme_point(const std::vector<std::vector<int>>& hulls,
171  const Eigen::Matrix<double, 2, Eigen::Dynamic>& points,
172  const std::function<bool(double, double)>& comp) {
173  // Initialise with the first point to start with
174  int extreme = hulls[0][0];
175 
176  // Loop through all points, updating the leftmost if a new leftmost is found
177  for (const auto& hull : hulls) {
178  for (int index : hull) {
179  if (comp(points(0, index), points(0, extreme))) {
180  extreme = index;
181  }
182  }
183  }
184 
185  return extreme;
186  }

188  /// @brief Finds the upper tangent of the convex hulls
189  /// Works by starting at the rightmost point and going counter-clockwise until the leftmost point is reached
190  /// @param hulls A vector of sub-hulls, each sub-hull is a vector of indices corresponding to points in `points`
191  /// @param points All the points in our space, including points not to be used in the convex hull algorithm
192  /// @param leftmost The index of the leftmost point in all the hulls
193  /// @param rightmost The index of the rightmost point in all the hulls
194  /// @return The indices of the points that make up the upper tangent
195  std::vector<int> find_upper_tangent(const std::vector<std::vector<int>>& hulls,
196  const Eigen::Matrix<double, 2, Eigen::Dynamic>& points,
197  int leftmost,
198  int rightmost) {
199  std::vector<int> upper_tangent{};
200 
201  // Start from the rightmost point and go counter-clockwise
202  int current_point = rightmost;
203 
204  do {
205  upper_tangent.push_back(current_point);
206 
207  // Find the next point that makes a counter-clockwise turn
208  int next_point = hulls[0][0];
209  for (const auto& hull : hulls) {
210  for (int index : hull) {
211  double orientation =
212  (points(1, index) - points(1, current_point)) * (points(0, next_point) - points(0, index))
213  - (points(0, index) - points(0, current_point)) * (points(1, next_point) - points(1, index));
214 
215  if (orientation < 0
216  || (orientation == 0
217  && (points.col(index) - points.col(current_point)).squaredNorm()
218  > (points.col(next_point) - points.col(current_point)).squaredNorm())) {
219  next_point = index;
220  }
221  }
222  }
223 
224  current_point = next_point;
225  } while (current_point != leftmost);
226 
227  return upper_tangent;
228  }

230  /// @brief Finds the lower tangent of the convex hulls
231  /// Works by starting at the leftmost point and going counter-clockwise until the rightmost point is reached
232  /// @param hulls A vector of sub-hulls, each sub-hull is a vector of indices corresponding to points in `points`
233  /// @param points All the points in our space, including points not to be used in the convex hull algorithm
234  /// @param leftmost The index of the leftmost point in all the hulls
235  /// @param rightmost The index of the rightmost point in all the hulls
236  /// @return The indices of the points that make up the upper tangent
237  std::vector<int> find_lower_tangent(const std::vector<std::vector<int>>& hulls,
238  const Eigen::Matrix<double, 2, Eigen::Dynamic>& points,
239  int leftmost,
240  int rightmost) {
241  std::vector<int> lower_tangent{};
242 
243  // Start from the leftmost point and go clockwise
244  int current_point = leftmost;
245 
246  do {
247  lower_tangent.push_back(current_point);
248 
249  // Find the next point that makes a clockwise turn
250  int next_point = hulls[0][0];
251  for (const auto& hull : hulls) {
252  for (int index : hull) {
253  double orientation =
254  (points(1, index) - points(1, current_point)) * (points(0, next_point) - points(0, index))
255  - (points(0, index) - points(0, current_point)) * (points(1, next_point) - points(1, index));
256 
257  if (orientation < 0
258  || (orientation == 0
259  && (points.col(index) - points.col(current_point)).squaredNorm()
260  > (points.col(next_point) - points.col(current_point)).squaredNorm())) {
261  next_point = index;
262  }
263  }
264  }
265 
266  current_point = next_point;
267  } while (current_point != rightmost);
268 
269  return lower_tangent;
270  }

272  /// @brief Merges the convex hulls together to form a single convex hull
273  /// @param hulls The convex hulls to be merged together
274  /// @param points All the points in our space, including points not to be used in the convex hull algorithm
275  /// @return The indices corresponding to points in `points` that make up the convex hull
276  std::vector<int> merge_hulls(const std::vector<std::vector<int>>& hulls,
277  const Eigen::Matrix<double, 2, Eigen::Dynamic>& points) {
278  // Vector to hold the convex hull from merging the sub-hulls
279  std::vector<int> merged_hull{};
280 
281  // Find the leftmost and rightmost points of all the hulls
282  int leftmost = find_extreme_point(hulls, points, std::less<double>());
283  int rightmost = find_extreme_point(hulls, points, std::greater<double>());
284 
285  // Add the leftmost and rightmost points to the merged hull
286  merged_hull.push_back(leftmost);
287 
288  // Find the upper and lower tangents of the hulls and add the points on the tangents to the merged hull
289  std::vector<int> upper_tangent = find_upper_tangent(hulls, points, leftmost, rightmost);
290  std::vector<int> lower_tangent = find_lower_tangent(hulls, points, leftmost, rightmost);
291 
292  // Add the points on the upper tangent to the merged hull in counter-clockwise order
293  for (auto it = upper_tangent.rbegin(); it != upper_tangent.rend(); ++it) {
294  merged_hull.push_back(*it);
295  }
296 
297  // Add the points on the lower tangent to the merged hull in counter-clockwise order
298  for (auto it = lower_tangent.rbegin(); it != lower_tangent.rend(); ++it) {
299  merged_hull.push_back(*it);
300  }
301 
302  return merged_hull;
303  }

305  /// @brief Algorithm to find a convex hull from a set of points using Chan's algorithm
306  /// Chan's algorithm finds the convex hull of subsets of points and then merges those convex hulls together
307  /// This is a type of 'divide-and-conquer' algorithm
308  /// @param indices The indices corresponding to points in `points` that we want to use in the algorithm
309  /// @param points All the points in our space, including points not to be used in the convex hull algorithm
310  /// @return The indices corresponding to points in `points` that make up the convex hull
311  inline std::vector<int> chans_convex_hull(const std::vector<int>& indices,
312  const Eigen::Matrix<double, 2, Eigen::Dynamic>& points) {
313  // Iterator variable to increase the number of subsets in each loop
314  // Start it higher than 0 since we have a lot of points
315  int t = 0;
316  // The number of subsets to divide the points into, aims to find the smallest `m` such that the convex hull
317  // algorithm returns a convex hull with no more than `m` points
318  size_t m = 0;
319  // Stores the best convex hull from these points, and is returned when the solution is found
320  std::vector<int> final_hull{};
321 
322  // This must run at least once, to find a convex hull
323  do {
324  // Find the number of subsets
325  t++;
326  m = std::pow(2, t);
327  // Create the subsets
328  std::vector<std::vector<int>> subsets = partition_points(indices, m);
329  // Find the convex hulls of each subset using graham scan
330  std::vector<std::vector<int>> hulls{};
331  for (auto& subset : subsets) {
332  // We can't use this if a subset is too small
333  if (subset.size() < 3) {
334  return final_hull;
335  }
336  hulls.push_back(graham_scan(subset, points));
337  }
338 
339  // Merge the convex hulls together
340  final_hull = merge_hulls(hulls, points);
341  } while (final_hull.size() > m); // keep trying until the smallest convex hull is found
342 
343  // Return the best hull
344  return final_hull;
345  }

347  // Overloaded so we can directly call with our 3d matrix from the visual mesh
348  inline std::vector<int> chans_convex_hull(const std::vector<int>& indices,
349  const Eigen::Matrix<double, 3, Eigen::Dynamic>& points) {
350  // Make the matrix of x-y points from the 3D points
351  Eigen::Matrix<double, 2, Eigen::Dynamic> points_2d(2, points.cols());
352  for (int i = 0; i < points.cols(); i++) {
353  points_2d.col(i) << points(0, i), points(1, i);
354  }
355 
356  return chans_convex_hull(indices, points_2d);
357  }

360  /// @brief Calculates the angle between three points
361  /// @param p1
362  /// @param p2
363  /// @param p3
364  /// @return
365  double calculate_angle(const Eigen::Vector3d& p1, const Eigen::Vector3d& p2, const Eigen::Vector3d& p3) {
366  Eigen::Vector3d v1 = p2 - p1;
367  Eigen::Vector3d v2 = p3 - p1;
368  return std::atan2(v1.x() * v2.y() - v1.y() * v2.x(), v1.x() * v2.x() + v1.y() * v2.y());
369  }

371  /// @brief Determine if point is in the convex hull
372  /// @param hull_indices The indices of the points in the convex hull, corresponding to `points`, ordered
373  /// @param point The point to check if it is in the convex hull
374  /// @return True if the point is in the convex hull, false otherwise
375  bool point_in_convex_hull(const std::vector<Eigen::Vector3d>& hull, const Eigen::Vector3d& point) {
376  double winding_number = 0.0;
377 
378  // Loop over each pair of consecutive points in the hull to find the total angle across the hull with the point
379  // If the point is outside, the angles should cancel out as it goes around the hull
380  // If the point is inside, the angles should add up to +-2pi, ie a full circle
381  for (size_t i = 0; i < hull.size(); ++i) {
382  // Get the current point and the next point in the hull
383  Eigen::Vector3d p1 = hull[i];
384  Eigen::Vector3d p2 = hull[(i + 1) % hull.size()];
385 
386  // Calculate the angle between the point and the current pair of hull points and add to the total
387  double angle = calculate_angle(point, p1, p2);
388  winding_number += angle;
389  }
390 
391  // Approximately zero as the angles cancel out
392  if (std::abs(winding_number) < 1e-6) {
393  return false;
394  }
395  else { // Approximately +-2pi as the angles add up to a circle
396  return true;
397  }
398  }

400  /// @brief Determine if point is in the convex hull. Converts to 3D and calls the 3D version.
401  /// @param hull_indices The indices of the points in the convex hull, corresponding to `points`, ordered
402  /// @param point The point to check if it is in the convex hull
403  /// @return True if the point is in the convex hull, false otherwise
404  bool point_in_convex_hull(const std::vector<Eigen::Vector2d>& hull, const Eigen::Vector2d& point) {
405  std::vector<Eigen::Vector3d> hull_3d;
406  for (const auto& p : hull) {
407  hull_3d.emplace_back(p.x(), p.y(), 0.0);
408  }
409  return point_in_convex_hull(hull_3d, Eigen::Vector3d(point.x(), point.y(), 0.0));
410  }

412  /// @brief Sort the indices by theta
413  /// @tparam Iterator Iterator type for the indices
414  /// @param first The first index to sort
415  /// @param last The last index to sort
416  /// @param rays The rays that the indices correspond to
417  /// @param world_offset The offset to apply to the theta values
418  template <typename Iterator>
419  void sort_by_theta(Iterator first,
420  Iterator last,
421  const Eigen::Matrix<double, 3, Eigen::Dynamic>& rays,
422  const double& world_offset) {
423  std::sort(first, last, [&](const int& a, const int& b) {
424  const Eigen::Vector3d& p0(rays.col(a));
425  const Eigen::Vector3d& p1(rays.col(b));
426 
427  double theta0 = std::fmod(std::atan2(p0.y(), p0.x()) + M_PI - world_offset, 2.0 * M_PI);
428  double theta1 = std::fmod(std::atan2(p1.y(), p1.x()) + M_PI - world_offset, 2.0 * M_PI);
429  return theta0 < theta1;
430  });
431  }
