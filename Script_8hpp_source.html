85  /// @brief Assumes the hostname is of the form <platform_name><robot_number> and returns <platform_name>.
86  /// @param hostname The hostname of the device the program is running on.
87  /// @return The platform name.
88  /// @throws System error when the platform cannot be extracted from the hostname.
89  static inline std::string get_platform(const std::string& hostname) {
90  // It is assumed that all hostnames are in the format <platform name><robot number>,
91  // such that the regular expression
92  // [a-z]+[0-9]+?
93  // will match all hostnames
94  std::regex re("([a-z]+)([0-9]+)?");
95  std::smatch match;
96 
97  if (std::regex_match(hostname, match, re)) {
98  // match[0] will be the full string
99  // match[1] the first match (platform name)
100  // match[2] the second match (robot number)
101  return match[1].str();
102  }
103 
104  throw std::system_error(-1,
105  std::system_category(),
106  ("Failed to extract platform name from '" + hostname + "'."));
107  }

109  /// @brief If a robot-specific script exists, load the YAML file as a script. If not, loads the
110  /// platform-specific script. If neither exist, throws a runtime error.
111  /// @param script The name of the script to run.
112  /// @return The script contents in the form off a YAML node
113  /// @throws Runtime error when the script doesn't exist.
114  static YAML::Node load(const std::string& script) {
115  // Set paths to the script files.
116  auto hostname = utility::support::get_hostname();
117  auto robot_path = "scripts/" + hostname + "/" + script;
118  auto platform_path = "scripts/" + get_platform(hostname) + "/" + script;
119  auto robot_name = utility::platform::get_robot_alias(hostname);
120  auto name_path = "scripts/" + robot_name + "/" + script;
121 
122  // If robot name script exists, use it
123  if (!robot_name.empty() && utility::file::exists(name_path)) {
124  NUClear::log<NUClear::LogLevel::INFO>("Parsing name specific script:", script);
125  return YAML::LoadFile(name_path);
126  }
127  // If robot name script doesn't exist, try to use the robot-specific script
128  else if (utility::file::exists(robot_path)) {
129  NUClear::log<NUClear::LogLevel::INFO>("Parsing robot specific script:", script);
130  return YAML::LoadFile(robot_path);
131  }
132  // If there was no robot-specific script, then get the platform-specific script
133  else if (utility::file::exists(platform_path)) {
134  NUClear::log<NUClear::LogLevel::INFO>("Parsing default platform script:", script);
135  return YAML::LoadFile(platform_path);
136  }
137  // The script doesn't exist, tell the user
138  else {
139  throw std::runtime_error("Script file '" + script + "' does not exist.");
140  }
141  }

143  /// @brief Creates sequences of servos from the script requests given
144  /// @param scripts Sequence of scripts to run
145  /// @param start When the first script should start executing. Default is now.
146  /// @return Sequence message which will be filled with sequences of servos and can then be emitted as a
147  /// Task.
148  template <typename Sequence>
149  static std::unique_ptr<Sequence> load_script(const std::vector<ScriptRequest> scripts,
150  NUClear::clock::time_point start = NUClear::clock::now()) {
151  // Create sequence message
152  auto msg = std::make_unique<Sequence>();
153 
154  // First script begins at start time
155  auto time = start;
156 
157  // Loop through all given scripts and add to the Sequence message
158  for (const auto& script : scripts) {
159  // Load the script to a vector of Frame structs
160  auto frames = load(script.name).as<std::vector<Frame>>();
161  // Time will be incremented through each frame
162  // Loop over the frames and add them as sequences of servos into the Sequence message
163  for (const auto& frame : frames) {
164  // This frame should finish after frame.duration time has passed
165  // A duraction modifier can be used to change the speed of the script
166  time += std::chrono::duration_cast<NUClear::clock::time_point::duration>(frame.duration
167  * script.duration_modifier);
168 
169  // Add the servos in the frame to a map
170  std::map<uint32_t, ServoCommand> servos{};
171  for (const auto& target : frame.targets) {
172  servos[target.id] = ServoCommand(time,
173  target.position,
174  ServoState(target.gain, target.torque),
175  ServoCommand::DoneType::TIME);
176  }
177 
178  // Add the map to the pack. This represents one sequence of servos.
179  msg->frames.emplace_back(servos);
180  }
181  }
182 
183  // Return the message which can then be emitted as a Task
184  return msg;
185  }

187  /// @brief Emits a script function with one script request which is then sent through to the multi script version
188  /// above
189  /// @param script Script to run
190  /// @param start When the script should start executing. Default is now.
191  /// @return Sequence message which will be filled with sequences of servos.
192  template <typename Sequence>
193  static std::unique_ptr<Sequence> load_script(const ScriptRequest script,
194  NUClear::clock::time_point start = NUClear::clock::now()) {

198  /// @brief Accepts a vector of strings and converts them into a ScriptRequest, and uses the ScriptRequest
199  /// function above
200  /// @param scripts Script names to run
201  /// @param start When the script should start executing. Default is now.
202  /// @return Sequence message which will be filled with sequences of servos.
203  template <typename Sequence>
204  static std::unique_ptr<Sequence> load_script(const std::vector<std::string> scripts,
205  NUClear::clock::time_point start = NUClear::clock::now()) {
206  // Convert the vector of strings into a vector of ScriptRequests
207  std::vector<ScriptRequest> script_requests;
208  for (const auto& script : scripts) {
209  script_requests.emplace_back(script);
210  }
211 
212  // Call the function above
213  return load_script<Sequence>(script_requests, start);
214  }

283  /// @brief Decodes a YAML Node as a Target

285  /// @param rhs A Target where the YAML Node information will be placed
286  /// @return True if decoding was successful, false otherwise
287  static inline bool decode(const Node& node, ::utility::skill::Frame::Target& rhs) {
288  // Try to read and save the target information
289  try {
290  rhs = ::utility::skill::Frame::Target{node["id"].as<std::string>(),
291  node["position"].as<float>(),

294  }

297  "Line:",

299  "Column:",

301  "Pos:",
302  e.mark.pos,
303  "Message:",
304  e.msg);
305  return false;
306  }
307 
308  return true;
309  }
