66  /// @brief Smart enum for the robot's position
67  struct Position {
68  enum Value { ALL_ROUNDER, STRIKER, GOALIE, DEFENDER, DYNAMIC };
69  Value value = Value::ALL_ROUNDER;
70 
71  Position() = default;
72  Position(std::string const& str) {
73  // clang-format off
74  if (str == "ALL_ROUNDER") { value = Value::ALL_ROUNDER; }
75  else if (str == "STRIKER") { value = Value::STRIKER; }
76  else if (str == "GOALIE") { value = Value::GOALIE; }
77  else if (str == "DEFENDER") { value = Value::DEFENDER; }
78  else if (str == "DYNAMIC") { value = Value::DYNAMIC; }
79  else { throw std::runtime_error("Invalid robot position"); }
80  // clang-format on
81  }
82 
83  /// @brief Convert the enum to a string
84  operator std::string() const {
85  switch (value) {
86  case Value::ALL_ROUNDER: return "ALL_ROUNDER";
87  case Value::STRIKER: return "STRIKER";
88  case Value::DEFENDER: return "DEFENDER";
89  case Value::GOALIE: return "GOALIE";
90  case Value::DYNAMIC: return "DYNAMIC";
91  default: throw std::runtime_error("enum Position's value is corrupt, unknown value stored");
92  }
93  }
94 
95  /// @brief Get the full yaml name of the config for the current position
96  /// @return The full yaml name of the config for the current position
97  std::string get_config_name() {
98  switch (value) {
99  case Value::ALL_ROUNDER: return "AllRounder.yaml";
100  case Value::STRIKER: return "Striker.yaml";
101  case Value::DEFENDER: return "Defender.yaml";
102  case Value::GOALIE: return "Goalie.yaml";
103  case Value::DYNAMIC: return "Dynamic.yaml";
104  default: throw std::runtime_error("enum Position's value is corrupt, unknown value stored");
105  }
106  }
107 
108  /// @brief Increment the enum, for toggle
109  void operator++() {
110  switch (value) {
111  case Value::ALL_ROUNDER: value = Value::STRIKER; break;
112  case Value::STRIKER: value = Value::GOALIE; break;
113  case Value::GOALIE: value = Value::DEFENDER; break;
114  case Value::DEFENDER: value = Value::DYNAMIC; break;
115  case Value::DYNAMIC: value = Value::ALL_ROUNDER; break;
116  default: value = Value::STRIKER;
117  }
118  }
